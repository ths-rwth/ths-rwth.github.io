55 inline Interval<Numeric> IntervalEvaluation::evaluate(const Monomial& m, const std::map<Variable, Interval<Numeric>>& map)
56 {
57  Interval<Numeric> result(1);
58  // TODO use iterator.
59  CARL_LOG_TRACE("carl.core.monomial", "Iterating over " << m);
60  for(unsigned i = 0; i < m.nrVariables(); ++i)
61  {
62  CARL_LOG_TRACE("carl.core.monomial", "Iterating: " << m[i].first);
63  // We expect every variable to be in the map.
64  CARL_LOG_ASSERT("carl.interval", map.count(m[i].first) > (size_t)0, "Every variable is expected to be in the map.");
65  result *= carl::pow(map.at(m[i].first), m[i].second);
66  if( result.isZero() )
67  return result;
68  }

94  if(isZero(p)) {

107 template<typename Numeric, typename Coeff, EnableIf<std::is_same<Numeric, Coeff>>>
108 inline Interval<Numeric> IntervalEvaluation::evaluate(const UnivariatePolynomial<Coeff>& p, const std::map<Variable, Interval<Numeric>>& map) {
109  CARL_LOG_FUNC("carl.core.monomial", p << ", " << map);
110  assert(map.count(p.mainVar()) > 0);
111  Interval<Numeric> res = Interval<Numeric>::emptyInterval();
112  const Interval<Numeric> varValue = map.at(p.mainVar());
113  Interval<Numeric> exp(1);
114  for (unsigned i = 0; i < p.degree(); i++) {
115  res += p.coefficients()[i] * exp;
116  if( res.isInfinite() )
117  return res;
118  exp = varValue.pow(i+1);
119  }
120  return res;
121 }

129  Interval<Numeric> exp(1);

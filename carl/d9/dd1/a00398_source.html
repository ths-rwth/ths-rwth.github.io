655  Logic logic() const {
656  if (carl::PROP_CONTAINS_BITVECTOR <= properties()
657  && !(carl::PROP_CONTAINS_UNINTERPRETED_EQUATIONS <= properties())
658  && !(carl::PROP_CONTAINS_INTEGER_VALUED_VARS <= properties())
659  && !(carl::PROP_CONTAINS_REAL_VALUED_VARS <= properties())
660  && !(carl::PROP_CONTAINS_PSEUDOBOOLEAN <= properties())) {
661  return Logic::QF_BV;
662  } else if (!(carl::PROP_CONTAINS_BITVECTOR <= properties())
663  && (carl::PROP_CONTAINS_UNINTERPRETED_EQUATIONS <= properties())
664  && !(carl::PROP_CONTAINS_INTEGER_VALUED_VARS <= properties())
665  && !(carl::PROP_CONTAINS_REAL_VALUED_VARS <= properties())
666  && !(carl::PROP_CONTAINS_PSEUDOBOOLEAN <= properties())) {
667  return Logic::QF_UF;
668  } else if (!(carl::PROP_CONTAINS_BITVECTOR <= properties())
669  && !(carl::PROP_CONTAINS_UNINTERPRETED_EQUATIONS <= properties())
670  && !(carl::PROP_CONTAINS_INTEGER_VALUED_VARS <= properties())
671  && !(carl::PROP_CONTAINS_REAL_VALUED_VARS <= properties())
672  && (carl::PROP_CONTAINS_PSEUDOBOOLEAN <= properties())) {
673  return Logic::QF_PB;
674  } else if (!(carl::PROP_CONTAINS_BITVECTOR <= properties())
675  && !(carl::PROP_CONTAINS_UNINTERPRETED_EQUATIONS <= properties())
676  && (carl::PROP_CONTAINS_INTEGER_VALUED_VARS <= properties())
677  && !(carl::PROP_CONTAINS_REAL_VALUED_VARS <= properties())
678  && !(carl::PROP_CONTAINS_PSEUDOBOOLEAN <= properties())) {
679  if (carl::PROP_CONTAINS_NONLINEAR_POLYNOMIAL <= properties()) {
680  return Logic::QF_NIA;
681  } else {
682  return Logic::QF_LIA;
683  }
684  } else if (!(carl::PROP_CONTAINS_BITVECTOR <= properties())

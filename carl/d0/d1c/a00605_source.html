21  assert(newtonSums.size() > 0);
22  std::size_t p = newtonSums.size() - 1;
23  std::vector<Coeff> res(p + 1);
24  res[p] = Coeff(1);
25  for(std::size_t i = 1; i <= p; i++) {
26  Coeff sum(0);
27  for (std::size_t j = 1; j <= i; j++) {
28  sum += res[p - i + j] * newtonSums[j];
29  }
30  sum *= -(Coeff(1) / Coeff(i));
31  res[p-i] = sum;
32  }
33  return res;
34 }

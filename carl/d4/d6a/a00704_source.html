51 /**
52  * Replace all variables except one of the multivariate polynomial 'p' by
53  * numbers as given in the mapping 'm', which creates a univariate polynomial,
54  * and return all roots of that created polynomial.
55  * Note that 'p' is represented as a univariate polynomial with polynomial coefficients.
56  * Its main variable is not replaced and stays the main variable of the created polynomial.
57  * However, all variables in the polynomial coefficients are replaced, which is why
58  * <ul>
59  * <li>the main variable of 'p' must not be in 'm'</li>
60  * <li>all variables from the coefficients of 'p' must be in 'm'</li>
61  * </ul>
62  * The roots are sorted in ascending order.
63  * Returns a real_roots_result indicating whether the roots could be isolated or the polynomial
64  * was not univariate or is nullified.
65  */
66 template<typename Coeff, typename Number>
67 real_roots_result<real_algebraic_number_interval<Number>> real_roots(
68  const UnivariatePolynomial<Coeff>& poly,
69  const ran::ran_assignment_t<real_algebraic_number_interval<Number>>& varToRANMap,
70  const Interval<Number>& interval = Interval<Number>::unboundedInterval()
71 ) {
72  CARL_LOG_FUNC("carl.ran.realroots", poly << " in " << poly.mainVar() << ", " << varToRANMap << ", " << interval);
73  assert(varToRANMap.count(poly.mainVar()) == 0);
74 
75  if (carl::isZero(poly)) {
76  CARL_LOG_TRACE("carl.ran.realroots", "poly is 0 -> nullified");
77  return real_roots_result<real_algebraic_number_interval<Number>>::nullified_response();
78  }
79  if (poly.isNumber()) {
80  CARL_LOG_TRACE("carl.ran.realroots", "poly is constant but not zero -> no root");
81  return real_roots_result<real_algebraic_number_interval<Number>>::no_roots_response();
82  }
83 
84  // We want to simplify 'poly', but it's const, so make a copy.
85  UnivariatePolynomial<Coeff> polyCopy(poly);
86  ran::ran_assignment_t<real_algebraic_number_interval<Number>> ir_map;
87 
88  for (Variable v: carl::variables(polyCopy)) {
89  if (v == poly.mainVar()) continue;
90  if (varToRANMap.count(v) == 0) {
91  CARL_LOG_TRACE("carl.ran.realroots", "poly still contains unassigned variable " << v << " -> non-univariate");
92  return real_roots_result<real_algebraic_number_interval<Number>>::non_univariate_response();
93  }
94  assert(varToRANMap.count(v) > 0);
95  if (varToRANMap.at(v).is_numeric()) {
96  substitute_inplace(polyCopy, v, Coeff(varToRANMap.at(v).value()));
97  } else {
98  ir_map.emplace(v, varToRANMap.at(v));
99  }
100  }
101  if (carl::isZero(polyCopy)) {
102  CARL_LOG_TRACE("carl.ran.realroots", "poly is 0 after substituting rational assignments -> nullified");
103  return real_roots_result<real_algebraic_number_interval<Number>>::nullified_response();
104  }
105  if (ir_map.empty()) {
106  assert(polyCopy.isUnivariate());
107  CARL_LOG_TRACE("carl.ran.realroots", "poly " << polyCopy << " is univariate after substituting rational assignments");
108  return real_roots(polyCopy, interval);
109  } else {
110  CARL_LOG_TRACE("carl.ran.realroots", polyCopy << " in " << polyCopy.mainVar() << ", " << varToRANMap << ", " << interval);
111  assert(ir_map.find(polyCopy.mainVar()) == ir_map.end());
112 
113  // substitute RANs with low degrees first
114  ran::ordered_ran_assignment_t<real_algebraic_number_interval<Number>> ord_ass;
115  for (const auto& ass : ir_map) ord_ass.emplace_back(ass);
116  std::sort(ord_ass.begin(), ord_ass.end(), [](const auto& a, const auto& b){
117  return a.second.polynomial().degree() > b.second.polynomial().degree();
118  });
119 
120  std::optional<UnivariatePolynomial<Number>> evaledpoly = substitute_rans_into_polynomial(polyCopy, ord_ass);
121  if (!evaledpoly) {
122  CARL_LOG_TRACE("carl.ran.realroots", "poly still contains unassigned variable -> non-univariate");
123  return real_roots_result<real_algebraic_number_interval<Number>>::non_univariate_response();
124  }
125  if (carl::isZero(*evaledpoly)) {
126  CARL_LOG_TRACE("carl.ran.realroots", "got zero polynomial -> nullified");
127  return real_roots_result<real_algebraic_number_interval<Number>>::nullified_response();
128  }
129 
130  CARL_LOG_TRACE("carl.ran.realroots", "Calling on " << *evaledpoly);
131  Constraint<MultivariatePolynomial<Number>> cons(MultivariatePolynomial<Number>(polyCopy), Relation::EQ);
132  std::vector<real_algebraic_number_interval<Number>> roots;
133  auto res = real_roots(*evaledpoly, interval);
134  for (const auto& r: res.roots()) { // TODO can be made more efficient!
135  CARL_LOG_TRACE("carl.ran.realroots", "Checking " << polyCopy.mainVar() << " = " << r);
136  ir_map[polyCopy.mainVar()] = r;
137  CARL_LOG_TRACE("carl.ran.realroots", "Evaluating " << cons << " on " << ir_map);
138  if (evaluate(cons, ir_map)) {
139  roots.emplace_back(r);
140  } else {
141  CARL_LOG_TRACE("carl.ran.realroots", "Purging spurious root " << r);
142  }
143  }
144  return real_roots_result<real_algebraic_number_interval<Number>>::roots_response(std::move(roots));
145  }

18 /**
19  * @file
20  * Represent a dynamic root, also known as a "root expression".
21  * It's a generalization of an algebraic real(= univariate poly having roots + root index).
22  * It uses a multivariate poly with distinguished, root-variable "_z"
23  * and an root index. If we want to "evaluate" this root, we need to
24  * plug in a subpoint with algebraic-real values for all variables that are not "_z".
25  * The result is basically an algebraic real.
26  * And since this algebraic real depends on the subpoint,
27  * you can also think of this as a multidimensional scalar function:
28  * A root(p(x1,x2,_z), 1) value represents a function f(x1,x2) of algReal^2 to
29  * an algebraic real. For example, f(a,b)= root(q(_z),1), where the resulting
30  * algebraic real is the 1. root of poly q(_z) := p(a,b,_z) after pluggin a and b into poly p.
31  */
32 template<typename Poly>
33 class MultivariateRoot {
34 public:
35  using Number = typename UnderlyingNumberType<Poly>::type;
36  using RAN = RealAlgebraicNumber<Number>;
37  using EvalMap = ran::RANMap<Number>;
38 private:
39  /**
40  * Distinguished, globally unique root-variable
41  */
42  static const Variable sVar;
43  /// Polynomial defining this root.
44  Poly mPoly;
45  /// Specifies which root to consider.
46  std::size_t mK;
47 public:
48  /**
49  * @param poly Must mention the root-variable "_z" and
50  * should have a at least 'rootIdx'-many roots in "_z" at each subpoint
51  * where it is intended to be evaluated.
52  * @param k The index of the root of the polynomial in "_z".
53  * The first root has index 1, the second has index 2 and so on.
54  */
55  MultivariateRoot(const Poly& poly, std::size_t k): mPoly(poly), mK(k) {
56  assert(mK > 0);
57  }

101  /**
102  * Return the emerging algebraic real after pluggin in a subpoint to replace
103  * all variables with algebraic reals that are not the root-variable "_z".
104  * @param m must contain algebraic real assignments for all variables that are not "_z".
105  * @return std::nullopt if the underlying polynomial has no root with index 'rootIdx' at
106  * the given subpoint.
107  */
108  std::optional<RAN> evaluate(const EvalMap& m) const {
109  CARL_LOG_DEBUG("carl.rootexpression", "Evaluate: " << *this << " against: " << m);
110  auto poly = carl::to_univariate_polynomial(mPoly, sVar);
111  auto result = carl::real_roots(poly, m);
112  if (!result.is_univariate()) {
113  CARL_LOG_TRACE("carl.rootexpression", poly << " is not univariate (nullified: " << result.is_nullified() << "; non-univariate: " << result.is_non_univariate() << ").");
114  return std::nullopt;
115  }
116  CARL_LOG_DEBUG("carl.rootexpression", "Roots: " << result.roots());
117  if (result.roots().size() < mK) {
118  CARL_LOG_TRACE("carl.rootexpression", mK << "th root does not exist.");
119  return std::nullopt;
120  }
121  CARL_LOG_TRACE("carl.rootexpression", "Take " << mK << "th of isolated roots " << result.roots());
122  assert(result.roots().size() >= mK);
123  assert(mK > 0);
124  CARL_LOG_DEBUG("carl.rootexpression", "Result is " << result.roots()[mK-1]);
125  return result.roots()[mK-1];
126  }
127 };

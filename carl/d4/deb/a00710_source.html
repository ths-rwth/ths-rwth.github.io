114  /**
115  * Construct from Polynomial and Interval
116  * Asserts that the polynomial has exactly one root in the given interval
117  * Libpoly uses dyadic intervals, to not approximate, we calculate the roots of the polynomial and filter out all roots not in the given interval
118  * @param p Polynomial, which roots represents the algebraic number
119  * @param i Interval, in which the root must lie
120  */
121  real_algebraic_number_libpoly(const carl::UnivariatePolynomial<Number>& p, const Interval<Number>& i) {
122  CARL_LOG_DEBUG("carl.ran.libpoly", " Create safe from poly: " << p << " in interval: " << i);
123 
124  poly::UPolynomial upoly = LibpolyConverter::getInstance().toLibpolyUPolynomial(p);
125  poly::Interval inter_poly = LibpolyConverter::getInstance().toInterval(i);
126 
127  CARL_LOG_DEBUG("carl.ran.libpoly", " Converted to poly: " << upoly << " in interval: " << inter_poly);
128 
129  //actual calculations
130  std::vector<poly::AlgebraicNumber> roots = poly::isolate_real_roots(upoly);
131  std::vector<poly::AlgebraicNumber> res;
132 
133  for (const auto& val : roots) {
134  // filter out roots not in interval
135  if (poly::contains(inter_poly, poly::Value(val))) {
136  CARL_LOG_DEBUG("carl.ran.libpoly", " Found Root " << val);
137  res.emplace_back(val);
138  }
139  }
140 
141  assert(res.size() == 1);
142  lp_algebraic_number_construct_zero(get_internal());
143  lp_algebraic_number_swap(get_internal(), res.front().get_internal());
144  //We dont need to free the items in res, as they're from the C++ interface
145  }

276  const Number value() const {

294  mpq_canonicalize(&result);

296  }else{
297  // b = 0
298  mpq_init(&result);
299  }
300  }
301  Number num = LibpolyConverter::getInstance().toNumber<Number>(&result) ;

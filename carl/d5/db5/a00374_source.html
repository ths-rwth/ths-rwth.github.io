45 template<typename Coeff>
46 Definiteness definiteness(const Term<Coeff>& t) {
47  if (t.monomial()) {
48  if (t.monomial()->is_square()) {
49  return (t.coeff() < Coeff(0) ? Definiteness::NEGATIVE_SEMI : Definiteness::POSITIVE_SEMI);
50  }
51  } else if (!carl::is_zero(t.coeff())) {
52  return (t.coeff() < Coeff(0) ? Definiteness::NEGATIVE : Definiteness::POSITIVE);
53  }
54  return Definiteness::NON;
55 }

280  Sign sgn(const Polynomial& p) const {
281  Polynomial tmp = replaceVariable(p);
282  if (polynomial_int() == tmp) return Sign::ZERO;
283  auto seq = carl::sturm_sequence(polynomial_int(), derivative(polynomial_int()) * tmp);
284  int variations = carl::count_real_roots(seq, interval_int());
285  assert((variations == -1) || (variations == 0) || (variations == 1));
286  switch (variations) {
287  case -1:
288  return Sign::NEGATIVE;
289  case 0:
290  return Sign::ZERO;
291  case 1:
292  return Sign::POSITIVE;
293  default:
294  CARL_LOG_ERROR("carl.ran", "Unexpected number of variations, should be -1, 0, 1 but was " << variations);

371 template<typename Number>
372 bool compare(const real_algebraic_number_interval<Number>& lhs, const real_algebraic_number_interval<Number>& rhs, const Relation relation) {
373  CARL_LOG_DEBUG("carl.ran", "Compare " << lhs << " " << relation << " " << rhs);
374 
375  if (lhs.m_content.get() == rhs.m_content.get()) {
376  CARL_LOG_TRACE("carl.ran", "Contents are equal");
377  return evaluate(Sign::ZERO, relation);
378  }
379 
380  if (lhs.interval_int().isPointInterval() && rhs.interval_int().isPointInterval()) {
381  CARL_LOG_TRACE("carl.ran", "Point interval comparison");
382  return evaluate(lhs.interval_int().lower(), relation, rhs.interval_int().lower());
383  }
384 
385  if (carl::set_have_intersection(lhs.interval_int(), rhs.interval_int())) {
386  CARL_LOG_TRACE("carl.ran", "Intervals " << lhs.interval_int() << " and " << rhs.interval_int() << " do intersect");
387  auto intersection = carl::set_intersection(lhs.interval_int(), rhs.interval_int());
388  assert(!intersection.isEmpty());
389  lhs.refine_using(intersection.lower());
390  rhs.refine_using(intersection.lower());
391  if (!intersection.isPointInterval()) {
392  lhs.refine_using(intersection.upper());
393  rhs.refine_using(intersection.upper());
394  }
395  }
396  // now: intervals are either equal or disjoint
397  assert(!carl::set_have_intersection(lhs.interval_int(), rhs.interval_int()) || lhs.interval_int() == rhs.interval_int());
398  if (lhs.interval_int() == rhs.interval_int()) {
399  CARL_LOG_TRACE("carl.ran", "Intervals " << lhs.interval_int() << " and " << rhs.interval_int() << " are equal");
400  if (lhs.is_numeric()) {
401  CARL_LOG_TRACE("carl.ran", "Interval " << lhs.interval_int() << " is a point interval");
402  return evaluate(Sign::ZERO, relation);
403  }
404  if (lhs.polynomial_int() == rhs.polynomial_int()) {
405  CARL_LOG_TRACE("carl.ran", "Polynomials " << lhs.polynomial_int() << " and " << rhs.polynomial_int() << " are equal");
406  return evaluate(Sign::ZERO, relation);
407  }
408  auto g = carl::gcd(lhs.polynomial_int(), rhs.polynomial_int());
409  auto lsgn = carl::sgn(carl::evaluate(g, lhs.interval_int().lower()));
410  auto usgn = carl::sgn(carl::evaluate(g, lhs.interval_int().upper()));
411  if (lsgn != usgn) {
412  CARL_LOG_TRACE("carl.ran", "gcd(lhs,rhs) has a zero in the common interval");
413  lhs.set_polynomial(g, lsgn);
414  rhs.set_polynomial(g, lsgn);
415  return evaluate(Sign::ZERO, relation);
416  } else {
417  CARL_LOG_TRACE("carl.ran", "gcd(lhs,rhs) has no zero in the common interval");
418  if (relation == Relation::EQ) return false;
419  if (relation == Relation::NEQ) return true;
420  CARL_LOG_TRACE("carl.ran", "Refine until intervals become disjoint");
421  while (lhs.interval_int() == rhs.interval_int()) {
422  lhs.refine();
423  rhs.refine();
424  }
425  }
426  }
427  // now: intervals are disjoint
428  CARL_LOG_TRACE("carl.ran", "Intervals " << lhs.interval_int() << " and " << rhs.interval_int() << " are disjoint");
429  assert(!carl::set_have_intersection(lhs.interval_int(), rhs.interval_int()));
430  if (lhs.interval_int().upper() <= rhs.interval_int().lower()) {
431  return relation == Relation::LESS || relation == Relation::LEQ;
432  }
433  if (lhs.interval_int().lower() >= rhs.interval_int().upper()) {
434  return relation == Relation::GREATER || relation == Relation::GEQ;
435  }
436 
437  assert(false);
438  return false;
439 }

442 bool compare(const real_algebraic_number_interval<Number>& lhs, const Number& rhs, const Relation relation) {
443  auto res = lhs.refine_using(rhs);
444  if (res) {
445  return evaluate(*res, relation);
446  } else {
447  if (relation == Relation::EQ)
448  return false;

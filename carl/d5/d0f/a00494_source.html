65  /**
66  * Convert this variable comparison "v < root(..)" into a simpler
67  * polynomial (in)equality against zero "p(..) < 0" if that is possible.
68  * @return std::nullopt if conversion impossible.
69  */
70  std::optional<Constraint<Poly>> asConstraint() const {
71  Relation rel = negated() ? inverse(mRelation) : mRelation;
72  if (std::holds_alternative<MR>(mValue)) {
73  const MR& mr = std::get<MR>(mValue);
74  if (mr.poly().degree(mr.var()) != 1) return std::nullopt;
75  if (mr.k() != 1) return std::nullopt;
76  auto lcoeff = mr.poly().coeff(mr.var(), 1);
77  if (!lcoeff.isConstant()) return std::nullopt;
78  auto ccoeff = mr.poly().coeff(mr.var(), 0);
79  return Constraint<Poly>(Poly(mVar) + ccoeff / lcoeff, rel);
80  }
81  if (!std::get<RAN>(mValue).is_numeric()) return std::nullopt;
82  return Constraint<Poly>(Poly(mVar) - Poly(std::get<RAN>(mValue).value()), rel);
83  }

85  /**
86  * Return a polynomial containing the lhs-variable that has a same root
87  * for the this lhs-variable as the value that rhs represent, e.g. if this
88  * variable comparison is 'v < 3' then a defining polynomial could be 'v-3',
89  * because it has the same root for variable v, i.e., v=3.
90  */
91  Poly definingPolynomial() const {
92  if (std::holds_alternative<RAN>(mValue)) {
93  const auto& ran = std::get<RAN>(mValue);
94  if (ran.is_numeric()) return Poly(mVar) - ran.value();
95  return Poly(carl::replace_main_variable(ran.polynomial(), mVar));
96  } else {
97  const auto& mr = std::get<MR>(mValue);
98  return mr.poly(mVar);
99  }
100  }

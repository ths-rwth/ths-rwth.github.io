61  {
62  auto it = mVariableInfo.find(ve.first);
63  // Variable did not occur before.
64  if(it == mVariableInfo.end())
65  {
66  // TODO GCC 4.7 does not support the (faster) emplace on maps
67  //it = mVariableInfo.emplace(ve.first, VariableInformation<collectCoeff, CoeffType>(ve.second)).first;
68  it = mVariableInfo.insert(std::make_pair(ve.first, VariableInformation<collectCoeff, CoeffType>(ve.second))).first;
69  }
70  else
71  {
72  // One more term in which the variable occurs.
73  it->second.increaseOccurence();
74  // Update minimal/maximal degree.
75  if(!it->second.raiseMaxDegree(ve.second))
76  {
77  // Only if raising failed, lowering can be successful.
78  it->second.lowerMinDegree(ve.second);
79  }
80  }
81  if(collectCoeff)
82  {
83  typename CoeffType::MonomType::Arg m = monomial.dropVariable(ve.first);
84  if(m == nullptr)
85  {
86  it->second.updateCoeff(ve.second, typename CoeffType::TermType(termCoeff));
87  }
88  else
89  {
90  it->second.updateCoeff(ve.second, typename CoeffType::TermType(termCoeff, m));
91  }

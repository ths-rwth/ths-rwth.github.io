84  void check_rehash() {
85  auto rehash = mRehashPolicy.needRehash(mPool.bucket_count(), mPool.size());
86  if (rehash.first) {
87  auto new_buckets = new typename underlying_set::bucket_type[rehash.second];
88  mPool.rehash(typename underlying_set::bucket_traits(new_buckets, rehash.second));

652  Logic logic() const {
653  if (carl::PROP_CONTAINS_BITVECTOR <= properties()
654  && !(carl::PROP_CONTAINS_UNINTERPRETED_EQUATIONS <= properties())
655  && !(carl::PROP_CONTAINS_INTEGER_VALUED_VARS <= properties())
656  && !(carl::PROP_CONTAINS_REAL_VALUED_VARS <= properties())
657  && !(carl::PROP_CONTAINS_PSEUDOBOOLEAN <= properties())) {
658  return Logic::QF_BV;
659  } else if (!(carl::PROP_CONTAINS_BITVECTOR <= properties())
660  && (carl::PROP_CONTAINS_UNINTERPRETED_EQUATIONS <= properties())
661  && !(carl::PROP_CONTAINS_INTEGER_VALUED_VARS <= properties())
662  && !(carl::PROP_CONTAINS_REAL_VALUED_VARS <= properties())
663  && !(carl::PROP_CONTAINS_PSEUDOBOOLEAN <= properties())) {
664  return Logic::QF_UF;
665  } else if (!(carl::PROP_CONTAINS_BITVECTOR <= properties())
666  && !(carl::PROP_CONTAINS_UNINTERPRETED_EQUATIONS <= properties())
667  && !(carl::PROP_CONTAINS_INTEGER_VALUED_VARS <= properties())
668  && !(carl::PROP_CONTAINS_REAL_VALUED_VARS <= properties())
669  && (carl::PROP_CONTAINS_PSEUDOBOOLEAN <= properties())) {
670  return Logic::QF_PB;
671  } else if (!(carl::PROP_CONTAINS_BITVECTOR <= properties())
672  && !(carl::PROP_CONTAINS_UNINTERPRETED_EQUATIONS <= properties())
673  && (carl::PROP_CONTAINS_INTEGER_VALUED_VARS <= properties())
674  && !(carl::PROP_CONTAINS_REAL_VALUED_VARS <= properties())
675  && !(carl::PROP_CONTAINS_PSEUDOBOOLEAN <= properties())) {

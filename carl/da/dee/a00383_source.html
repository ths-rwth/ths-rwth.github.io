468  /**
469  * @return true, if this constraint is a lower bound.
470  */
471  bool isLowerBound() const
472  {
473  if( isBound() )
474  {
475  if( mpContent->mRelation == Relation::EQ ) return true;
476  const typename Pol::NumberType& coeff = mpContent->mLhs.lterm().coeff();
477  if( coeff < 0 )
478  return (mpContent->mRelation == Relation::LEQ || mpContent->mRelation == Relation::LESS );
479  else
480  {
481  assert( coeff > 0 );

704  * -2, if it is easy to decide that _constraintA has no solution common with _constraintB; not(_constraintA and _constraintB)
705  * -3, if it is easy to decide that _constraintA and _constraintB can be intersected; _constraintA and _constraintB = _constraintC
706  * -4, if it is easy to decide that _constraintA is the inverse of _constraintB; _constraintA xor _constraintB
707  * 0, otherwise.
708  */
709  template<typename Pol>
710  signed compare( const Constraint<Pol>& _constraintA, const Constraint<Pol>& _constraintB )
711  {
712  /*
713  * Check whether it holds that
714  *
715  * _constraintA = a_1*m_1+...+a_k*m_k + c ~ 0
716  * and
717  * _constraintB = b_1*m_1+...+b_k*m_k + d ~ 0,
718  *
719  * where a_1,..., a_k, b_1,..., b_k, c, d are rational coefficients,
720  * m_1,..., m_k are non-constant monomials and
721  * exists a rational g such that
722  *
723  * a_i = g * b_i for all 1<=i<=k
724  * or b_i = g * a_i for all 1<=i<=k
725  */
726  typename Pol::NumberType one_divided_by_a = _constraintA.lhs().coprimeFactorWithoutConstant();
727  typename Pol::NumberType one_divided_by_b = _constraintB.lhs().coprimeFactorWithoutConstant();
728  typename Pol::NumberType c = _constraintA.lhs().constantPart();
729  typename Pol::NumberType d = _constraintB.lhs().constantPart();
730  assert( carl::isOne(carl::getNum(carl::abs(one_divided_by_b))) );
731  Pol tmpA = (_constraintA.lhs() - c) * one_divided_by_a;
732  Pol tmpB = (_constraintB.lhs() - d) * one_divided_by_b;
733 // std::cout << "tmpA = " << tmpA << std::endl;
734 // std::cout << "tmpB = " << tmpB << std::endl;
735  if( tmpA != tmpB ) return 0;
736  bool termACoeffGreater = false;
737  bool signsDiffer = (one_divided_by_a < carl::constant_zero<typename Pol::NumberType>::get()) != (one_divided_by_b < carl::constant_zero<typename Pol::NumberType>::get());
738  typename Pol::NumberType g;
739  if( carl::getDenom(one_divided_by_a) > carl::getDenom(one_divided_by_b) )
740  {
741  g = typename Pol::NumberType(carl::getDenom(one_divided_by_a))/carl::getDenom(one_divided_by_b);
742  if( signsDiffer )
743  g = -g;
744  termACoeffGreater = true;
745  d *= g;
746  }
747  else
748  {
749  g = typename Pol::NumberType(carl::getDenom(one_divided_by_b))/carl::getDenom(one_divided_by_a);
750  if( signsDiffer )
751  g = -g;
752  c *= g;
753  }
754  // Apply the multiplication by a negative g to the according relation symbol, which
755  // has to be turned around then.
756  Relation relA = _constraintA.relation();
757  Relation relB = _constraintB.relation();
758  if( g < 0 )
759  {
760  if( termACoeffGreater )
761  {
762  switch( relB )
763  {
764  case Relation::LEQ:
765  relB = Relation::GEQ;
766  break;
767  case Relation::GEQ:
768  relB = Relation::LEQ;
769  break;
770  case Relation::LESS:
771  relB = Relation::GREATER;
772  break;
773  case Relation::GREATER:
774  relB = Relation::LESS;
775  break;
776  default:
777  break;
778  }
779  }
780  else
781  {
782  switch( relA )
783  {
784  case Relation::LEQ:
785  relA = Relation::GEQ;
786  break;
787  case Relation::GEQ:
788  relA = Relation::LEQ;
789  break;
790  case Relation::LESS:
791  relA = Relation::GREATER;
792  break;
793  case Relation::GREATER:
794  relA = Relation::LESS;
795  break;
796  default:
797  break;
798  }
799  }

8 /**
9  * Constructs the equivalent of
10  * (iff lhs (and *rhs_and)))
11  * The result is the list
12  * (=> lhs (and *rhs_and))
13  * (=> rhs !lhs) (for each rhs in rhs_and)
14  */
15 template<typename Poly>
16 std::vector<Formula<Poly>> construct_iff(const Formula<Poly>& lhs, const std::vector<Formula<Poly>>& rhs_and) {
17  std::vector<Formula<Poly>> res;
18  Formulas<Poly> subs = { lhs };
19  for (const auto& sub: rhs_and) {
20  subs.emplace_back(!sub);
21  res.emplace_back(FormulaType::OR, sub, !lhs);
22  }
23  res.emplace_back(FormulaType::OR, std::move(subs));
24  return res;
25 }

38  // Checks for immediate tautologies among constraints
39  ConstraintBounds<Poly> constraint_bounds;
40  // Resulting subformulas
41  Formulas<Poly> subformulas;
42  // Queue of subformulas to process
43  std::vector<Formula<Poly>> subformula_queue = { f };
44  while (!subformula_queue.empty()) {
45  auto current = subformula_queue.back();
46  CARL_LOG_DEBUG("carl.formula.cnf", "Processing " << current << " from " << subformula_queue);
47  subformula_queue.pop_back();
48 
49  switch (current.getType()) {
50  case FormulaType::TRUE:
51  return Formula<Poly>(FormulaType::TRUE);
52  case FormulaType::FALSE:
53  break;
54  case FormulaType::BITVECTOR:
55  case FormulaType::BOOL:
56  case FormulaType::UEQ:
57  case FormulaType::VARASSIGN:
58  case FormulaType::VARCOMPARE:
59  subformulas.emplace_back(current);
60  break;
61  case FormulaType::CONSTRAINT:
62  // Try simplification with ConstraintBounds
63  if (simplify_combinations) {
64  if (Formula<Poly>::addConstraintBound(constraint_bounds, current, false).isFalse()) {
65  CARL_LOG_DEBUG("carl.formula.cnf", "Adding " << current << " to constraint bounds yielded a tautology");
66  return Formula<Poly>(FormulaType::TRUE);
67  }
68  } else {
69  subformulas.emplace_back(current);
70  }
71  break;
72  case FormulaType::NOT: {
73  // Resolve negation
74  auto resolved = current.resolveNegation(keep_constraints);
75  if (resolved.isLiteral()) {
76  subformulas.emplace_back(resolved);
77  } else {
78  subformula_queue.emplace_back(resolved);
79  }
80  break;
81  }
82  case FormulaType::IMPLIES:
83  // (=> A B) -> (not A), B
84  subformula_queue.emplace_back(!current.premise());
85  subformula_queue.emplace_back(current.conclusion());
86  break;
87  case FormulaType::ITE:
88  // (ite C T E) -> (and (=> C T) (=> (not C) E)) -> (and (or (not C) T) (or C E))
89  subformula_queue.emplace_back(Formula<Poly>(FormulaType::AND, {
90  Formula<Poly>(FormulaType::OR, {
91  !current.condition(), current.firstCase()
92  }),
93  Formula<Poly>(FormulaType::OR, {
94  current.condition(), current.secondCase()
95  })
96  }));
97  break;
98  case FormulaType::IFF: {
99  // (iff A ...) -> (and A ...), (and (not A) ...)
100  Formulas<Poly> subA;
101  Formulas<Poly> subB;
102  for (const auto& sub: current.subformulas()) {
103  subA.emplace_back(sub);
104  subB.emplace_back(!sub);
105  }
106  subformula_queue.emplace_back(Formula<Poly>(FormulaType::AND, std::move(subA)));
107  subformula_queue.emplace_back(Formula<Poly>(FormulaType::AND, std::move(subB)));
108  break;
109  }
110  case FormulaType::XOR: {
111  // (xor A B) -> (and A (not B)), (and (not A) B)
112  auto lhs = current.connectPrecedingSubformulas();
113  const auto& rhs = current.subformulas().back();
114  subformula_queue.emplace_back(Formula<Poly>(FormulaType::AND, { lhs, !rhs }));
115  subformula_queue.emplace_back(Formula<Poly>(FormulaType::AND, { !lhs, rhs }));
116  break;
117  }
118  case FormulaType::OR:
119  // Simply add subformulas to the queue
120  for (const auto& sub: current.subformulas()) {
121  subformula_queue.emplace_back(sub);
122  }
123  break;
124  case FormulaType::AND: {
125  // Replace by a fresh tseitin variable.
126  auto tseitinVar = FormulaPool<Poly>::getInstance().createTseitinVar(current);
127  if (tseitin_equivalence) {
128  tseitin.emplace_back(Formula<Poly>(FormulaType::IFF, { tseitinVar, current }));
129  } else {
130  tseitin.emplace_back(Formula<Poly>(FormulaType::IMPLIES, { tseitinVar, current }));
131  }
132  subformulas.emplace_back(tseitinVar);
133  break;
134  }
135  case FormulaType::EXISTS:
136  case FormulaType::FORALL:
137  CARL_LOG_ERROR("carl.formula.cnf", "Cannot transform quantified formula to CNF");
138  assert(false);
139  break;
140  }
141  }
142  if (simplify_combinations && Formula<Poly>::swapConstraintBounds(constraint_bounds, subformulas, false)) {
143  return Formula<Poly>(FormulaType::TRUE);
144  } else if (subformulas.empty()) {
145  tseitin.clear();
146  return Formula<Poly>(FormulaType::FALSE);
147  } else if (subformulas.size() == 1) {
148  return subformulas.front();
149  }
150  return Formula<Poly>(FormulaType::OR, std::move(subformulas));
151 }

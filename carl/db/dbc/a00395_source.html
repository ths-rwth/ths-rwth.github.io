167 template<typename Pol>
168 inline void RawConstraint<Pol>::set_constraint(const Pol& _lhs, Relation _rel) {
169  if (_lhs.isConstant()) {
170  CARL_LOG_DEBUG("carl.formula.constraint", _lhs << " is constant, we simply evaluate.");
171  evaluate(_lhs.constantPart(), _rel) ? set_trivial(true) : set_trivial(false);
172  } else if (_lhs.totalDegree() == 1 && (_rel != Relation::EQ && _rel != Relation::NEQ) && _lhs.isUnivariate()) {
173  if (carl::isNegative(_lhs.lcoeff())) {
174  CARL_LOG_DEBUG("carl.formula.constraint", "Normalizing leading coefficient of linear poly.");
175  switch (_rel) {
176  case Relation::LESS:
177  _rel = Relation::GREATER;
178  break;
179  case Relation::GREATER:
180  _rel = Relation::LESS;
181  break;
182  case Relation::LEQ:
183  _rel = Relation::GEQ;
184  break;
185  case Relation::GEQ:
186  _rel = Relation::LEQ;
187  break;
188  default:
189  assert(false);
190  break;
191  }
192  }
193  CARL_LOG_DEBUG("carl.formula.constraint", "Rewriting to bound");
194  set_bound(_lhs.getSingleVariable(), _rel, (-_lhs.constantPart()) / _lhs.lcoeff());
195  } else {
196  CARL_LOG_DEBUG("carl.formula.constraint", "Normalizing " << _lhs << " " << _rel << " 0");
197  mLhs = isZero(_lhs) ? Pol(typename Pol::NumberType(0)) : _lhs.coprimeCoefficients();
198  if (_rel == Relation::EQ || _rel == Relation::NEQ) {
199  if (!isZero(_lhs) && mLhs.lterm().coeff() < typename Pol::NumberType(0)) mLhs = -mLhs;
200  mRelation = _rel;
201  } else if (_rel == Relation::LEQ || _rel == Relation::LESS) {
202  if (!isZero(mLhs) && (_lhs.lterm().coeff() < 0) != (mLhs.lterm().coeff() < 0)) mLhs = -mLhs;
203  mRelation = _rel;
204  } else if (_rel == Relation::GREATER) {
205  if (!isZero(mLhs) && (_lhs.lterm().coeff() < 0) == (mLhs.lterm().coeff() < 0)) mLhs = -mLhs;
206  mRelation = Relation::LESS;
207  } else if (_rel == Relation::GEQ) {
208  if (!isZero(mLhs) && (_lhs.lterm().coeff() < 0) == (mLhs.lterm().coeff() < 0)) mLhs = -mLhs;
209  mRelation = Relation::LEQ;
210  }
211  CARL_LOG_DEBUG("carl.formula.constraint", "-> " << mLhs << " " << mRelation << " 0");
212  }

238 template<typename Pol>
239 inline unsigned RawConstraint<Pol>::is_consistent() const {
240  if (mLhs.isConstant()) {
241  CARL_LOG_DEBUG("carl.formula.constraint", "Lhs " << mLhs << " is constant");
242  return carl::evaluate(mLhs.constantPart(), mRelation) ? 1 : 0;
243  } else {
244  CARL_LOG_DEBUG("carl.formula.constraint", "Checking " << mRelation << " against " << mLhsDefiniteness);
245  switch (mRelation) {
246  case Relation::EQ: {
247  if (mLhsDefiniteness == Definiteness::POSITIVE || mLhsDefiniteness == Definiteness::NEGATIVE) return 0;
248  break;
249  }
250  case Relation::NEQ: {
251  if (mLhsDefiniteness == Definiteness::POSITIVE || mLhsDefiniteness == Definiteness::NEGATIVE) return 1;
252  break;
253  }
254  case Relation::LESS: {
255  if (mLhsDefiniteness == Definiteness::NEGATIVE) return 1;
256  if (mLhsDefiniteness >= Definiteness::POSITIVE_SEMI) return 0;
257  break;
258  }
259  case Relation::GREATER: {
260  if (mLhsDefiniteness == Definiteness::POSITIVE) return 1;
261  if (mLhsDefiniteness <= Definiteness::NEGATIVE_SEMI) return 0;
262  break;
263  }
264  case Relation::LEQ: {
265  if (mLhsDefiniteness <= Definiteness::NEGATIVE_SEMI) return 1;
266  if (mLhsDefiniteness == Definiteness::POSITIVE) return 0;
267  break;
268  }
269  case Relation::GEQ: {
270  if (mLhsDefiniteness >= Definiteness::POSITIVE_SEMI) return 1;
271  if (mLhsDefiniteness == Definiteness::NEGATIVE) return 0;
272  break;
273  }
274  default:
275  assert(false);
276  return false;
277  }
278  return 2;

282 template<typename Pol>
283 inline bool RawConstraint<Pol>::simplify_nonlinear_univariate_monomial() {
284  if (!(mVariables.size() == 1 && !mLhs.isLinear() && mLhs.nrTerms() == 1)) return false;
285 
286  // Left-hand side is a non-linear univariate monomial
287  Relation rel = mRelation;
288  if ((mLhsDefiniteness == Definiteness::POSITIVE_SEMI && rel == Relation::LEQ) || (mLhsDefiniteness == Definiteness::NEGATIVE_SEMI && rel == Relation::GEQ))
289  rel = Relation::EQ;
290  Variable var = mVariables.as_vector()[0];
291 
292  switch (rel) {
293  case Relation::EQ:
294  set_constraint(PolyT(var), rel);
295  break;
296  case Relation::NEQ:
297  set_constraint(PolyT(var), rel);
298  break;
299  case Relation::LEQ:
300  if (mLhsDefiniteness == Definiteness::NEGATIVE_SEMI) {
301  set_constraint(PolyT(typename Pol::NumberType(-1)) * PolyT(var) * PolyT(var), rel);
302  } else {
303  set_constraint((mLhs.trailingTerm().coeff() > 0 ? PolyT(typename Pol::NumberType(1)) : PolyT(typename Pol::NumberType(-1))) * PolyT(var), rel);
304  }
305  break;
306  case Relation::GEQ:
307  if (mLhsDefiniteness == Definiteness::POSITIVE_SEMI) {
308  set_constraint(PolyT(var) * PolyT(var), rel);
309  } else {
310  set_constraint((mLhs.trailingTerm().coeff() > 0 ? PolyT(typename Pol::NumberType(1)) : PolyT(typename Pol::NumberType(-1))) * PolyT(var), rel);
311  }
312  break;
313  case Relation::LESS:
314  if (mLhsDefiniteness == Definiteness::NEGATIVE_SEMI) {
315  set_constraint(PolyT(var), Relation::NEQ);
316  } else {
317  if (mLhsDefiniteness == Definiteness::POSITIVE_SEMI) {
318  set_constraint(PolyT(var) * PolyT(var), rel);
319  } else {
320  set_constraint((mLhs.trailingTerm().coeff() > 0 ? PolyT(typename Pol::NumberType(1)) : PolyT(typename Pol::NumberType(-1))) * PolyT(var), rel);
321  }
322  }
323  break;
324  case Relation::GREATER:
325  if (mLhsDefiniteness == Definiteness::POSITIVE_SEMI) {
326  set_constraint(PolyT(var), Relation::NEQ);
327  } else {
328  if (mLhsDefiniteness == Definiteness::NEGATIVE_SEMI) {
329  set_constraint(PolyT(typename Pol::NumberType(-1)) * PolyT(var) * PolyT(var), rel);
330  } else {
331  set_constraint((mLhs.trailingTerm().coeff() > 0 ? PolyT(typename Pol::NumberType(1)) : PolyT(typename Pol::NumberType(-1))) * PolyT(var), rel);
332  }
333  }
334  break;

342 template<typename Pol>
343 inline bool RawConstraint<Pol>::simplify_integer() {
344  if (!is_integer()) return false;
345 
346  typename Pol::NumberType constPart = mLhs.constantPart();
347  if (constPart != typename Pol::NumberType(0)) {
348  // Find the gcd of the coefficients of the non-constant terms.
349  typename Pol::NumberType g = carl::abs(mLhs.coprimeFactorWithoutConstant());
350  assert(g != typename Pol::NumberType(0));
351  if (carl::mod(carl::getNum(constPart), carl::getDenom(g)) != 0) {
352  switch (mRelation) {
353  case Relation::EQ: {
354  set_constraint(Pol(typename Pol::NumberType(0)), Relation::LESS);
355  break;
356  }
357  case Relation::NEQ: {
358  set_constraint(Pol(typename Pol::NumberType(0)), Relation::EQ);
359  break;
360  }
361  case Relation::LEQ: {
362  Pol newLhs = ((mLhs - constPart) * g);
363  newLhs += carl::floor((constPart * g)) + typename Pol::NumberType(1);
364  set_constraint(newLhs, Relation::LEQ);
365  break;
366  }
367  case Relation::GEQ: {
368  Pol newLhs = ((mLhs - constPart) * g);
369  newLhs += carl::floor((constPart * g));
370  set_constraint(newLhs, Relation::GEQ);
371  break;
372  }
373  case Relation::LESS: {
374  Pol newLhs = ((mLhs - constPart) * g);
375  newLhs += carl::floor((constPart * g)) + typename Pol::NumberType(1);
376  set_constraint(newLhs, Relation::LEQ);
377  break;
378  }
379  case Relation::GREATER: {
380  Pol newLhs = ((mLhs - constPart) * g);
381  newLhs += carl::floor((constPart * g));
382  set_constraint(newLhs, Relation::GEQ);
383  break;
384  }
385  default:
386  assert(false);
387  }
388  return true;
389  }
390  }
391  return false;

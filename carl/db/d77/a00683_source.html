3 /**
4  * @file AlgebraicSubstitution.h
5  * This file contains carl::algebraic_substitution which performs what we call an ``algebraic substitution``.
6  * We substitute the ``algebraic part`` of a variable assignment into a multivariate polynomial to obtain a univariate polynomial in the remaining variable.
7  * The algebraic part is a minimal polynomial, in practice the defining polynomial of a real algebraic number.
8  * We implement two strategies: GROEBNER and RESULTANT.
9  *
10  * In both cases you need to make sure that the minimal polynomials are not only reducible over Q, but are actually minimal polynomials within a tower of field extensions characterized by the given polynomials.
11  * If one fails to do this, the resulting polynomial may vanish identically.
12  * In some cases one can exclude that this happens.
13  *
14  * Such minimal polynomials can be computed using FieldExtensions.
15  * If the resultant strategy is used, it is important that the defining polynomials are used from top to bottom.
16  * Thus the first defining polynomials may contain all variables, the last must be univariate.
17  */

30  * Implements algebraic substitution by Gröbner basis computation.
31  * Essentially we take all polynomials and compute a Gröbner basis with respect to an elimination order, having the remaining variable at the end.
32  * The result is then the polynomial in the last variable only.
33  */
34 template<typename Number>
35 std::optional<UnivariatePolynomial<Number>> algebraic_substitution_groebner(
36  const std::vector<MultivariatePolynomial<Number>>& polynomials,
37  const std::vector<Variable>& variables
38 ) {
39  Variable target = variables.back();
40  try {
41  CoCoAAdaptor<MultivariatePolynomial<Number>> ca(variables, true);
42  CARL_LOG_DEBUG("carl.algsubs", "Computing GBasis of " << polynomials << " with order " << variables);
43  auto res = ca.GBasis(polynomials);
44  CARL_LOG_DEBUG("carl.algsubs", "-> " << res);
45  for (const auto& poly: res) {
46  if (carl::variables(poly) == carlVariables({ target })) {
47  CARL_LOG_DEBUG("carl.algsubs", "-> " << poly)
48  return carl::to_univariate_polynomial(poly);
49  }
50  }
51  } catch (const CoCoA::ErrorInfo& e) {
52  CARL_LOG_ERROR("carl.algsubs", "Computation of GBasis failed: " << e << " -> " << CoCoA::context(e));
53  }
54  return std::nullopt;

81 /**
82  * Implements algebraic substitution by resultant computation.
83  * We iteratively compute the resultant of the input polynomial with each of the defining polynomials.
84  * Eventually we obtain a polynomial univariate in the remaining variable, our result.
85  *
86  * Note that we assume that the polynomials are in a triangular form where any polynomial may contain variables that are ``defined'' by the previous polynomials.
87  */
88 template<typename Number>
89 std::optional<UnivariatePolynomial<Number>> algebraic_substitution_resultant(
90  const UnivariatePolynomial<MultivariatePolynomial<Number>>& p,
91  const std::vector<UnivariatePolynomial<MultivariatePolynomial<Number>>>& polynomials
92 ) {
93  Variable v = p.mainVar();
94  UnivariatePolynomial<MultivariatePolynomial<Number>> cur = p;
95  for (auto it = polynomials.rbegin(); it != polynomials.rend(); ++it) {
96  const auto& poly = *it;
97  if (!cur.has(poly.mainVar())) {
98  continue;
99  }
100  cur = pseudo_remainder(switch_main_variable(cur, poly.mainVar()), poly);
101  CARL_LOG_DEBUG("carl.algsubs", "Computing resultant of " << cur << " and " << poly);
102  cur = carl::resultant(cur, poly);
103  CARL_LOG_DEBUG("carl.algsubs", "-> " << cur);
104  }
105  auto swpoly = switch_main_variable(cur, v);
106  if (!swpoly.isUnivariate()) {
107  return std::nullopt;
108  }
109  UnivariatePolynomial<Number> result = swpoly.toNumberCoefficients();
110  CARL_LOG_DEBUG("carl.algsubs", "Result: " << result);
111  return result;
112 }

115  * Implements algebraic substitution by resultant computation.
116  * We iteratively compute the resultant of the input polynomial with each of the defining polynomials.
117  * Eventually we obtain a polynomial univariate in the remaining variable, our result.
118  *
119  * Note that we assume that the polynomials are in a triangular form where any polynomial may contain variables that are ``defined'' by the previous polynomials.
120  */
121 template<typename Number>
122 std::optional<UnivariatePolynomial<Number>> algebraic_substitution_resultant(
123  const std::vector<MultivariatePolynomial<Number>>& polynomials,
124  const std::vector<Variable>& variables
125 ) {
126  auto p = carl::to_univariate_polynomial(polynomials.back(), variables.back());
127  std::vector<UnivariatePolynomial<MultivariatePolynomial<Number>>> polys;
128 
129  for (std::size_t i = 0; i < polynomials.size() - 1; ++i) {
130  polys.emplace_back(carl::to_univariate_polynomial(polynomials[i], variables[i]));
131  }
132 
133  CARL_LOG_DEBUG("carl.algsubs", "Converted " << polynomials << " under " << variables << " to " << p << " and " << polys);
134  return algebraic_substitution_resultant(p, polys);
135 }

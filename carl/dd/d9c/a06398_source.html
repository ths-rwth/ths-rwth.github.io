115 /**
116  * Tries to convert num = a/b into a dyadic rational of the form c/2^d
117  * We take d = precision * ceil(log(2,b)) and c = ceil((a * 2^d)/(b))
118  * @return Approximation to num by a dyadic rational
119  */
120 inline poly::DyadicRational get_libpoly_dyadic_approximation(const mpq_class& num, const unsigned int& precision) {
121  CARL_LOG_DEBUG("carl.converter", "Starting Dyadic Approximation with: " << num);
122 
123  mpz_class a = num.get_num();
124  mpz_class b = num.get_den();
125 
126  mpz_class d = (precision)*carl::ceil(carl::log(b)); //unsigned long
127  assert(d.fits_ulong_p());
128 
129  mpz_class c = carl::ceil((a * carl::pow(mpq_class(2), d.get_ui())) / (b));
130 
131  if (!c.fits_slong_p()) {
132  CARL_LOG_DEBUG("carl.converter", "Dyadic Rational: Fallback for overflow of numerator");
133  poly::DyadicRational tmp = get_libpoly_dyadic_approximation(c); //Construct with arbitrary precision
134  return poly::div_2exp(tmp, d.get_ui());
135  }
136 
137  CARL_LOG_DEBUG("carl.converter", "Got d: " << d << " and c " << c);
138  CARL_LOG_DEBUG("carl.converter", "Got dyadic number: " << poly::DyadicRational(c.get_si(), d.get_ui()));
139 
140  return poly::DyadicRational(c.get_si(), d.get_ui());
141 }

147 inline poly::Interval to_libpoly_interval(const carl::Interval<mpq_class>& inter) {
148  poly::Value low;
149  bool low_open;
150  switch (inter.lower_bound_type()) {
151  case BoundType::STRICT:
152  low = poly::Value(poly::Rational(inter.lower()));
153  low_open = true;
154  break;
155  case BoundType::WEAK:
156  low = poly::Value(poly::Rational(inter.lower()));
157  low_open = false;
158  break;
159  case BoundType::INFTY:
160  low = poly::Value::minus_infty();
161  low_open = true;
162  break;
163  }
164  poly::Value up;
165  bool up_open;
166  switch (inter.upper_bound_type()) {
167  case BoundType::STRICT:
168  up = poly::Value(poly::Rational(inter.upper()));
169  up_open = true;
170  break;
171  case BoundType::WEAK:
172  up = poly::Value(poly::Rational(inter.upper()));
173  up_open = false;
174  break;
175  case BoundType::INFTY:

177  up_open = false;
178  break;
179  }

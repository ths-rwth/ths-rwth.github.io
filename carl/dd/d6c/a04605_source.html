283 inline mpz_class round(const mpq_class& n) {
284  if (isZero(mpz_class(n.get_num_mpz_t()))) return carl::constant_zero<mpz_class>::get();
285  mpz_class res;
286  mpz_class rem;
287  mpz_fdiv_qr(res.get_mpz_t(), rem.get_mpz_t(), n.get_num_mpz_t(), n.get_den_mpz_t());
288  rem *= 2;
289  if (rem >= getDenom(n)) ++res;
290  return res;
291 }

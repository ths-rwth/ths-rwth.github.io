53  template<typename Poly>
54  Poly convertMV(const CoCoA::RingElem& p) const {
55  Poly res;
56  for (CoCoA::SparsePolyIter i = CoCoA::BeginIter(p); !CoCoA::IsEnded(i); ++i) {
57  Poly coeff;
58  CoCoA::BigRat numcoeff;
59  if (CoCoA::IsRational(numcoeff, CoCoA::coeff(i))) {
60  coeff = Poly(convert<typename Poly::CoeffType>(numcoeff));
61  } else {
62  coeff = convertMV<Poly>(CoCoA::CanonicalRepr(CoCoA::coeff(i)));
63  }
64  if (CoCoA::IsOne(CoCoA::PP(i))) {
65  res += coeff;
66  } else {
67  std::vector<long> exponents;
68  CoCoA::exponents(exponents, CoCoA::PP(i));
69  Monomial::Content monContent;
70  std::size_t tdeg = 0;
71  for (std::size_t i = 0; i < exponents.size(); ++i) {
72  if (exponents[i] == 0) continue;
73  const auto& ring = CoCoA::owner(p);
74  monContent.emplace_back(mSymbolBack.at(std::make_pair(CoCoA::RingID(ring),i)), exponents[i]);
75  tdeg += std::size_t(exponents[i]);
76  }
77  res += coeff * createMonomial(std::move(monContent), tdeg);
78  }
79  }
80  return res;
81  }

83  template<typename Poly>
84  CoCoA::RingElem convertMV(const Poly& p, const CoCoA::ring& ring) const {
85  CoCoA::RingElem res(ring);
86  for (const auto& t: p) {
87  if (!t.monomial()) {
88  res += convert(t.coeff());
89  continue;
90  }
91  std::vector<long> exponents(mSymbolBack.size());
92  for (const auto& p: *t.monomial()) {
93  auto it = mSymbolThere.find(p.first);
94  assert(it != mSymbolThere.end());
95  long indetIndex;
96  if (CoCoA::IsIndet(indetIndex, it->second)) {
97  exponents[std::size_t(indetIndex)] = long(p.second);
98  } else {
99  assert(false && "The symbol is not an inderminant.");
100  }
101  }
102  res += CoCoA::monomial(ring, convert(t.coeff()), exponents);
103  }
104  return res;

122  * This class can be used to construct iterated field extensions from a sequence of real algebraic numbers.
123  * In particular it makes sure that the minimal polynomials are "reduced", i.e. making sure that they are minimal polynomial w.r.t. the current extension field.
124  */

137  bool evaluatesToZero(const CoCoA::RingElem& p) const {
138  auto mp = cc.convertMV<Poly>(p);
139  auto res = carl::evaluate(Constraint<Poly>(mp, Relation::EQ), mModel);
140  CARL_LOG_DEBUG("carl.fieldext", "Evaluated " << p << " -> " << mp << " == 0 -> " << res);
141  assert(!indeterminate(res));
142  return (bool)res;
143  }

145  void extendRing(const CoCoA::ring& ring, const CoCoA::RingElem& p) {
146  mQ = CoCoA::NewQuotientRing(ring, CoCoA::ideal(p));
147  }
148 public:
149  /**
150  * Extend the current number field with the field extension defined by r.
151  * The minimal polynomial of r (with is a minimal polynomials in Q[x]) is
152  * embedded into the current number field and the minimal polynomial for r within this number field is computed.
153  * The resulting polynomial is this minimal polynomial over the current number field.
154  *
155  * We may have one of two cases:
156  * - We can eliminate v by substitution with some term
157  * - We create a new field extension and may have to reduce the lifting polynomial
158  *
159  * In the first case, we return true and the term to substitute with.
160  * In the second case, we return false and the new minimal polynomial.
161  */
162  std::pair<bool,Poly> extend(Variable v, const real_algebraic_number_interval<Rational>& r) {
163  mModel.emplace(v, r);
164  if (r.is_numeric()) {
165  CARL_LOG_DEBUG("carl.fieldext", "Is numeric: " << v << " -> " << r);
166  return std::make_pair(true, Poly(r.value()));
167  }
168  auto ci = buildPolyRing(v);
169  CoCoA::RingElem p = cc.convertUV(replace_main_variable(r.polynomial(), v), ci);
170  CARL_LOG_DEBUG("carl.fieldext", "Factorization of " << p << " on " << ci);
171  auto factorization = CoCoA::factor(p);
172  CARL_LOG_DEBUG("carl.fieldext", "-> " << factorization);
173  for (const auto& f: factorization.myFactors()) {
174  if (evaluatesToZero(f)) {
175  CARL_LOG_DEBUG("carl.fieldext", "Factor " << f << " is zero in assignment.");
176  if (CoCoA::deg(f) == 1) {
177  auto cf =-(f -CoCoA::LF(f)) / CoCoA::CoeffEmbeddingHom(CoCoA::owner(f))(CoCoA::LC(f));
178  return std::make_pair(true, cc.convertMV<Poly>(cf));
179  } else {
180  extendRing(ci, f);
181  return std::make_pair(false, cc.convertMV<Poly>(f));
182  }
183  }
184  }
185  CARL_LOG_ERROR("carl.fieldext", "No factor is zero in assignment.");
186  assert(false);
187  return std::make_pair(false, Poly());
188  }
189 
190  Poly embed(const Poly& poly) { // TODO not functional yet
191  CARL_LOG_DEBUG("carl.fieldext", "Embed " << poly << " into " << mQ);
192  auto f = cc.convertMV(poly, mQ);
193  CARL_LOG_DEBUG("carl.fieldext", "Embedding is " << f);
194  return cc.convertMV<Poly>(f);
195  }

85  break;

110  * Substitutes all variables from a model within a formula.
111  * May fail to substitute some variables, for example if the values are RANs or SqrtEx.
112  */
113  template<typename Rational, typename Poly>
114  void substituteIn(Formula<Poly>& f, const Model<Rational,Poly>& m) {
115  switch (f.getType()) {
116  case FormulaType::ITE: {
117  f = Formula<Poly>(FormulaType::ITE, substitute(f.condition(), m), substitute(f.firstCase(), m), substitute(f.secondCase(), m));
118  break;
119  }
120  case FormulaType::EXISTS:
121  CARL_LOG_WARN("carl.model.evaluation", "Evaluation of exists not yet implemented.");
122  break;
123  case FormulaType::FORALL:
124  CARL_LOG_WARN("carl.model.evaluation", "Evaluation of forall not yet implemented.");
125  break;
126  case FormulaType::TRUE: break;
127  case FormulaType::FALSE: break;
128  case FormulaType::BOOL: {
129  auto it = m.find(f.boolean());
130  if (it == m.end()) break;
131  if (it->second.isBool()) {
132  f = Formula<Poly>(it->second.asBool() ? FormulaType::TRUE : FormulaType::FALSE);
133  } else {
134  CARL_LOG_WARN("carl.model.evaluation", "Could not evaluate " << it->first << " as a boolean, value is " << it->second);
135  }
136  break;
137  }
138  case FormulaType::NOT: {
139  f = Formula<Poly>(FormulaType::NOT, substitute(f.subformula(), m));
140  break;
141  }
142  case FormulaType::IMPLIES: {
143  f = Formula<Poly>(FormulaType::IMPLIES, substitute(f.premise(), m), substitute(f.conclusion(), m));
144  break;
145  }
146  case FormulaType::AND: substituteSubformulas(f, m); break;
147  case FormulaType::OR: substituteSubformulas(f, m); break;
148  case FormulaType::XOR: substituteSubformulas(f, m); break;
149  case FormulaType::IFF: substituteSubformulas(f, m); break;
150  case FormulaType::CONSTRAINT: {
151  auto res = evaluate(f.constraint(), m);
152  if (res.isBool()) {
153  if (res.asBool()) f = Formula<Poly>(FormulaType::TRUE);
154  else f = Formula<Poly>(FormulaType::FALSE);
155  } else {
156  assert(res.isSubstitution());
157  const auto& subs = res.asSubstitution();
158  auto fsubs = static_cast<ModelFormulaSubstitution<Rational,Poly>*>(subs.get());
159  f = fsubs->getFormula();
160  }
161  break;
162  }
163  case FormulaType::VARCOMPARE: {
164  evaluateVarCompare(f, m);
165  break;
166  }
167  case FormulaType::VARASSIGN: {
168  evaluateVarAssign(f, m);
169  break;
170  }
171  case FormulaType::BITVECTOR: {
172  BVConstraint bvc = substitute(f.bvConstraint(), m);
173  if (bvc.isAlwaysConsistent()) f = Formula<Poly>(FormulaType::TRUE);
174  else if (bvc.isAlwaysInconsistent()) f = Formula<Poly>(FormulaType::FALSE);
175  else f = Formula<Poly>(bvc);
176  break;
177  }
178  case FormulaType::UEQ: {
179  std::set<UVariable> vars;
180  f.uequality().gatherUVariables(vars);
181  if (m.contains(vars)) {
182  auto val = evaluate(f.uequality(), m);
183  assert(val.isBool());
184  f = Formula<Poly>(val.asBool() ? FormulaType::TRUE : FormulaType::FALSE);
185  } else {
186  CARL_LOG_WARN("carl.model.evaluation", "Could not evaluate " << f.uequality() << " as some variables are missing from the model.");
187  }
188  break;
189  }
190  }
191  }

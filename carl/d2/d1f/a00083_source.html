45  template<typename Coeff, EnableIf<is_subset_of_rationals<Coeff>> = dummy>
46  poly::Polynomial toLibpolyPolynomial(const carl::MultivariatePolynomial<Coeff>& p, mpz_class& denominator) {
47  CARL_LOG_DEBUG("carl.converter", "Converting Carl Multivariate Poly " << p);
48  if (carl::is_constant(p)) {
49  CARL_LOG_DEBUG("carl.converter", "Poly is constant");
50  denominator = carl::getDenom(p.constantPart());
51  auto res = poly::Polynomial(poly::Integer(carl::getNum(p.constantPart())));
52  return res;
53  }
54  //Libpoly polynomials have integer coefficients -> so we have to store the lcm of the coefficients of every term
55  Coeff coprimeFactor = p.mainDenom();
56  if (carl::getDenom(coprimeFactor) != 1) {
57  //if coprime factor is not an integer
58  denominator = coprimeFactor > 0 ? mpz_class(1) : mpz_class(-1);
59  } else if (coprimeFactor == 0) {
60  //Bug? This case is just to be safe
61  denominator = mpz_class(1);
62  } else {
63  denominator = mpz_class(coprimeFactor);
64  }
65  CARL_LOG_DEBUG("carl.converter", "Coprime Factor/ Denominator: " << denominator);
66  poly::Polynomial res(poly::Integer(0));
67  //iterate over terms
68  for (const carl::Term<Coeff>& term : p) {
69  //Multiply by denominator to make the coefficient an integer
70  poly::Polynomial t(poly::Integer(mpz_class(term.coeff() * denominator)));
71  //iterate over monomial
72  if (term.monomial()) {
73  //A monomial is a vector of pairs <variable, power>
74  for (const std::pair<carl::Variable, std::size_t>& var_pow : (*term.monomial()).exponents()) {
75  //multiply 1*var^pow
76  t *= poly::Polynomial(poly::Integer(1), VariableMapper::getInstance().getLibpolyVariable(var_pow.first), (unsigned)var_pow.second);
77  }
78  }
79  res += t;
80  }
81  CARL_LOG_DEBUG("carl.converter", "Got Polynomial: " << res);
82  return res;
83  }

164  template<class Settings>
165  void UnionFindModule<Settings>::propagate_induces_equalities(
166  const std::unordered_set<carl::UVariable>& vars) noexcept
167  {
168  auto induced = [&] (auto const& lhs, auto const& rhs) -> bool {
169  return (classes.has_variable(lhs) && classes.has_variable(rhs))
170  && (classes.find(lhs) == classes.find(rhs));
171  };
172 
173  for (const auto& ueq : informed) {
174  const auto& lhs = ueq.lhs().asUVariable();
175  const auto& rhs = ueq.rhs().asUVariable();
176 
177  if (vars.count(lhs) && vars.count(rhs) && induced(lhs, rhs)) {
178  if (auto path = graph.get_path(lhs, rhs, Settings::lemma_length_bound)) {
179  FormulasT eqs;
180 
181  for (const auto& [u, v]: path.value()) {
182  eqs.emplace_back(u, v, true);
183  }
184 
185  eqs.emplace_back(ueq);
186 
187  using carl::FormulaType;
188  addLemma(FormulaT{ FormulaType::OR, eqs });
189  }
190  }
191  }

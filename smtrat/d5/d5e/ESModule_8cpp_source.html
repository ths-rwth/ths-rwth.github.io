23  template<class Settings>
24  void ESModule<Settings>::updateModel() const
25  {
26  clearModel();
27  if( solverState() == SAT || (solverState() != UNSAT && appliedPreprocessing()) )
28  {
29  getBackendsModel();
30  for( const auto& iter : mBoolSubs )
31  {
32  if( iter.first.getType() == carl::FormulaType::BOOL )
33  {
34  assert( mModel.find( iter.first.boolean() ) == mModel.end() );
35  mModel.emplace( iter.first.boolean(), iter.second );
36  }
37  }
38  for( const auto& iter : mArithSubs )
39  {
40  assert( mModel.find( iter.first ) == mModel.end() );
41  mModel.emplace( iter.first, carl::createSubstitution<Rational,Poly,ModelPolynomialSubstitution>( iter.second ) );
42  }
43  // All variables which occur in the root of the constructed state tree but were incidentally eliminated
44  // (during the elimination of another variable) can have an arbitrary assignment. If the variable has the
45  // real domain, we leave at as a parameter, and, if it has the integer domain we assign 0 to it.
46  carl::carlVariables _receivedVars;
47  rReceivedFormula().gatherVariables(_receivedVars);
48  auto receivedVars = _receivedVars.as_set();
49  if( solverState() != SAT && appliedPreprocessing() )
50  {
51  carl::carlVariables passedVars;
52  rPassedFormula().gatherVariables(passedVars);
53  auto rvIter = receivedVars.begin();
54  auto pvIter = passedVars.begin();
55  while( rvIter != receivedVars.end() && pvIter != passedVars.end() )
56  {
57  if( *rvIter < *pvIter )
58  ++rvIter;
59  else if( *pvIter < *rvIter )
60  ++pvIter;
61  else
62  {
63  rvIter = receivedVars.erase( rvIter );
64  ++pvIter;
65  }
66  }
67  }
68  for( auto var : receivedVars )
69  {
70  if( var.type() == carl::VariableType::VT_BOOL )
71  mModel.insert(std::make_pair(var, false));
72  else
73  mModel.insert(std::make_pair(var, carl::createSubstitution<Rational,Poly,ModelPolynomialSubstitution>(Poly())));
74  }
75  }
76  }

78  template<class Settings>
79  Answer ESModule<Settings>::checkCore()
80  {
81  mBoolSubs.clear();
82  mArithSubs.clear();
83  FormulaT formula = elimSubstitutions( (FormulaT) rReceivedFormula(), true, true );
84  Answer ans = SAT;
85  if( formula.isFalse() )
86  ans = UNSAT;
87  else
88  {
89  addSubformulaToPassedFormula( formula );
90  ans = runBackends();
91  }
92  if( ans == UNSAT )
93  generateTrivialInfeasibleSubset(); // TODO: compute a better infeasible subset
94  return ans;
95  }

97  template<typename Settings>
98  FormulaT ESModule<Settings>::elimSubstitutions( const FormulaT& _formula, bool _elimSubstitutions, bool _outermost )
99  {
100 
101  auto iter = mBoolSubs.find( _formula );
102  if( iter != mBoolSubs.end() )
103  {
104  SMTRAT_LOG_DEBUG("smtrat.es", _formula << " ----> " << (iter->second ? FormulaT( carl::FormulaType::TRUE ) : FormulaT( carl::FormulaType::FALSE )));
105  return iter->second ? FormulaT( carl::FormulaType::TRUE ) : FormulaT( carl::FormulaType::FALSE );
106  }
107  FormulaT result = _formula;
108  switch( _formula.getType() )
109  {
110  case carl::FormulaType::AND:
111  {
112  std::vector<std::map<carl::Variable,Poly>::const_iterator> addedArithSubs;
113  std::unordered_map<FormulaT,std::unordered_map<FormulaT, bool>::const_iterator> foundBooleanSubstitutions;
114  bool foundNewSubstitution = true;
115  // we use sets, as we want the sub-formulas to be sorted according to their IDs
116  // a formula has always a greater id than its formulas (and, hence, their sub-formulas etc)
117  // then, for instance, for (and b (or a b)) we would first see that b->true and afterwards
118  // replace b for true in (or a b) as it has a higher ID
119  FormulaSetT foundSubstitutions;
120  FormulaSetT currentSubformulas;
121  for( const FormulaT& subf : result.subformulas() )
122  currentSubformulas.insert( currentSubformulas.end(), subf ); // as sub-formulas are already sorted use hint
123  while( foundNewSubstitution )
124  {
125  FormulaSetT sfs;
126  foundNewSubstitution = false;
127  // Process all equations first.
128  for( const auto& sf : currentSubformulas )
129  {
130  if( sf.getType() == carl::FormulaType::CONSTRAINT && sf.constraint().relation() == carl::Relation::EQ )
131  {
132  FormulaT tmp = elimSubstitutions( sf );
133  if( tmp.getType() == carl::FormulaType::FALSE )
134  {
135  result = tmp;
136  goto Return;
137  }
138  else if( tmp.getType() != carl::FormulaType::TRUE )
139  {
140  carl::Variable subVar;
141  Poly subPoly;
142  if( tmp.constraint().getSubstitution( subVar, subPoly, false, objective() ) )
143  {
144  if (subVar != objective()) {
145  SMTRAT_LOG_INFO("smtrat.es", "found substitution [" << subVar << " -> " << subPoly << "]");
146  assert( mArithSubs.find( subVar ) == mArithSubs.end() );
147  addedArithSubs.push_back( mArithSubs.emplace( subVar, subPoly ).first );
148  foundSubstitutions.insert( tmp );
149  foundNewSubstitution = true;
150  }
151  }
152  else
153  {
154  sfs.insert( sfs.end(), tmp );
155  }
156  }
157  }
158  }
159  // Now the other sub-formulas.
160  for( const auto& sf : currentSubformulas )
161  {
162  if( sf.getType() != carl::FormulaType::CONSTRAINT || sf.constraint().relation() != carl::Relation::EQ || !sf.constraint().lhs().isLinear() )
163  {
164  auto iterC = foundBooleanSubstitutions.find( sf );
165  if( iterC != foundBooleanSubstitutions.end() )
166  {
167  mBoolSubs.erase( iterC->second );
168  foundBooleanSubstitutions.erase( iterC );
169  }
170  FormulaT sfSimplified = elimSubstitutions( sf );
171  if( sfSimplified.isFalse() )
172  {
173  result = sfSimplified;
174  goto Return;
175  }
176  else if( !sfSimplified.isTrue() )
177  {
178  if( sf != sfSimplified )
179  {
180  foundNewSubstitution = true;
181  if( sfSimplified.getType() == carl::FormulaType::AND )
182  sfs.insert( sfSimplified.subformulas().begin(), sfSimplified.subformulas().end() );
183  else
184  sfs.insert( sfs.end(), sfSimplified );
185  }
186  else
187  {
188  if( !(_outermost && sfSimplified.isLiteral() && sfSimplified.isOnlyPropositional()) )
189  sfs.insert( sfs.end(), sfSimplified );
190  if( sfSimplified.getType() == carl::FormulaType::NOT )
191  {
192  SMTRAT_LOG_TRACE("smtrat.es", "found boolean substitution [" << sfSimplified.subformula() << " -> false]");
193  assert( mBoolSubs.find( sfSimplified.subformula() ) == mBoolSubs.end() );
194  assert( foundBooleanSubstitutions.find( sfSimplified ) == foundBooleanSubstitutions.end() );
195  foundBooleanSubstitutions.emplace( sfSimplified, mBoolSubs.insert( std::make_pair( sfSimplified.subformula(), false ) ).first );
196  }
197  else
198  {
199  SMTRAT_LOG_TRACE("smtrat.es", "found boolean substitution [" << sfSimplified << " -> true]");
200  assert( mBoolSubs.find( sfSimplified ) == mBoolSubs.end() );
201  assert( foundBooleanSubstitutions.find( sfSimplified ) == foundBooleanSubstitutions.end() );
202  foundBooleanSubstitutions.emplace( sfSimplified, mBoolSubs.insert( std::make_pair( sfSimplified, true ) ).first );
203  }
204  }
205  }
206  }
207  }
208  currentSubformulas = std::move(sfs);
209  }
210  if( currentSubformulas.empty() )
211  {
212  if( foundSubstitutions.empty() )
213  result = FormulaT( carl::FormulaType::TRUE );
214  else if( !_elimSubstitutions )
215  result = FormulaT( carl::FormulaType::AND, std::move(foundSubstitutions) );
216  }
217  else
218  {
219  if( !_elimSubstitutions )
220  currentSubformulas.insert( foundSubstitutions.begin(), foundSubstitutions.end() );
221  result = FormulaT( carl::FormulaType::AND, std::move(currentSubformulas) );
222  }
223  Return:
224  if( !_outermost )
225  {
226  while( !addedArithSubs.empty() )
227  {
228  assert( std::count( addedArithSubs.begin(), addedArithSubs.end(), addedArithSubs.back() ) == 1 );
229  mArithSubs.erase( addedArithSubs.back() );
230  addedArithSubs.pop_back();
231  }
232  while( !foundBooleanSubstitutions.empty() )
233  {
234  mBoolSubs.erase( foundBooleanSubstitutions.begin()->second );
235  foundBooleanSubstitutions.erase( foundBooleanSubstitutions.begin() );
236  }
237  }
238  break;
239  }
240  case carl::FormulaType::ITE:
241  {
242  FormulaT cond = elimSubstitutions( _formula.condition() );
243  if( cond.getType() == carl::FormulaType::CONSTRAINT )
244  {
245  carl::Variable subVar;
246  Poly subPoly;
247  if( cond.constraint().getSubstitution( subVar, subPoly, false ) )
248  {
249  SMTRAT_LOG_DEBUG("smtrat.es", __LINE__ << " found substitution [" << subVar << " -> " << subPoly << "]" );
250  auto addedBoolSub = cond.getType() == carl::FormulaType::NOT ? mBoolSubs.emplace( cond.subformula(), false ) : mBoolSubs.emplace( cond, true );
251  SMTRAT_LOG_DEBUG("smtrat.es", __LINE__ << " found boolean substitution [" << addedBoolSub.first->first << " -> " << (addedBoolSub.first->second ? "true" : "false") << "]");
252  assert( addedBoolSub.second );
253  auto iterB = mArithSubs.emplace( subVar, subPoly ).first;
254  FormulaT firstCaseTmp = elimSubstitutions( _formula.firstCase() );
255  mArithSubs.erase( iterB );
256  mBoolSubs.erase( addedBoolSub.first );
257  addedBoolSub = cond.getType() == carl::FormulaType::NOT ? mBoolSubs.emplace( cond.subformula(), true ) : mBoolSubs.emplace( cond, false );
258  SMTRAT_LOG_DEBUG("smtrat.es", __LINE__ << " found boolean substitution [" << addedBoolSub.first->first << " -> " << (addedBoolSub.first->second ? "true" : "false") << "]" );
259  assert( addedBoolSub.second );
260  FormulaT secondCaseTmp = elimSubstitutions( _formula.secondCase() );
261  mBoolSubs.erase( addedBoolSub.first );
262  result = FormulaT( carl::FormulaType::ITE, {cond, firstCaseTmp, secondCaseTmp} );
263  break;
264  }
265  else if( cond.constraint().getSubstitution( subVar, subPoly, true ) )
266  {
267  SMTRAT_LOG_DEBUG("smtrat.es", __LINE__ << " found substitution [" << subVar << " -> " << subPoly << "]" );
268  auto addedBoolSub = cond.getType() == carl::FormulaType::NOT ? mBoolSubs.emplace( cond.subformula(), false ) : mBoolSubs.emplace( cond, true );
269  SMTRAT_LOG_DEBUG("smtrat.es", __LINE__ << " found boolean substitution [" << addedBoolSub.first->first << " -> " << (addedBoolSub.first->second ? "true" : "false") << "]" );
270  assert( addedBoolSub.second );
271  FormulaT firstCaseTmp = elimSubstitutions( _formula.firstCase() );
272  mBoolSubs.erase( addedBoolSub.first );
273  addedBoolSub = cond.getType() == carl::FormulaType::NOT ? mBoolSubs.emplace( cond.subformula(), true ) : mBoolSubs.emplace( cond, false );
274  SMTRAT_LOG_DEBUG("smtrat.es", __LINE__ << " found boolean substitution [" << addedBoolSub.first->first << " -> " << (addedBoolSub.first->second ? "true" : "false") << "]" );
275  assert( addedBoolSub.second );
276  auto iterB = mArithSubs.emplace( subVar, subPoly ).first;
277  FormulaT secondCaseTmp = elimSubstitutions( _formula.secondCase() );
278  mArithSubs.erase( iterB );
279  mBoolSubs.erase( addedBoolSub.first );
280  result = FormulaT( carl::FormulaType::ITE, {cond, firstCaseTmp, secondCaseTmp} );
281  break;
282  }
283  }
284  if( cond.isTrue() )
285  {
286  result = elimSubstitutions( _formula.firstCase() );
287  }
288  else if( cond.isFalse() )
289  {
290  result = elimSubstitutions( _formula.secondCase() );
291  }
292  else
293  {
294  auto addedBoolSub = cond.getType() == carl::FormulaType::NOT ? mBoolSubs.emplace( cond.subformula(), false ) : mBoolSubs.emplace( cond, true );
295  SMTRAT_LOG_DEBUG("smtrat.es", __LINE__ << " found boolean substitution [" << addedBoolSub.first->first << " -> " << (addedBoolSub.first->second ? "true" : "false") << "]");
296  assert( addedBoolSub.second );
297  FormulaT firstCaseTmp = elimSubstitutions( _formula.firstCase() );
298  mBoolSubs.erase( addedBoolSub.first );
299  addedBoolSub = cond.getType() == carl::FormulaType::NOT ? mBoolSubs.emplace( cond.subformula(), true ) : mBoolSubs.emplace( cond, false );
300  SMTRAT_LOG_DEBUG("smtrat.es", __LINE__ << " found boolean substitution [" << addedBoolSub.first->first << " -> " << (addedBoolSub.first->second ? "true" : "false") << "]");
301  assert( addedBoolSub.second );
302  FormulaT secondCaseTmp = elimSubstitutions( _formula.secondCase() );
303  mBoolSubs.erase( addedBoolSub.first );
304  result = FormulaT( carl::FormulaType::ITE, {cond, firstCaseTmp, secondCaseTmp} );
305  }
306  break;
307  }
308  case carl::FormulaType::OR:
309  case carl::FormulaType::IFF:
310  case carl::FormulaType::XOR: {
311  FormulasT newSubformulas;
312  bool changed = false;
313  for (const auto& cur: _formula.subformulas()) {
314  FormulaT newCur = elimSubstitutions(cur);
315  if (newCur != cur) changed = true;
316  newSubformulas.push_back(newCur);
317  }
318  if (changed)
319  result = FormulaT(_formula.getType(), std::move(newSubformulas));
320  break;
321  }
322  case carl::FormulaType::NOT: {
323  FormulaT cur = elimSubstitutions(_formula.subformula());
324  if (cur != _formula.subformula())
325  result = FormulaT(carl::FormulaType::NOT, cur);
326  break;
327  }
328  case carl::FormulaType::IMPLIES: {
329  FormulaT prem = elimSubstitutions(_formula.premise());
330  FormulaT conc = elimSubstitutions(_formula.conclusion());
331  if ((prem != _formula.premise()) || (conc != _formula.conclusion()))
332  result = FormulaT(carl::FormulaType::IMPLIES, {prem, conc});
333  break;
334  }
335  case carl::FormulaType::CONSTRAINT: {
336  FormulaT tmp = result;
337  while( result != (tmp = tmp.substitute( mArithSubs )) )
338  result = tmp;
339  break;
340  }
341  case carl::FormulaType::BOOL:
342  case carl::FormulaType::BITVECTOR:
343  case carl::FormulaType::UEQ:
344  case carl::FormulaType::TRUE:
345  case carl::FormulaType::FALSE:
346  if (_formula != result) {
347  SMTRAT_LOG_DEBUG("smtrat.es", _formula << " ----> " << result);
348  }
349  return result;
350  case carl::FormulaType::EXISTS:
351  case carl::FormulaType::FORALL: {
352  FormulaT sub = elimSubstitutions(_formula.quantifiedFormula());
353  if (sub != _formula.quantifiedFormula())
354  result = FormulaT(_formula.getType(), _formula.quantifiedVariables(), sub);
355  }
356  default: {}
357  }
358  if (!_outermost) {
359  iter = mBoolSubs.find( result );
360  if( iter != mBoolSubs.end() )
361  {
362  SMTRAT_LOG_DEBUG("smtrat.es", _formula << " ----> " << (iter->second ? FormulaT( carl::FormulaType::TRUE ) : FormulaT( carl::FormulaType::FALSE )));
363  return iter->second ? FormulaT( carl::FormulaType::TRUE ) : FormulaT( carl::FormulaType::FALSE );
364  }
365  }
366  if (_formula != result) {
367  SMTRAT_LOG_DEBUG("smtrat.es", _formula << " ----> " << result);
368  }
369  return result;
370  }
371 }

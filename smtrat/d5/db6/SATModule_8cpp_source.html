197  {
198  mModelComputed = false;

200  mInfeasibleSubsets.emplace_back();
201  mInfeasibleSubsets.back().insert( _subformula->formula() );
202  return false;
203  }
204  else if( !_subformula->formula().isTrue() )
205  {
206  if( !_subformula->formula().isOnlyPropositional() )
207  mReceivedFormulaPurelyPropositional = false;
208  mModelComputed = false;
209  mOptimumComputed = false;
210  //TODO Matthias: better solution?
211  cancelUntil(0, true);
212  adaptPassedFormula();
213  if( _subformula->formula().propertyHolds( carl::PROP_IS_A_LITERAL ) )
214  {
215  assumptions.push( createLiteral( _subformula->formula(), _subformula->formula() ) );
216  assert( mFormulaAssumptionMap.find( _subformula->formula() ) == mFormulaAssumptionMap.end() );
217  mFormulaAssumptionMap.emplace( _subformula->formula(), assumptions.last() );
218  }
219  else
220  {
221  addClauses( _subformula->formula(), NORMAL_CLAUSE, 0, _subformula->formula() );
222  }
223  if ( isLemmaLevel(NORMAL) && decisionLevel() == 0)
224  {
225  if (_subformula->formula().propertyHolds(carl::PROP_IS_A_LITERAL) && _subformula->formula().propertyHolds(carl::PROP_CONTAINS_BOOLEAN))
226  {
227  // Add literal from unary clause to lemmas
228  carl::carlVariables vars = carl::boolean_variables(_subformula->formula());
229  assert(vars.size() == 1);
230  // Get corresponding Minisat variable
231  BooleanVarMap::const_iterator itVar = mBooleanVarMap.find(*vars.begin());
232  assert(itVar != mBooleanVarMap.end());
233  Minisat::Var var = itVar->second;
234  // Insert new propagated lemma
235  mPropagatedLemmas[var].push_back(_subformula->formula());
236  }
237  }
238  }
239  if( !ok )
240  updateInfeasibleSubset();
241  return ok;

244  template<class Settings>
245  void SATModule<Settings>::removeCore( ModuleInput::const_iterator _subformula )
246  {
247  if( _subformula->formula().isFalse() || _subformula->formula().isTrue() )
248  return;
249  cancelUntil( 0, true ); // can we do better than this?
250  if( !mReceivedFormulaPurelyPropositional )
251  adaptPassedFormula();
252  assert( rPassedFormula().empty() );
253  for( int i = 0; i < learnts.size(); ++i )
254  {
255  assert( learnts[i] != CRef_Undef );
256  removeClause( learnts[i] );
257  }
258  learnts.clear();
259  mUnorderedClauseLookup.clear();
260  ok = true;
261  if( _subformula->formula().propertyHolds( carl::PROP_IS_A_LITERAL ) )
262  {
263  auto iter = mFormulaAssumptionMap.find( _subformula->formula() );
264  assert( iter != mFormulaAssumptionMap.end() );
265  int i = 0;
266  while( assumptions[i] != iter->second ) ++i;
267  while( i < assumptions.size() - 1 )
268  {
269  assumptions[i] = assumptions[i+1];
270  ++i;
271  }
272  assumptions.pop();
273  mFormulaAssumptionMap.erase( iter );
274  ConstraintLiteralsMap::iterator constraintLiteralPair = mConstraintLiteralMap.find( _subformula->formula() );
275  if( constraintLiteralPair != mConstraintLiteralMap.end() )
276  removeLiteralOrigin( constraintLiteralPair->second.front(), _subformula->formula() );
277  }
278  else
279  {
280  auto cnfInfoIter = mFormulaCNFInfosMap.find( _subformula->formula().removeNegations() );
281  assert( cnfInfoIter != mFormulaCNFInfosMap.end() );
282  updateCNFInfoCounter( cnfInfoIter, _subformula->formula(), false );
283  if( cnfInfoIter->second.mClauses.empty() )
284  {
285  mFormulaCNFInfosMap.erase( cnfInfoIter );
286  }
287  std::vector<FormulaT> constraints;
288  _subformula->formula().getConstraints( constraints );
289  for( const FormulaT& constraint : constraints )
290  {
291  ConstraintLiteralsMap::iterator constraintLiteralPair = mConstraintLiteralMap.find( constraint );
292  if( constraintLiteralPair != mConstraintLiteralMap.end() )
293  removeLiteralOrigin( constraintLiteralPair->second.front(), _subformula->formula() );
294  constraintLiteralPair = mConstraintLiteralMap.find( constraint.negated() );
295  if( constraintLiteralPair != mConstraintLiteralMap.end() )
296  removeLiteralOrigin( constraintLiteralPair->second.front(), _subformula->formula() );
297  }
298  }

315  {
316  if (iter != --origs.end())
317  {
318  *iter = origs.back();
319  origs.pop_back();
320  }
321  else
322  {
323  origs.pop_back();
324  break;

326  }
327  else

339  template<class Settings>
340  double SATModule<Settings>::luby( double y, int x )
341  {
342  // Find the finite subsequence that contains index 'x', and the
343  // size of that subsequence:
344  int size, seq;
345  for( size = 1, seq = 0; size < x + 1; seq++, size = 2 * size + 1 );
346 
347  while( size - 1 != x )
348  {
349  size = (size - 1) >> 1;
350  seq--;
351  x = x % size;
352  }
353 
354  return pow( y, seq );
355  }

357  template<class Settings>
358  Answer SATModule<Settings>::checkCore()
359  {
360  //for( const auto& f : rReceivedFormula() )
361  // std::cout << " " << f.formula() << std::endl;
362 // std::cout << ((FormulaT) rReceivedFormula()) << std::endl;
363  #ifdef SMTRAT_DEVOPTION_Statistics
364  mStatistics.rNrTotalVariablesBefore() = (size_t) nVars();
365  mStatistics.rNrClauses() = (size_t) nClauses();
366  #endif
367  ScopedBool scopedBool( mBusy, true );
368  budgetOff();
369 // assumptions.clear();
370  Module::init();
371  processLemmas();
372 
373  if (Settings::mc_sat) {
374  assert(Settings::use_new_var_scheduler);
375  #ifdef DEBUG_SATMODULE
376  std::cout << "### Processing clause" << std::endl;
377  print(std::cout, "###");
378  #endif
379  mMCSAT.initVariables(mBooleanConstraintMap);
380  for (const auto& v : mMCSAT.theoryVarAbstractions()) {
381  var_scheduler.insert(v);
382  }
383  assert(mMCSAT.level() == 0);
384  if (var_scheduler.empty()) {
385  SMTRAT_LOG_DEBUG("smtrat.sat", "We have no variables in our variable scheduler.");
386  } else {
387  var_scheduler.rebuildTheoryVars(mBooleanConstraintMap);
388  }
389  }
390  ++solves;
391  // compared to original minisat we add the number of clauses with size 1 (nAssigns()) and learnts, we got after init()
392  max_learnts = (nAssigns() + nClauses() + nLearnts() ) * learntsize_factor;
393  learntsize_adjust_confl = learntsize_adjust_start_confl;
394  learntsize_adjust_cnt = (int)learntsize_adjust_confl;
395 
396  if( !ok )
397  {
398  assert( !mInfeasibleSubsets.empty() );
399  #ifdef SMTRAT_DEVOPTION_Statistics
400  collectStats();
401  #endif
402  mBusy = false;
403  return UNSAT;
404  }
405  mReceivedFormulaPurelyPropositional = rReceivedFormula().isOnlyPropositional();
406  if( mReceivedFormulaPurelyPropositional )
407  {
408  mAllActivitiesChanged = false;
409  mChangedBooleans.clear();
410  mChangedActivities.clear();
411  }
412  else if( Settings::initiate_activities )
413  {
414  double highestActivity = 0;
415  for( int pos = 0; pos < activity.size(); ++pos )
416  {
417  double& act = activity[pos];
418  act = 1;
419  if( Settings::check_active_literal_occurrences && false )
420  {
421  const auto& litActOccs = mLiteralsActivOccurrences[(size_t)pos];
422  act = (double)litActOccs.first + (double)litActOccs.second;
423  }
424  if( mBooleanConstraintMap[pos].first != nullptr )
425  {
426  act /= (double)mBooleanConstraintMap[pos].first->reabstraction.complexity();
427  }
428  else
429  {
430  auto tvfIter = mTseitinVarFormulaMap.find( pos );
431  if( tvfIter != mTseitinVarFormulaMap.end() )
432  act /= (double)tvfIter->second.complexity();
433  }
434  if( act > highestActivity )
435  highestActivity = act;
436  }
437  if( highestActivity > 1 )
438  {
439  for( int pos = 0; pos < activity.size(); ++pos )
440  {
441  activity[pos] /= highestActivity;
442  }
443  }
444  if (Settings::use_new_var_scheduler) {
445  var_scheduler.rebuildActivities();
446  } else {
447  rebuildOrderHeap();
448  }
449  }
450  Minisat::lbool result = l_Undef;
451  mUpperBoundOnMinimal = passedFormulaEnd();
452  bool isOptimal = true;
453  while( true )
454  {
455  if( Settings::use_restarts )
456  {
457  mCurr_Restarts = 0;
458  int current_restarts = -1;
459  while( current_restarts < mCurr_Restarts )
460  {
461  current_restarts = mCurr_Restarts;
462  double rest_base = luby_restart ? luby( restart_inc, mCurr_Restarts ) : pow( restart_inc, mCurr_Restarts );
463  result = search( (int)rest_base * restart_first );
464  // if( !withinBudget() ) break;
465  }
466  }
467  else
468  result = search();
469 
470  if (isLemmaLevel(LemmaLevel::ADVANCED))
471  {
472  assert(result == l_True);
473  computeAdvancedLemmas();
474  }
475  if( !Settings::stop_search_after_first_unknown )
476  {
477  unknown_excludes.clear();
478  mExcludedAssignments = false;
479  }
480  // ##### Stop here if not in optimization mode!
481  if( !is_minimizing() )
482  break;
483  std::vector<CRef> excludedAssignments;
484  if( result == l_Undef )
485  break;
486  else if( result == l_False )
487  {
488  if( mUpperBoundOnMinimal != rPassedFormula().end() )
489  {
490  cleanUpAfterOptimizing( excludedAssignments );
491  result = l_True;
492  }
493  break;
494  }
495  else
496  {
497  assert( result == l_True );
498  Answer runBackendAnswer = runBackends( true, mFullCheck, objective() );
499  assert(is_sat(runBackendAnswer));
500  isOptimal = isOptimal && (runBackendAnswer == OPTIMAL);
501  updateModel();
502  auto modelIter = mModel.find( objective() );
503  assert( modelIter != mModel.end() );
504  const ModelValue& mv = modelIter->second;
505  if( mv.isMinusInfinity() )
506  {
507  cleanUpAfterOptimizing( excludedAssignments );
508  break;
509  }
510  assert( mv.isRational() ); // @todo: how do we handle the other model value types?
511  // Add a new upper bound on the yet computed minimum
512  removeUpperBoundOnMinimal();
513  FormulaT newUpperBoundOnMinimal( objective() - mv.asRational(), carl::Relation::LESS );
514  addConstraintToInform_( newUpperBoundOnMinimal );
515  mUpperBoundOnMinimal = addSubformulaToPassedFormula( newUpperBoundOnMinimal, newUpperBoundOnMinimal ).first;
516  // Exclude the last theory call with a clause.
517  vec<Lit> excludeClause;
518  for( int k = 0; k < mBooleanConstraintMap.size(); ++k )
519  {
520  if( assigns[k] != l_Undef )
521  {
522  if( mBooleanConstraintMap[k].first != nullptr )
523  {
524  assert( mBooleanConstraintMap[k].second != nullptr );
525  const Abstraction& abstr = assigns[k] == l_False ? *mBooleanConstraintMap[k].second : *mBooleanConstraintMap[k].first;
526  if( !abstr.reabstraction.isTrue() && abstr.consistencyRelevant && (abstr.reabstraction.getType() == carl::FormulaType::UEQ || abstr.reabstraction.getType() == carl::FormulaType::BITVECTOR || abstr.reabstraction.constraint().isConsistent() != 1))
527  {
528  excludeClause.push( mkLit( k, assigns[k] != l_False ) );
529  }
530  }
531  }
532  }
533  addClause( excludeClause, PERMANENT_CLAUSE );
534  CRef confl = storeLemmas();
535  if( confl != CRef_Undef )
536  excludedAssignments.push_back( confl );
537  if( !ok || decisionLevel() <= assumptions.size() )
538  {
539  cleanUpAfterOptimizing( excludedAssignments );
540  break;
541  }
542  if( confl != CRef_Undef )
543  handleConflict( confl );
544  }
545  }
546 
547  #ifdef SMTRAT_DEVOPTION_Statistics
548  collectStats();
549  #endif
550  if( result == l_True )
551  {
552  return (is_minimizing() && isOptimal) ? OPTIMAL : SAT;
553  }
554  else if( result == l_False )
555  {
556  SMTRAT_LOG_DEBUG("smtrat.sat", "ok = false");
557  ok = false;
558  updateInfeasibleSubset();
559  return UNSAT;
560  }
561  return UNKNOWN;
562  }

894  template<class Settings>
895  void SATModule<Settings>::updateCNFInfoCounter( typename FormulaCNFInfosMap::iterator _iter, const FormulaT& _origin, bool _increment )
896  {
897  assert( _iter != mFormulaCNFInfosMap.end() );
898  if( _increment )
899  ++_iter->second.mCounter;
900  else
901  --_iter->second.mCounter;
902  for( std::size_t pos = 0; pos < _iter->second.mClauses.size(); )
903  {
904  Minisat::CRef cr = _iter->second.mClauses[pos];
905  assert( cr != CRef_Undef );
906  auto ciIter = mClauseInformation.find( cr );
907  assert( ciIter != mClauseInformation.end() );
908  if( _increment )
909  {
910  ciIter->second.addOrigin( _origin );
911  ++pos;
912  }
913  else
914  {
915  ciIter->second.removeOrigin( _origin );
916  // if the counter becomes zero, remove the clause
917  if( _iter->second.mCounter == 0 )
918  {
919  // remove this clause and each information we stored for it
920  assert( ciIter->second.mOrigins.size() == 0 );
921  vec<CRef>& cls = ciIter->second.mStoredInSatisfied ? satisfiedClauses : clauses;
922  if( ciIter->second.mPosition < cls.size() - 1 )
923  {
924  cls[ciIter->second.mPosition] = cls.last();
925  auto ciIterB = mClauseInformation.find( cls[ciIter->second.mPosition] );
926  assert( ciIterB != mClauseInformation.end() );
927  ciIterB->second.mPosition = ciIter->second.mPosition;
928  cls.pop();
929  }
930  else
931  {
932  cls.pop();
933  }
934  mClauseInformation.erase( ciIter );
935  if( Settings::check_if_all_clauses_are_satisfied )
936  {
937  const Clause& c = ca[cr];
938  for( int i = 0; i < c.size(); ++i )
939  mLiteralClausesMap[Minisat::toInt(c[i])].erase( cr );
940  }
941  removeClause( cr );
942  if( pos < _iter->second.mClauses.size() - 1 )
943  _iter->second.mClauses[pos] = _iter->second.mClauses.back();
944  _iter->second.mClauses.pop_back();
945  }
946  else
947  ++pos;
948  }
949  }
950  // Invoke this procedure recursively on all sub-formulas, which again contain sub-formulas
951  if( _iter->first.isNary() )
952  {
953  for( const FormulaT& formula : _iter->first.subformulas() )
954  {
955  if( formula.isNary() )
956  {
957  auto cnfInfoIter = mFormulaCNFInfosMap.find( formula.removeNegations() );
958  if( cnfInfoIter != mFormulaCNFInfosMap.end() )
959  {
960  updateCNFInfoCounter( cnfInfoIter, _origin, _increment );
961  if( !_increment && cnfInfoIter->second.mClauses.empty() )
962  {
963  mFormulaCNFInfosMap.erase( cnfInfoIter );
964  }
965  }
966  }
967  }
968  }

971  template<class Settings>
972  Lit SATModule<Settings>::addClauses( const FormulaT& _formula, unsigned _type, unsigned _depth, const FormulaT& _original )
973  {
974  SMTRAT_LOG_DEBUG("smtrat.sat", "Adding formula " << _formula);
975  assert( _type < 4 );
976  bool everythingDecisionRelevant = !Settings::formula_guided_decision_heuristic;
977  unsigned nextDepth = _depth+1;
978  switch( _formula.getType() )
979  {
980  case carl::FormulaType::TRUE:
981  case carl::FormulaType::FALSE:
982  assert( false );
983  break;
984  case carl::FormulaType::BOOL:
985  case carl::FormulaType::UEQ:
986  case carl::FormulaType::CONSTRAINT:
987  case carl::FormulaType::VARCOMPARE:
988  case carl::FormulaType::VARASSIGN:
989  case carl::FormulaType::BITVECTOR:
990  return createLiteral( _formula, _original, everythingDecisionRelevant || _depth <= 1 );
991  case carl::FormulaType::NOT:
992  {
993  SMTRAT_LOG_TRACE("smtrat.sat", "Adding a negation: " << _formula);
994  Lit l = lit_Undef;
995  if( _formula.isLiteral() )
996  {
997  l = createLiteral( _formula, _original, everythingDecisionRelevant || _depth <= 1 );
998  SMTRAT_LOG_TRACE("smtrat.sat", "It is a literal: " << l);
999  }
1000  else {
1001  l = neg( addClauses( _formula.subformula(), _type, nextDepth, _original ) );
1002  SMTRAT_LOG_TRACE("smtrat.sat", "It is not a literal, but now: " << l);
1003  }
1004  if( _depth == 0 )
1005  {
1006  SMTRAT_LOG_DEBUG("smtrat.sat", "Depth is zero");
1007  vec<Lit> c;
1008  c.push(l);
1009  addClause(c, LEMMA_CLAUSE);
1010  //assumptions.push( l );
1011  //assert( mFormulaAssumptionMap.find( _formula ) == mFormulaAssumptionMap.end() );
1012  //mFormulaAssumptionMap.emplace( _formula, assumptions.last() );
1013  return lit_Undef;
1014  }
1015  return l;
1016  }
1017  default:
1018  {
1019  auto cnfInfoIter = mFormulaCNFInfosMap.end();
1020  if( _type == NORMAL_CLAUSE )
1021  {
1022  cnfInfoIter = mFormulaCNFInfosMap.find( _formula );
1023  if( cnfInfoIter != mFormulaCNFInfosMap.end() )
1024  {
1025  updateCNFInfoCounter( cnfInfoIter, _original, true );
1026  SMTRAT_LOG_DEBUG("smtrat.sat", "Recovered literal for " << _original << ": " << cnfInfoIter->second.mLiteral);
1027  Lit l = cnfInfoIter->second.mLiteral;
1028  // If it was not assumed yet
1029  if (!mAssumedTseitinVariable.test(std::size_t(var(l)))) {
1030  SMTRAT_LOG_DEBUG("smtrat.sat", _original << " is not assumed yet");
1031  // If we have a top-level clause right now and it is already used somewhere else
1032  // Or there already was a top-level clause but it is not assumed yet
1033  if (
1034  (_depth == 0 && cnfInfoIter->second.mCounter > 1) ||
1035  mNonassumedTseitinVariable.test(std::size_t(var(l)))
1036  ) {
1037  SMTRAT_LOG_DEBUG("smtrat.sat", _original << " should be assumed, adding it to assumptions");
1038  /*
1039  * If this literal is a tseitin variable, it may belong to a top-level clause.
1040  * In this case, it is not eagerly added to the assumptions but only lazily when it is actually reused.
1041  * This is the case now. We backtrack to DL0 (+ assumptions.size()) and add it to the assumptions now.
1042  * This can only happen if a formula is added with some boolean structure, as only then the tseitin variable will be used.
1043  * Examples are addCore() or a lemma from a backend, in these cases it is safe to reset.
1044  * In particular this can not happen for infeasible subsets or conflict clauses, where a reset might not be safe.
1045  */
1046  cancelUntil(assumptions.size());
1047  assumptions.push(l);
1048  assert(mFormulaAssumptionMap.find(_formula) == mFormulaAssumptionMap.end());
1049  mFormulaAssumptionMap.emplace(_formula, assumptions.last());
1050  mNonassumedTseitinVariable.reset(std::size_t(var(l)));
1051  mAssumedTseitinVariable.set(std::size_t(var(l)));
1052  }
1053  }
1054  return l;
1055  }
1056  cnfInfoIter = mFormulaCNFInfosMap.emplace( _formula, CNFInfos() ).first;
1057  }
1058  vec<Lit> lits;
1059  FormulaT tsVar = carl::FormulaPool<Poly>::getInstance().createTseitinVar( _formula );
1060  Lit tsLit = createLiteral( tsVar, _original, everythingDecisionRelevant || _depth <= 1 );
1061  mTseitinVariable.set(static_cast<std::size_t>(var(tsLit)));
1062  if( _type == NORMAL_CLAUSE )
1063  cnfInfoIter->second.mLiteral = tsLit;
1064  switch( _formula.getType() )
1065  {
1066  case carl::FormulaType::ITE:
1067  {
1068  Lit condLit = addClauses( _formula.condition(), _type, nextDepth, _original );
1069  Lit negCondLit = _formula.condition().isLiteral() ? addClauses( _formula.condition().negated(), _type, nextDepth, _original ) : neg( condLit );
1070  Lit thenLit = addClauses( _formula.firstCase(), _type, nextDepth, _original );
1071  Lit elseLit = addClauses( _formula.secondCase(), _type, nextDepth, _original );
1072  if( _depth == 0 )
1073  {
1074  // (or -cond then)
1075  lits.push( negCondLit ); lits.push( thenLit ); addClause_( lits, _type, _original, cnfInfoIter );
1076  // (or cond else)
1077  lits.clear(); lits.push( condLit ); lits.push( elseLit ); addClause_( lits, _type, _original, cnfInfoIter );
1078  return lit_Undef;
1079  }
1080  Lit negThenLit = _formula.firstCase().isLiteral() ? addClauses( _formula.firstCase().negated(), _type, nextDepth, _original ) : neg( thenLit );
1081  Lit negElseLit = _formula.secondCase().isLiteral() ? addClauses( _formula.secondCase().negated(), _type, nextDepth, _original ) : neg( elseLit );
1082  if( !mReceivedFormulaPurelyPropositional && Settings::initiate_activities )
1083  {
1084  mTseitinVarFormulaMap.emplace( (int)var(tsLit), _formula );
1085  }
1086  if( !mReceivedFormulaPurelyPropositional && Settings::formula_guided_decision_heuristic )
1087  {
1088  mTseitinVarShadows.emplace( (signed)var(tsLit), std::vector<signed>{ (signed)var(condLit), (signed)var(thenLit), (signed)var(elseLit)} );
1089  }
1090  // (or ts -cond -then)
1091  lits.push( tsLit ); lits.push( negCondLit ); lits.push( negThenLit ); addClause_( lits, _type, _original, cnfInfoIter );
1092  // (or ts cond -else)
1093  lits.clear(); lits.push( tsLit ); lits.push( condLit ); lits.push( negElseLit ); addClause_( lits, _type, _original, cnfInfoIter );
1094  // (or -ts -cond then)
1095  lits.clear(); lits.push( neg( tsLit ) ); lits.push( negCondLit ); lits.push( thenLit ); addClause_( lits, _type, _original, cnfInfoIter );
1096  // (or -ts cond else)
1097  lits.clear(); lits.push( neg( tsLit ) ); lits.push( condLit ); lits.push( elseLit ); addClause_( lits, _type, _original, cnfInfoIter );
1098 
1099  return tsLit;
1100  }
1101  case carl::FormulaType::IMPLIES:
1102  {
1103  Lit premLit = addClauses( _formula.premise(), _type, nextDepth, _original );
1104  Lit negPremLit = _formula.premise().isLiteral() ? addClauses( _formula.premise().negated(), _type, nextDepth, _original ) : neg( premLit );
1105  Lit conLit = addClauses( _formula.conclusion(), _type, nextDepth, _original );
1106  if( _depth == 0 )
1107  {
1108  // (or -premise conclusion)
1109  lits.push( neg( premLit ) ); lits.push( conLit ); addClause_( lits, _type, _original, cnfInfoIter );
1110  return lit_Undef;
1111  }
1112  Lit negConLit = _formula.conclusion().isLiteral() ? addClauses( _formula.conclusion().negated(), _type, nextDepth, _original ) : neg( conLit );
1113  if( !mReceivedFormulaPurelyPropositional && Settings::initiate_activities )
1114  {
1115  mTseitinVarFormulaMap.emplace( (int)var(tsLit), _formula );
1116  }
1117  if( !mReceivedFormulaPurelyPropositional && Settings::formula_guided_decision_heuristic )
1118  {
1119  mTseitinVarShadows.emplace( (signed)var(tsLit), std::vector<signed>{ (signed)var(premLit), (signed)var(conLit)} );
1120  }
1121  // (or -ts -prem con)
1122  lits.push( neg( tsLit ) ); lits.push( negPremLit ); lits.push( conLit ); addClause_( lits, _type, _original, cnfInfoIter );
1123  // (or ts prem)
1124  lits.clear(); lits.push( tsLit ); lits.push( premLit ); addClause_( lits, _type, _original, cnfInfoIter );
1125  // (or ts -con)
1126  lits.clear(); lits.push( tsLit ); lits.push( negConLit ); addClause_( lits, _type, _original, cnfInfoIter );
1127  return tsLit;
1128  }
1129  case carl::FormulaType::OR:
1130  {
1131  for( const auto& sf : _formula.subformulas() )
1132  lits.push( addClauses( sf, _type, nextDepth, _original ) );
1133  if( _depth == 0 )
1134  {
1135  /*
1136  * This is a top-level clause. The full tseitin encoding would be:
1137  * ts and (or -ts a1 ... an) and (or ts -a1) ... (or ts -an)
1138  * However ts will become an assumption and thus -ts can be skipped and (or ts -ak) is satisfied anyway.
1139  * We therefore only add (or a1 .. an).
1140  * However if the formula is reused in a nested formula somewhere else, we need ts to be forced to true.
1141  * To avoid work (and because always doing that induces problems when adding infeasible subsets) we do this lazily.
1142  * We add ts to mNonassumedTseitinVariable and only add it to the assumptions when it is actually reused in another formula.
1143  */
1144  if (_type == NORMAL_CLAUSE) {
1145  SMTRAT_LOG_DEBUG("smtrat.sat", "top-level clause has new tseitin literal " << tsLit << ", marking it as non-assumed");
1146  assert(!mAssumedTseitinVariable.test(std::size_t(var(tsLit))));
1147  assert(cnfInfoIter->second.mCounter == 1);
1148  mNonassumedTseitinVariable.set(std::size_t(var(tsLit)));
1149  }
1150  addClause_( lits, _type, _original, cnfInfoIter );
1151  return lit_Undef;
1152  }
1153  if( !mReceivedFormulaPurelyPropositional && Settings::initiate_activities )
1154  {
1155  mTseitinVarFormulaMap.emplace( (int)var(tsLit), _formula );
1156  }
1157  if( !mReceivedFormulaPurelyPropositional && Settings::formula_guided_decision_heuristic )
1158  {
1159  std::vector<signed> vars;
1160  for( int pos = 0; pos < lits.size(); ++pos )
1161  vars.push_back( (signed)var(lits[pos]) );
1162  mTseitinVarShadows.emplace( (signed)var(tsLit), std::move(vars) );
1163  }
1164  // (or -ts a1 .. an)
1165  lits.push( neg( tsLit ) );
1166  addClause_( lits, _type, _original, cnfInfoIter );
1167  // Only add on deeper levels, otherwise ts is an assumption and the clauses are immediately satisfied anyway.
1168  if (_depth != 0) {
1169  // (or ts -a1) .. (or ts -an)
1170  vec<Lit> litsTmp;
1171  litsTmp.push( tsLit );
1172  int i = 0;
1173  for( const auto& sf : _formula.subformulas() )
1174  {
1175  assert( i < lits.size() );
1176  litsTmp.push( sf.isLiteral() ? addClauses( sf.negated(), _type, nextDepth, _original ) : neg( lits[i] ) );
1177  addClause_( litsTmp, _type, _original, cnfInfoIter );
1178  litsTmp.pop();
1179  ++i;
1180  }
1181  }
1182  SMTRAT_LOG_DEBUG("smtrat.sat", "Added formula " << _formula << " -> " << tsLit);
1183  return tsLit;
1184  }
1185  case carl::FormulaType::AND:
1186  {
1187  assert( _depth != 0 ); // because, this should be split in the module input
1188  if( !mReceivedFormulaPurelyPropositional && Settings::initiate_activities )
1189  {
1190  mTseitinVarFormulaMap.emplace( (int)var(tsLit), _formula );
1191  }
1192  // (or -ts a1) .. (or -ts an)
1193  // (or ts -a1 .. -an)
1194  vec<Lit> litsTmp;
1195  litsTmp.push( neg( tsLit ) );
1196  for( const auto& sf : _formula.subformulas() )
1197  {
1198  Lit l = addClauses( sf, _type, nextDepth, _original );
1199  assert( l != lit_Undef );
1200  litsTmp.push( l );
1201  addClause_( litsTmp, _type, _original, cnfInfoIter );
1202  litsTmp.pop();
1203  Lit negL = sf.isLiteral() ? addClauses( sf.negated(), _type, nextDepth, _original ) : neg( l );
1204  lits.push( negL );
1205  }
1206  if( !mReceivedFormulaPurelyPropositional && Settings::formula_guided_decision_heuristic )
1207  {
1208  std::vector<signed> vars;
1209  for( int pos = 0; pos < lits.size(); ++pos )
1210  vars.push_back( (signed)var(lits[pos]) );
1211  mTseitinVarShadows.emplace( (signed)var(tsLit), std::move(vars) );
1212  }
1213  lits.push( tsLit );
1214  addClause_( lits, _type, _original, cnfInfoIter );
1215  if( _type == NORMAL_CLAUSE )
1216  cnfInfoIter->second.mLiteral = tsLit;
1217  return tsLit;
1218  }
1219  case carl::FormulaType::IFF:
1220  {
1221  vec<Lit> tmp;
1222  if( _depth == 0 )
1223  {
1224  auto sfIter = _formula.subformulas().begin();
1225  Lit l = addClauses( *sfIter, _type, nextDepth, _original );
1226  Lit negL = sfIter->isLiteral() ? addClauses( sfIter->negated(), _type, nextDepth, _original ) : neg( l );
1227  ++sfIter;
1228  for( ; sfIter != _formula.subformulas().end(); ++sfIter )
1229  {
1230  Lit k = addClauses( *sfIter, _type, nextDepth, _original );
1231  Lit negK = sfIter->isLiteral() ? addClauses( sfIter->negated(), _type, nextDepth, _original ) : neg( k );
1232  // (or -l k)
1233  tmp.clear(); tmp.push( negL ); tmp.push( k ); addClause_( tmp, _type, _original, cnfInfoIter );
1234  // (or l -k)
1235  tmp.clear(); tmp.push( l ); tmp.push( negK ); addClause_( tmp, _type, _original, cnfInfoIter );
1236  l = k;
1237  negL = negK;
1238  }
1239  return lit_Undef;
1240  }
1241  for( const auto& sf : _formula.subformulas() )
1242  {
1243  Lit l = addClauses( sf, _type, nextDepth, _original );
1244  Lit negL = sf.isLiteral() ? addClauses( sf.negated(), _type, nextDepth, _original ) : neg( l );
1245  lits.push( l );
1246  tmp.push( negL );
1247  }
1248  if( !mReceivedFormulaPurelyPropositional && Settings::initiate_activities )
1249  {
1250  mTseitinVarFormulaMap.emplace( (int)var(tsLit), _formula );
1251  }
1252  if( !mReceivedFormulaPurelyPropositional && Settings::formula_guided_decision_heuristic )
1253  {
1254  std::vector<signed> vars;
1255  for( int pos = 0; pos < lits.size(); ++pos )
1256  vars.push_back( (signed)var(lits[pos]) );
1257  mTseitinVarShadows.emplace( (signed)var(tsLit), std::move(vars) );
1258  }
1259  // (or a1 .. an h)
1260  lits.push( tsLit ); addClause_( lits, _type, _original, cnfInfoIter );
1261  lits.pop();
1262  // (or -a1 .. -an h)
1263  tmp.push( tsLit ); addClause_( tmp, _type, _original, cnfInfoIter );
1264  // (or -a1 a2 -h) (or a1 -a2 -h) .. (or -a{n-1} an -h) (or a{n-1} -an -h)
1265  vec<Lit> tmpB;
1266  for( int i = 1; i < lits.size(); ++i )
1267  {
1268  tmpB.clear(); tmpB.push( tmp[i-1] ); tmpB.push( lits[i] ); tmpB.push( neg( tsLit ) ); addClause_( tmpB, _type, _original, cnfInfoIter );
1269  tmpB.clear(); tmpB.push( lits[i-1] ); tmpB.push( tmp[i] ); tmpB.push( neg( tsLit ) ); addClause_( tmpB, _type, _original, cnfInfoIter );
1270  }
1271  return tsLit;
1272  }
1273  case carl::FormulaType::XOR:
1274  {
1275  vec<Lit> negLits;
1276  vec<Lit> tmp;
1277  for( const auto& sf : _formula.subformulas() )
1278  {
1279  lits.push( addClauses( sf, _type, nextDepth, _original ) );
1280  negLits.push( sf.isLiteral() ? addClauses( sf.negated(), _type, nextDepth, _original ) : neg( lits.last() ) );
1281  }
1282  if( _type == NORMAL_CLAUSE )
1283  cnfInfoIter->second.mLiteral = tsLit;
1284  if( _depth == 0 )
1285  {
1286  addXorClauses( lits, negLits, 0, true, _type, tmp, _original, cnfInfoIter );
1287  return lit_Undef;
1288  }
1289  if( !mReceivedFormulaPurelyPropositional && Settings::initiate_activities )
1290  {
1291  mTseitinVarFormulaMap.emplace( (int)var(tsLit), _formula );
1292  }
1293  if( !mReceivedFormulaPurelyPropositional && Settings::formula_guided_decision_heuristic )
1294  {
1295  std::vector<signed> vars;
1296  for( int pos = 0; pos < lits.size(); ++pos )
1297  vars.push_back( (signed)var(lits[pos]) );
1298  mTseitinVarShadows.emplace( (signed)var(tsLit), std::move(vars) );
1299  }
1300  lits.push( neg( tsLit ) );
1301  negLits.push( tsLit );
1302  addXorClauses( lits, negLits, 0, true, _type, tmp, _original, cnfInfoIter );
1303  return tsLit;
1304  }
1305  case carl::FormulaType::EXISTS:
1306  case carl::FormulaType::FORALL:
1307  assert( false );
1308  std::cerr << "Formula must be quantifier-free!" << std::endl;
1309  break;
1310  default:
1311  SMTRAT_LOG_ERROR("smtrat.sat", "Unexpected formula type " << _formula.getType());
1312  SMTRAT_LOG_ERROR("smtrat.sat", _formula);
1313  assert( false );
1314  }
1315  }
1316  }
1317  return lit_Undef;

69  struct OptionLt
70  {
71  bool operator ()( const Option* x, const Option* y )
72  {
73  int test1 = strcmp( x->category, y->category );
74  return (test1 < 0 || (test1 == 0 && strcmp( x->type_name, y->type_name ) < 0));
75  }

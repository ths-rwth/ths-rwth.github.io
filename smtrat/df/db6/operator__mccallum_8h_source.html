96  rules::poly_irreducible_nonzero_sgn_inv(*deriv.delineated(), prop.poly);
97  } else if (deriv.delin().nullified().find(prop.poly) != deriv.delin().nullified().end()) {
98  rules::poly_irreducible_null_sgn_inv(deriv, prop.poly);

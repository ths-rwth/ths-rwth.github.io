18 /**
19  * This class manages features that are used to valuate variables on objects.
20  * Each feature consists of a valuation function, a level and a weight.
21  * All feature valuations of a certain level are combined linearly using the respective weights.
22  * Valuations are then compared lexicographically.
23  */
24 template<typename Objects>
25 struct FeatureCollector {
26  using Extractor = std::function<double(Objects, carl::Variable)>;
27  using Valuation = std::vector<double>;
28  std::vector<std::tuple<Extractor,std::size_t,double>> mFeatures;
29 
30  void addFeature(Extractor&& e, std::size_t level, double weight) {
31  mFeatures.emplace_back(std::move(e), level, weight);
32  }

34  Valuation valuateVariable(const Objects& o, carl::Variable v) const {
35  Valuation res;
36  for (const auto& f: mFeatures) {
37  if (res.size() <= std::get<1>(f)) {
38  res.resize(std::get<1>(f) + 1);
39  }
40  res[std::get<1>(f)] += std::get<2>(f) * std::get<0>(f)(o, v);
41  }
42  SMTRAT_LOG_DEBUG("smtrat.mcsat.variableorder", "Valuation of " << v << " is " << res);
43  return res;
44  }
45  std::vector<carl::Variable> sortVariables(const Objects& o, std::vector<carl::Variable> vars) const {
46  std::vector<std::pair<Valuation,carl::Variable>> valuations;
47  for (auto v: vars) {
48  valuations.emplace_back(valuateVariable(o, v), v);
49  }
50  std::sort(valuations.begin(), valuations.end());
51  SMTRAT_LOG_DEBUG("smtrat.mcsat.variableorder", "Evaluated to " << valuations);
52  std::transform(valuations.begin(), valuations.end(), vars.begin(), [](const auto& p){ return p.second; });
53  return vars;
54  }
55 };

80  void rehash () {
81  const vec<Pair>* old = table;

83  int old_cap = cap;
84  int newsize = primes[0];
85  for (int i = 1; newsize <= cap && i < nprimes; i++)

92  for (int j = 0; j < old[i].size(); j++){
93  _insert(old[i][j].key, old[i][j].data); }}
94 
95  delete [] old;

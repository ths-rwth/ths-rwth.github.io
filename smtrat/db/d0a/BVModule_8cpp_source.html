72  template<class Settings>
73  bool BVModule<Settings>::addCore( ModuleInput::const_iterator _subformula )
74  {
75  if( _subformula->formula().isTrue() )
76  return true;
77  mModelComputed = false;
78  if( _subformula->formula().isFalse() )
79  {
80  receivedFormulasAsInfeasibleSubset( _subformula );
81  return false;
82  }
83  if( Settings::incremental_flattening )
84  {
85  auto sortKey = std::make_pair( evaluateBVFormula(_subformula->formula()), _subformula->formula().getId() );
86  #ifdef DEBUG_BVMODULE
87  std::cout << std::endl << std::endl << "add formula" << std::endl;
88  std::cout << _subformula->formula() << std::endl;
89  std::cout << "Evaluation: " << sortKey << std::endl;
90  #endif
91  auto ret = mFormulasToBlast.insert( std::make_pair(sortKey, _subformula->formula() ) );
92  assert( ret.second );
93  assert( mPositionInFormulasToBlast.find( _subformula->formula() ) == mPositionInFormulasToBlast.end() );
94  mPositionInFormulasToBlast[_subformula->formula()] = ret.first;
95  }
96  return true;
97  }

141  template<class Settings>
142  void BVModule<Settings>::transferBackendModel() const
143  {
144  clearModel();
145  const Model& bModel = backendsModel();
146  // Build bitvector values from the values of the single bits
147  auto& blastings = mEncoder.bitvectorBlastings();
148 
149  for(auto const & bitvectorToBits : blastings)
150  {
151  carl::BVValue composedValue(bitvectorToBits.first.width());
152 
153  for(std::size_t i=0;i<bitvectorToBits.second.size();++i)
154  {
155  auto iter = bModel.find( bitvectorToBits.second[i] );
156  assert( iter != bModel.end() );
157  composedValue[i] = iter->second.asBool();
158  }
159  mModel.emplace(bitvectorToBits.first, composedValue);
160  }
161  if( rReceivedFormula().containsBooleanVariables() )
162  {
163  carl::carlVariables bvars;
164  rReceivedFormula().gatherVariables(bvars);
165  auto modelIter = mModel.begin();
166  for( carl::Variable var : bvars )
167  {
168  if (var.type() != carl::VariableType::VT_BOOL) continue;
169  auto bmodelIter = bModel.find( var );
170  assert( bmodelIter != bModel.end() );
171  modelIter = mModel.emplace_hint( modelIter, var, bmodelIter->second.asBool() );
172  }
173  }
174  getDefaultModel( mModel, (FormulaT) rReceivedFormula() );
175  }

177  template<class Settings>
178  Answer BVModule<Settings>::checkCore()
179  {
180  if( Settings::incremental_flattening )
181  {
182  #ifdef DEBUG_BVMODULE
183  std::cout << "Check in BVModule" << std::endl;
184  #endif
185  getDefaultModel( mModel, (FormulaT) rReceivedFormula() );
186  #ifdef DEBUG_BVMODULE
187  std::cout << "initial model:" << std::endl;
188  std::cout << mModel << std::endl << std::endl;
189  #endif
190  if( !mFormulasToBlast.empty() )
191  {
192  FormulaT origin = mFormulasToBlast.begin()->second;
193  #ifdef DEBUG_BVMODULE
194  std::cout << std::endl << std::endl << "Take into account: " << std::endl;
195  std::cout << origin << std::endl;
196  #endif
197  const FormulaSetT& formulasToPass = mEncoder.encode( origin );
198  mFormulasToBlast.erase( mFormulasToBlast.begin() );
199  for( const FormulaT& formulaToPass : formulasToPass )
200  addSubformulaToPassedFormula( formulaToPass, origin );
201  }
202  while( true )
203  {
204  #ifdef DEBUG_BVMODULE
205  std::cout << std::endl << "Run backends" << std::endl;
206  #endif
207  Answer backendAnswer = runBackends();
208  #ifdef DEBUG_BVMODULE
209  std::cout << " --> " << backendAnswer << std::endl << std::endl;
210  #endif
211  switch( backendAnswer )
212  {
213  case UNSAT:
214  {
215  getInfeasibleSubsets();
216  return UNSAT;
217  }
218  case SAT:
219  {
220  transferBackendModel();
221  const Model& currentModel = model();
222  #ifdef DEBUG_BVMODULE
223  std::cout << "current model: " << std::endl;
224  std::cout << currentModel << std::endl;
225  #endif
226  auto iter = mFormulasToBlast.begin();
227  for( ; iter != mFormulasToBlast.end(); ++iter )
228  {
229  unsigned tmp = satisfies( currentModel, iter->second );
230  #ifdef DEBUG_BVMODULE
231  std::cout << "currentModel satisfies" << std::endl << iter->second << std::endl << " -> " << tmp << std::endl;
232  #endif
233  assert( tmp != 2 );
234  if( tmp == 0 )
235  {
236  #ifdef DEBUG_BVMODULE
237  std::cout << std::endl << std::endl << "Not yet satisfied: " << std::endl;
238  std::cout << iter->second << std::endl;
239  #endif
240  break;
241  }
242  }
243  if( iter == mFormulasToBlast.end() )
244  {
245  #ifdef DEBUG_BVMODULE
246  std::cout << "All formulas satisfied!" << std::endl;
247  #endif
248  mModelComputed = true;
249  return SAT;
250  }
251  else
252  {
253  #ifdef DEBUG_BVMODULE
254  std::cout << std::endl << std::endl << "Take into account: " << std::endl;
255  std::cout << iter->second << std::endl;
256  #endif
257  const FormulaSetT& formulasToPass = mEncoder.encode( iter->second );
258  for( const FormulaT& formulaToPass : formulasToPass )
259  addSubformulaToPassedFormula( formulaToPass, iter->second );
260  mPositionInFormulasToBlast.erase( iter->second );
261  mFormulasToBlast.erase( iter );
262  }
263  break;
264  }
265  default:
266  assert( backendAnswer == UNKNOWN );
267  return UNKNOWN;
268  }
269  }
270  assert( false );
271  return SAT;
272  }
273  else
274  {
275  auto receivedSubformula = firstUncheckedReceivedSubformula();
276  while(receivedSubformula != rReceivedFormula().end())
277  {
278  const FormulaWithOrigins& fwo = *receivedSubformula;
279  const FormulaT& formula = fwo.formula();
280 
281  const FormulaSetT& formulasToPass = mEncoder.encode(formula);
282 
283  for(const FormulaT& formulaToPass : formulasToPass)
284  addSubformulaToPassedFormula(formulaToPass, formula);
285  ++receivedSubformula;
286  }
287 
288  Answer backendAnswer = runBackends();
289  if(backendAnswer == UNSAT)
290  {
291  getInfeasibleSubsets();
292  }
293 
294  return backendAnswer;
295  }

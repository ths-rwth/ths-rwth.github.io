63 template<class Settings>
64 size_t NewCoveringModule<Settings>::addConstraintsSAT() {
65  SMTRAT_LOG_DEBUG("smtrat.covering", "Adding constraints: " << mUnknownConstraints);
66  // Hard case:
67  // Recheck the unknown constraints with the last satisfying assignment
68  SMTRAT_LOG_DEBUG("smtrat.covering", "Rechecking the unknown constraints with the last satisfying assignment");
69  std::size_t lowestLevelWithUnsatisfiedConstraint = mVariableOrdering.size()+1;
70  bool foundUnsatisfiedConstraint = false;
71 
72  std::map<size_t, std::vector<ConstraintT>> constraintsByLevel;
73 
74  for(const auto& formula : mUnknownConstraints) {
75  ConstraintT constraint = formula.constraint();
76  size_t level = helper::level_of(mVariableOrdering, constraint.lhs()) - 1;
77  constraintsByLevel[level].push_back(std::move(constraint));
78  }
79  mUnknownConstraints.clear();
80 
81  SMTRAT_LOG_DEBUG("smtrat.covering", "Build constraint map: " << constraintsByLevel);
82 
83  // Now iterate over the constraints by level, starting with lowest first (the keys in a std::map are implicitly sorted using std::less)
84  for (const auto& levelConstraints : constraintsByLevel) {
85  SMTRAT_LOG_DEBUG("smtrat.covering", "Checking level " << levelConstraints.first);
86  if (foundUnsatisfiedConstraint) break;
87  for (const auto& constraint : levelConstraints.second) {
88  if (carl::evaluate(constraint, mLastAssignment) != true) {
89  // This constraint is unsat with the last assignment
90  SMTRAT_LOG_DEBUG("smtrat.covering", "Found unsatisfied constraint on level:" << levelConstraints.first);
91  foundUnsatisfiedConstraint = true;
92  lowestLevelWithUnsatisfiedConstraint = levelConstraints.first;
93  break ;
94  }
95  }
96  }
97 
98  // We can add the new constraints to the backend now
99  for (const auto& levelConstraints : constraintsByLevel) {
100  backend.addConstraint(levelConstraints.first, std::move(levelConstraints.second));
101  }
102  constraintsByLevel.clear();
103 
104  return lowestLevelWithUnsatisfiedConstraint ;
105 
106 }

132 template<class Settings>
133 bool NewCoveringModule<Settings>::removeConstraintsUNSAT() {
134  SMTRAT_LOG_DEBUG("smtrat.covering", "Removing constraints: " << mRemoveConstraints);
135  assert(backend.getCoveringInformation()[0].isFullCovering());
136  // Hard case:
137  // We have to remove the constraints and the corresponding stored information (derivations with the constraint as origin)
138  // This might include information in the stored full coverings
139  // If not: Nothing changes and the model stays unsatisfiable
140  // If yes: the model might become satisfiable or stays unsatisfiable -> Needs recalculation of the covering at level 0
141  // If level 0 is still full after removal of constraints: the model is still unsatisfiable
142  // If level 0 is not full after removal of constraints: the model might become satisfiable or stays unsatisfiable -> Needs recalculation off all the higher levels : TODO Ask Jasper if this is correct
143 
144  // First: remove all constraints from the backend, this also removes the corresponding derivations and invalidates the coverings, if they used a removed derivations
145  for (const auto& constraint : mRemoveConstraints) {
146  backend.removeConstraint(std::move(constraint.constraint()));
147  }
148  mRemoveConstraints.clear();
149 
150  // Second: Check if the covering on level 0 has changed/ was invalidated
151  bool hasChanged = backend.getCoveringInformation()[0].isUnknownCovering();
152 
153  SMTRAT_LOG_DEBUG("smtrat.covering", "Covering on level 0 has changed: " << hasChanged);
154 
155  // Third: If the covering has changed, we need to recompute it
156  if (hasChanged) {
157  backend.getCoveringInformation()[0].computeCovering();
158  }
159 
160  SMTRAT_LOG_DEBUG("smtrat.covering", "Covering on level is still full: " << backend.getCoveringInformation()[0].isFullCovering());
161 
162  return backend.getCoveringInformation()[0].isFullCovering();
163 }

165 template<typename Settings>
166 Answer NewCoveringModule<Settings>::checkCore() {
167 
168  SMTRAT_STATISTICS_CALL(getStatistics().called());
169 
170  SMTRAT_LOG_DEBUG("smtrat.covering", "Check Core called with unknown constraints: " << mUnknownConstraints);
171  SMTRAT_LOG_DEBUG("smtrat.covering", "Check Core called with remove constraints: " << mRemoveConstraints);
172  SMTRAT_LOG_DEBUG("smtrat.covering", "Last Answer: " << mLastAnswer);
173  SMTRAT_LOG_DEBUG("smtrat.covering", "Check Core called with last assignment: " << mLastAssignment);
174 
175  // Check if this is the first time checkCore is called
176  if (mVariableOrdering.empty()) {
177 
178  // Init variable odering, we use the variable ordering which was declared in the settings
179  mVariableOrdering = mcsat::calculate_variable_order<Settings::variableOrderingStrategy>(mAllConstraints);
180 
181  SMTRAT_STATISTICS_CALL(getStatistics().setDimension(mVariableOrdering.size()));
182 
183  // We can clear mAllConstraints now, as we don't need it anymore -> Its only needed to calculate the variable ordering
184  mAllConstraints.clear();
185 
186  SMTRAT_LOG_DEBUG("smtrat.covering", "Got Variable Ordering : " << mVariableOrdering);
187 
188  // init backend
189  backend.init(mVariableOrdering);
190 
191  // Add unknown constraints to backend
192  for (const auto& constraint : mUnknownConstraints) {
193  // Asserts that it is indeed a constraint
194  backend.addConstraint(constraint.constraint());
195  }
196 
197  mUnknownConstraints.clear();
198  } else if (mLastAnswer == Answer::UNSAT || mLastAnswer == Answer::SAT) {
199  if (mUnknownConstraints.empty() && mRemoveConstraints.empty()) {
200  // We dont have any new constraints and can just return the last value derived by the backend
201  // Why does this even happen??
202  return mLastAnswer;
203  }
204 
205  if (mRemoveConstraints.empty() && !mUnknownConstraints.empty()) {
206  // we only have constraints to add, and no constraints to remove
207 
208  SMTRAT_STATISTICS_CALL(getStatistics().calledIncrementalOnly());
209 
210  if (mLastAnswer == Answer::SAT) {
211  // Hard case:
212  auto lowestLevel = addConstraintsSAT();
213 
214  if (lowestLevel == mVariableOrdering.size() + 1) {
215  // The assignment is still satisfiable
216  return Answer::SAT;
217  } else {
218  // Remove all the data from the levels higher than lowestLevel
219  SMTRAT_LOG_DEBUG("smtrat.covering", "Removing data from level " << lowestLevel);
220  backend.resetStoredData(lowestLevel);
221  }
222  } else {
223  // Easy Case:
224  addConstraintsUNSAT();
225  // NOTE: Trivially the infeasible subset did not change
226  return Answer::UNSAT;
227  }
228 
229  } else if (!mRemoveConstraints.empty() && mUnknownConstraints.empty()) {
230  // We only have constraints to remove, and no constraints to add
231 
232  SMTRAT_STATISTICS_CALL(getStatistics().calledBacktrackingOnly());
233 
234  if (mLastAnswer == Answer::SAT) {
235  // Easy case
236  removeConstraintsSAT();
237  return Answer::SAT;
238  } else {
239  // Hard case
240  bool stillFullCovering = removeConstraintsUNSAT();
241  if (stillFullCovering) {
242  // The assignment is still unsatisfiable
243 
244  // we have to recompute the infeasible subset
245  mInfeasibleSubsets.push_back(backend.getInfeasibleSubset());
246  SMTRAT_LOG_DEBUG("smtrat.covering", "Infeasible Subset: " << mInfeasibleSubsets.back());
247  return Answer::UNSAT;
248  } else {
249  backend.resetStoredData(1);
250  }
251  }
252  } else if (!mRemoveConstraints.empty() && !mUnknownConstraints.empty()) {
253 
254  SMTRAT_STATISTICS_CALL(getStatistics().calledIncrementalAndBacktracking());
255 
256  // We have both constraints to add and constraints to remove
257  // First make sure that the vectors are disjoint
258  FormulasT intersection;
259  // we need to sort both vectors to make sure that the intersection is correct
260  std::sort(mRemoveConstraints.begin(), mRemoveConstraints.end());
261  std::sort(mUnknownConstraints.begin(), mUnknownConstraints.end());
262  std::set_intersection(mUnknownConstraints.begin(), mUnknownConstraints.end(), mRemoveConstraints.begin(), mRemoveConstraints.end(), std::back_inserter(intersection));
263 
264  if (intersection.size() > 0) {
265  // There is an intersection between the two vectors
266  // This means that we need to remove the intersection from the remove constraints
267  // NOTE: This is a rare case, but it can happen
268  // We remove the intersection from the remove constraints
269  SMTRAT_LOG_DEBUG("smtrat.covering", "Intersection between unknown and remove constraints: " << intersection);
270  for (const auto& constraint : intersection) {
271  mRemoveConstraints.erase(std::remove(mRemoveConstraints.begin(), mRemoveConstraints.end(), constraint), mRemoveConstraints.end());
272  }
273  }
274 
275  if (mLastAnswer == Answer::SAT) {
276  // first trivially remove the constraints
277  removeConstraintsSAT();
278  // then add the constraints
279  auto lowestLevel = addConstraintsSAT();
280  if (lowestLevel == mVariableOrdering.size() + 1) {
281  // The assignment is still satisfiable
282  return Answer::SAT;
283  } else {
284  // Remove all the data from the levels higher than lowestLevel
285  backend.resetStoredData(lowestLevel);
286  }
287  } else {
288  // first trivially add the constraints
289  addConstraintsUNSAT();
290  // then remove the constraints
291  bool stillFullCovering = removeConstraintsUNSAT();
292  if (stillFullCovering) {
293  // The assignment is still unsatisfiable
294  // we have to recompute the infeasible subset
295  mInfeasibleSubsets.push_back(backend.getInfeasibleSubset());
296  SMTRAT_LOG_DEBUG("smtrat.covering", "Infeasible Subset: " << mInfeasibleSubsets.back());
297  return Answer::UNSAT;
298  } else {
299  //delete everything, but keep level 0
300  backend.resetStoredData(1);
301  }
302  }
303  }
304  } else {
305  // The last call returned UNKNOWN
306  // we have to delete all stored information and completely re-initialize the backend
307  backend.resetStoredData(0);
308  backend.resetDerivationToConstraintMap();
309  }
310 
311  // As getUnsatCover is recursive we always have to start at level 0
312  mLastAnswer = backend.getUnsatCover(0);
313 
314  SMTRAT_LOG_DEBUG("smtrat.covering", "Check Core returned: " << mLastAnswer);
315 
316  if (mLastAnswer == Answer::UNSAT) {
317  mInfeasibleSubsets.push_back(backend.getInfeasibleSubset());
318  SMTRAT_LOG_DEBUG("smtrat.covering", "Infeasible Subset: " << mInfeasibleSubsets.back());
319  mLastAssignment.clear(); // There is no satisfying assignment
320 
321  } else if (mLastAnswer == Answer::SAT) {
322  mLastAssignment = backend.getCurrentAssignment();
323  } else {
324  // Answer is UNKNOWN and something went wrong
325  SMTRAT_LOG_DEBUG("smtrat.covering", "Backend encountered an error: " << mLastAnswer);
326 
327  // remove all stored information in the backend
328  backend.resetStoredData(0);
329  mLastAssignment.clear(); // There is no satisfying assignment
330  }
331 
332  updateModel(); // Update the model according to the last assignment from the backend
333 
334  return mLastAnswer;
335 }

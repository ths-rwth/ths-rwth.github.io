24  template<class Settings>
25  bool STropModule<Settings>::addCore(ModuleInput::const_iterator _subformula)
26  {
27  addReceivedSubformulaToPassedFormula(_subformula);
28  const FormulaT& formula{_subformula->formula()};
29  if (formula.getType() == carl::FormulaType::FALSE)
30  mInfeasibleSubsets.push_back({formula});
31  else if (formula.getType() == carl::FormulaType::CONSTRAINT)
32  {
33  /// Normalize the left hand side of the constraint and turn the relation accordingly
34  const ConstraintT& constraint{formula.constraint()};
35  const Poly normalization{constraint.lhs().normalize()};
36  carl::Relation relation{constraint.relation()};
37  if (carl::isNegative(constraint.lhs().lcoeff()))
38  relation = carl::turn_around(relation);
39 
40  /// Store the normalized constraint and mark the separator object as changed
41  Separator& separator{mSeparators.emplace(normalization, normalization).first->second};
42  separator.mRelations.insert(relation);
43  mChangedSeparators.insert(&separator);
44 
45  /// Check if the asserted constraint prohibits the application of this method
46  if (relation == carl::Relation::EQ
47  || (relation == carl::Relation::LEQ
48  && separator.mRelations.count(carl::Relation::GEQ))
49  || (relation == carl::Relation::GEQ
50  && separator.mRelations.count(carl::Relation::LEQ)))
51  ++mRelationalConflicts;
52 
53  /// Check if the asserted relation trivially conflicts with other asserted relations
54  switch (relation)
55  {
56  case carl::Relation::EQ:
57  if (separator.mRelations.count(carl::Relation::NEQ))
58  mInfeasibleSubsets.push_back({
59  FormulaT(normalization, carl::Relation::EQ),
60  FormulaT(normalization, carl::Relation::NEQ)
61  });
62  if (separator.mRelations.count(carl::Relation::LESS))
63  mInfeasibleSubsets.push_back({
64  FormulaT(normalization, carl::Relation::EQ),
65  FormulaT(normalization, carl::Relation::LESS)
66  });
67  if (separator.mRelations.count(carl::Relation::GREATER))
68  mInfeasibleSubsets.push_back({
69  FormulaT(normalization, carl::Relation::EQ),
70  FormulaT(normalization, carl::Relation::GREATER)
71  });
72  break;
73  case carl::Relation::NEQ:
74  if (separator.mRelations.count(carl::Relation::EQ))
75  mInfeasibleSubsets.push_back({
76  FormulaT(normalization, carl::Relation::NEQ),
77  FormulaT(normalization, carl::Relation::EQ)
78  });
79  break;
80  case carl::Relation::LESS:
81  if (separator.mRelations.count(carl::Relation::EQ))
82  mInfeasibleSubsets.push_back({
83  FormulaT(normalization, carl::Relation::LESS),
84  FormulaT(normalization, carl::Relation::EQ)
85  });
86  if (separator.mRelations.count(carl::Relation::GEQ))
87  mInfeasibleSubsets.push_back({
88  FormulaT(normalization, carl::Relation::LESS),
89  FormulaT(normalization, carl::Relation::GEQ)
90  });
91  case carl::Relation::LEQ:
92  if (separator.mRelations.count(carl::Relation::GREATER))
93  mInfeasibleSubsets.push_back({
94  FormulaT(normalization, relation),
95  FormulaT(normalization, carl::Relation::GREATER)
96  });
97  break;
98  case carl::Relation::GREATER:
99  if (separator.mRelations.count(carl::Relation::EQ))
100  mInfeasibleSubsets.push_back({
101  FormulaT(normalization, carl::Relation::GREATER),
102  FormulaT(normalization, carl::Relation::EQ)
103  });
104  if (separator.mRelations.count(carl::Relation::LEQ))
105  mInfeasibleSubsets.push_back({
106  FormulaT(normalization, carl::Relation::GREATER),
107  FormulaT(normalization, carl::Relation::LEQ)
108  });
109  case carl::Relation::GEQ:
110  if (separator.mRelations.count(carl::Relation::LESS))
111  mInfeasibleSubsets.push_back({
112  FormulaT(normalization, relation),
113  FormulaT(normalization, carl::Relation::LESS)
114  });
115  break;
116  default:
117  assert(false);
118  }
119  }
120  return mInfeasibleSubsets.empty();

151  template<class Settings>
152  void STropModule<Settings>::updateModel() const
153  {
154  if (!mModelComputed)
155  {
156  if (mCheckedWithBackends)
157  {
158  clearModel();
159  getBackendsModel();
160  excludeNotReceivedVariablesFromModel();
161  }
162  else
163  {
164  /// Stores all informations retrieved from the LRA solver to construct the model
165  struct Weight
166  {
167  const carl::Variable& mVariable;
168  Rational mExponent;
169  const bool mSign;
170 
171  Weight(const carl::Variable& variable, const Rational& exponent, const bool sign)
172  : mVariable(variable)
173  , mExponent(exponent)
174  , mSign(sign)
175  {}
176  };
177  std::vector<Weight> weights;
178 
179  /// Retrieve the sign and exponent for every active variable
180  const Model& LRAModel{mLRAModule.model()};
181  Rational gcd(0);
182  for (const auto& momentsEntry : mMoments)
183  {
184  const carl::Variable& variable{momentsEntry.first};
185  const Moment& moment{momentsEntry.second};
186  if (moment.mUsed)
187  {
188  auto signIter{LRAModel.find(moment.mSignVariant)};
189  weights.emplace_back(
190  variable,
191  LRAModel.at(moment.mNormalVector).asRational(),
192  signIter != LRAModel.end() && signIter->second.asBool()
193  );
194  carl::gcd_assign(gcd, weights.back().mExponent);
195  }
196  }
197 
198  /// Calculate smallest possible integer valued exponents
199  if (!carl::isZero(gcd) && !carl::isOne(gcd))
200  for (Weight& weight : weights)
201  weight.mExponent /= gcd;
202 
203  /// Find model by increasingly testing the sample base
204  Rational base = 0;
205  do
206  {
207  ++base;
208  clearModel();
209  for (const Weight& weight : weights)
210  {
211  Rational value{carl::isNegative(weight.mExponent) ? carl::reciprocal(base) : base};
212  carl::pow_assign(value, carl::toInt<carl::uint>(carl::abs(weight.mExponent)));
213  if (weight.mSign)
214  value *= -1;
215  mModel.emplace(weight.mVariable, value);
216  }
217  }
218  while (!rReceivedFormula().satisfiedBy(mModel));
219  }
220  mModelComputed = true;
221  }
222  }

224  template<class Settings>
225  Answer STropModule<Settings>::checkCore()
226  {
227  /// Report unsatisfiability if the already found conflicts are still unresolved
228  if (!mInfeasibleSubsets.empty())
229  return Answer::UNSAT;
230 
231  /// Predicate that decides if the given conflict is a subset of the asserted constraints
232  const auto hasConflict = [&](const Conflict& conflict) {
233  return std::all_of(
234  conflict.begin(),
235  conflict.end(),
236  [&](const auto& conflictEntry) {
237  return ((conflictEntry.second == Direction::NEGATIVE
238  || conflictEntry.second == Direction::BOTH)
239  && (conflictEntry.first->mRelations.count(carl::Relation::LESS)
240  || conflictEntry.first->mRelations.count(carl::Relation::LEQ)))
241  || ((conflictEntry.second == Direction::POSITIVE
242  || conflictEntry.second == Direction::BOTH)
243  && (conflictEntry.first->mRelations.count(carl::Relation::GREATER)
244  || conflictEntry.first->mRelations.count(carl::Relation::GEQ)))
245  || (conflictEntry.second == Direction::BOTH
246  && conflictEntry.first->mRelations.count(carl::Relation::NEQ));
247  }
248  );
249  };
250 
251  /// Apply the method only if the asserted formula is not trivially undecidable
252  if (!mRelationalConflicts
253  && rReceivedFormula().isConstraintConjunction()
254  && std::none_of(mLinearizationConflicts.begin(), mLinearizationConflicts.end(), hasConflict))
255  {
256  /// Update the linearization of all changed separators
257  for (Separator *separatorPtr : mChangedSeparators)
258  {
259  Separator& separator{*separatorPtr};
260 
261  /// Determine the direction that shall be active
262  boost::optional<Direction> direction;
263  if (!separator.mRelations.empty())
264  {
265  if (separator.mActiveDirection
266  && *separator.mActiveDirection == Direction::NEGATIVE
267  && ((separator.mRelations.count(carl::Relation::LESS)
268  || separator.mRelations.count(carl::Relation::LEQ))))
269  direction = Direction::NEGATIVE;
270  else if (separator.mActiveDirection
271  && *separator.mActiveDirection == Direction::POSITIVE
272  && ((separator.mRelations.count(carl::Relation::GREATER)
273  || separator.mRelations.count(carl::Relation::GEQ))))
274  direction = Direction::POSITIVE;
275  else
276  switch (*separator.mRelations.rbegin())
277  {
278  case carl::Relation::EQ:
279  direction = boost::none;
280  break;
281  case carl::Relation::NEQ:
282  direction = Direction::BOTH;
283  break;
284  case carl::Relation::LESS:
285  case carl::Relation::LEQ:
286  direction = Direction::NEGATIVE;
287  break;
288  case carl::Relation::GREATER:
289  case carl::Relation::GEQ:
290  direction = Direction::POSITIVE;
291  break;
292  default:
293  assert(false);
294  }
295  }
296 
297  /// Update the linearization if the direction has changed
298  if (separator.mActiveDirection != direction)
299  {
300  if (separator.mActiveDirection)
301  propagateFormula(createLinearization(separator), false);
302  separator.mActiveDirection = direction;
303  if (separator.mActiveDirection)
304  propagateFormula(createLinearization(separator), true);
305  }
306  }
307  mChangedSeparators.clear();
308 
309  /// Restrict the normal vector component of integral variables to positive values
310  for (auto& momentsEntry : mMoments)
311  {
312  const carl::Variable& variable{momentsEntry.first};
313  Moment& moment{momentsEntry.second};
314  if (variable.type() == carl::VariableType::VT_INT
315  && moment.mUsed != receivedVariable(variable))
316  {
317  moment.mUsed = !moment.mUsed;
318  propagateFormula(FormulaT(Poly(moment.mNormalVector), carl::Relation::GEQ), moment.mUsed);
319  }
320  }
321 
322  /// Check the constructed linearization with the LRA solver
323  if (mLRAModule.check(true) == Answer::SAT)
324  {
325  mCheckedWithBackends = false;
326  return Answer::SAT;
327  }
328  else
329  {
330  /// Learn the conflicting set of separators to avoid its recheck in the future
331  const std::vector<FormulaSetT> LRAConflicts{mLRAModule.infeasibleSubsets()};
332  for (const FormulaSetT& LRAConflict : LRAConflicts)
333  {
334  carl::carlVariables variables;
335  for (const FormulaT& formula : LRAConflict)
336  formula.gatherVariables(variables);
337  Conflict conflict;
338  for (const auto& separatorsEntry : mSeparators)
339  {
340  const Separator& separator{separatorsEntry.second};
341  if (separator.mActiveDirection
342  && variables.has(separator.mBias))
343  conflict.emplace_back(&separator, *separator.mActiveDirection);
344  }
345  mLinearizationConflicts.emplace_back(std::move(conflict));
346  }
347  }
348  }
349 
350  /// Check the asserted formula with the backends
351  mCheckedWithBackends = true;
352  Answer answer{runBackends()};
353  if (answer == Answer::UNSAT)
354  getInfeasibleSubsets();
355  return answer;
356  }

378  template<class Settings>
379  FormulaT STropModule<Settings>::createSeparator(const Separator& separator, bool negated)
380  {
381  Poly totalRating;
382  FormulasT disjunctions, conjunctions;
383  for (const Vertex& vertex : separator.mVertices)
384  {
385  /// Create the hyperplane and sign change formula
386  Poly hyperplane{separator.mBias};
387  FormulaT signChangeFormula;
388  if (vertex.mMonomial)
389  {
390  FormulasT signChangeSubformulas;
391  for (const auto& exponent : vertex.mMonomial->exponents())
392  {
393  const auto& moment{mMoments[exponent.first]};
394  hyperplane += carl::fromInt<Rational>(exponent.second)*moment.mNormalVector;
395  if (exponent.second % 2 == 1)
396  signChangeSubformulas.emplace_back(moment.mSignVariant);
397  }
398  signChangeFormula = FormulaT(carl::FormulaType::XOR, std::move(signChangeSubformulas));
399  }
400 
401  /// Create the rating case distinction formula
402  if (Settings::separatorType == SeparatorType::WEAK)
403  {
404  totalRating += vertex.mRating;
405  conjunctions.emplace_back(
406  carl::FormulaType::IMPLIES,
407  FormulaT(hyperplane, carl::Relation::LESS),
408  FormulaT(Poly(vertex.mRating), carl::Relation::EQ)
409  );
410  const Rational coefficient{negated ? -vertex.mCoefficient : vertex.mCoefficient};
411  conjunctions.emplace_back(
412  carl::FormulaType::IMPLIES,
413  FormulaT(hyperplane, carl::Relation::EQ),
414  FormulaT(
415  carl::FormulaType::AND,
416  FormulaT(
417  carl::FormulaType::IMPLIES,
418  signChangeFormula,
419  FormulaT(vertex.mRating+coefficient, carl::Relation::EQ)
420  ),
421  FormulaT(
422  carl::FormulaType::IMPLIES,
423  signChangeFormula.negated(),
424  FormulaT(vertex.mRating-coefficient, carl::Relation::EQ)
425  )
426  )
427  );
428  }
429 
430  /// Create the strict/weak linear saparating hyperplane
431  bool positive{carl::isPositive(vertex.mCoefficient) != negated};
432  disjunctions.emplace_back(
433  FormulaT(
434  carl::FormulaType::IMPLIES,
435  positive ? signChangeFormula.negated() : signChangeFormula,
436  FormulaT(hyperplane, Settings::separatorType == SeparatorType::STRICT ? carl::Relation::LEQ : carl::Relation::LESS)
437  ).negated()
438  );
439  conjunctions.emplace_back(
440  carl::FormulaType::IMPLIES,
441  positive ? std::move(signChangeFormula) : std::move(signChangeFormula.negated()),
442  FormulaT(std::move(hyperplane), Settings::separatorType == SeparatorType::STRICT ? carl::Relation::LESS : carl::Relation::LEQ)
443  );
444  }
445  if (Settings::separatorType == SeparatorType::WEAK)
446  conjunctions.emplace_back(totalRating, carl::Relation::GREATER);
447  return FormulaT(
448  carl::FormulaType::AND,
449  FormulaT(carl::FormulaType::OR, std::move(disjunctions)),
450  FormulaT(carl::FormulaType::AND, std::move(conjunctions))
451  );
452  }

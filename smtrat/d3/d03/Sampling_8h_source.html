69 template<>
70 struct sampling<SamplingAlgorithm::LOWER_UPPER_BETWEEN_SAMPLING> {
71  template<typename T>
72  static size_t sample_outside(cadcells::RAN& sample, const std::set<cadcells::datastructures::SampledDerivationRef<T>, SampledDerivationRefCompare>& derivationsSet) {
73 
74  // remove redundancies of the first kind
75  // the derivations are already sorted by lower bound
76  std::vector<cadcells::datastructures::SampledDerivationRef<T>> derivationsVector;
77  auto iter = derivationsSet.begin();
78  while (iter != derivationsSet.end()) {
79  derivationsVector.push_back(*iter);
80  auto& last_cell = (*iter)->cell();
81  iter++;
82  while (iter != derivationsSet.end() && !cadcells::datastructures::upper_lt_upper(last_cell, (*iter)->cell()))
83  iter++;
84  }
85 
86  if (derivationsVector.empty()) {
87  // There are no cells, just take trivially 0
88  sample = cadcells::RAN(0);
89  return 0;
90  }
91 
92  if (!derivationsVector.front()->cell().lower_unbounded()) {
93  // Lower bound is finite, just take a sufficiently large negative number
94  sample = carl::sample_below(derivationsVector.front()->cell().lower()->first);
95  return 0;
96  }
97 
98  if (!derivationsVector.back()->cell().upper_unbounded()) {
99  // Upper bound is finite, just take a sufficiently large positive number
100  sample = carl::sample_above(derivationsVector.back()->cell().upper()->first);
101  return 0;
102  }
103 
104  // Search for adjacent disjoint cells and sample between
105  for (size_t i = 0; i + 1 < derivationsVector.size(); i++) {
106  // We know that the cells are ordered by lower bound - so for checking disjointness the following suffices
107  if (!derivationsVector[i]->cell().upper_unbounded() && !derivationsVector[i + 1]->cell().lower_unbounded() && derivationsVector[i]->cell().upper()->first < derivationsVector[i + 1]->cell().lower()->first) {
108  sample = carl::sample_between(derivationsVector[i]->cell().upper()->first, derivationsVector[i + 1]->cell().lower()->first);
109  return 0;
110 
111  // The check above does not care for open bounds
112  // i.e if we have (x, y), (y, z) we can still choose y as a sample point
113  } else if (derivationsVector[i]->cell().is_sector() && derivationsVector[i + 1]->cell().is_sector() && derivationsVector[i]->cell().upper()->first == derivationsVector[i + 1]->cell().lower()->first) {
114  sample = derivationsVector[i]->cell().upper()->first;
115  return 0;
116  }
117  }
118 
119  // The cells cover the number line -> There is no sample to be found
120  return 1;
121  }
122 };

124 template<>
125 struct is_sample_outside<IsSampleOutsideAlgorithm::DEFAULT> {
126  template<typename T>
127  static bool is_outside(const cadcells::RAN& sample, const std::set<cadcells::datastructures::SampledDerivationRef<T>, SampledDerivationRefCompare>& derivations) {
128 
129  SMTRAT_LOG_DEBUG("smtrat.covering", "Checking if " << sample << " is outside of " << derivations);
130 
131  if (derivations.empty()) {
132  SMTRAT_LOG_DEBUG("smtrat.covering", "No cells, so " << sample << " is outside");
133  return true;
134  }
135 
136  for (const auto& deriv : derivations) {
137  auto& cell = deriv->cell();
138  SMTRAT_LOG_DEBUG("smtrat.covering", "Checking if " << sample << " is outside of " << cell);
139  SMTRAT_LOG_DEBUG("smtrat.covering", " Lower unbounded " << cell.lower_unbounded() << " Upper unbounded " << cell.upper_unbounded());
140 
141  // When one bound is infty, we must have a sector
142  if (cell.lower_unbounded() && cell.upper_unbounded()) {
143  // (-infty, infty)
144  return false;
145  } else if (!cell.lower_unbounded() && cell.upper_unbounded()) {
146  // (a, infty)
147  if (sample > cell.lower()->first) {
148  return false;
149  }
150  } else if (cell.lower_unbounded() && !cell.upper_unbounded()) {
151  // (-infty, a)
152  if (sample < cell.upper()->first) {
153  return false;
154  }
155  } else {
156  // both cells bounds are not infty
157  if (cell.is_section()) {
158  // [a,b]
159  // Non-strict bounds
160  if (cell.lower()->first <= sample && sample <= cell.upper()->first) {
161  return false;
162  }
163  } else {
164  // Strict bounds
165  // (a,b)
166  if (cell.lower()->first < sample && sample < cell.upper()->first) {
167  return false;
168  }
169  }
170  }
171  }
172  return true;
173  }
174 };

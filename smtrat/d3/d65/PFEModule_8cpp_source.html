42  template<class Settings>
43  Answer PFEModule<Settings>::checkCore() {
44  auto receivedFormula = firstUncheckedReceivedSubformula();
45  if (boundsChanged) {
46  clearPassedFormula();
47  receivedFormula = rReceivedFormula().begin();
48  boundsChanged = false;
49  }
50  while (receivedFormula != rReceivedFormula().end()) {
51  if (is_minimizing() && receivedFormula->formula().variables().find(objective()) != receivedFormula->formula().variables().end()) {
52  SMTRAT_LOG_DEBUG("smtrat.pfe", "Ignoring " << receivedFormula->formula() << " as it containts the objective variable " << objective());
53  addReceivedSubformulaToPassedFormula(receivedFormula);
54  ++receivedFormula;
55  continue;
56  }
57  if (receivedFormula->formula().isBound()) {
58  addReceivedSubformulaToPassedFormula(receivedFormula);
59  ++receivedFormula;
60  continue;
61  }
62  FormulaT formula = visitor.visitResult(receivedFormula->formula(), removeFactorsFunction);
63  formula = visitor.visitResult(formula, removeSquaresFunction);
64  //formula = visitor.visitResult(formula, implyDefinitenessFunction);
65  if (receivedFormula->formula() != formula) {
66  SMTRAT_LOG_DEBUG("smtrat.pfe", "Simplified " << receivedFormula->formula());
67  SMTRAT_LOG_DEBUG("smtrat.pfe", "to " << formula);
68  SMTRAT_LOG_DEBUG("smtrat.pfe", "due to bounds " << varbounds.getEvalIntervalMap());
69  }
70 
71  if (formula.isFalse()) {
72  mInfeasibleSubsets.clear();
73  carl::carlVariables vars;
74  receivedFormula->formula().gatherVariables(vars);
75  FormulaSetT infeasibleSubset = varbounds.getOriginSetOfBounds(vars.as_set());
76  infeasibleSubset.insert(receivedFormula->formula());
77  mInfeasibleSubsets.push_back(std::move(infeasibleSubset));
78  return UNSAT;
79  }
80  if (!formula.isTrue()) {
81  if (formula == receivedFormula->formula()) {
82  addReceivedSubformulaToPassedFormula(receivedFormula);
83  } else {
84  carl::carlVariables vars;
85  receivedFormula->formula().gatherVariables(vars);
86  FormulasT origins = varbounds.getOriginsOfBounds(vars.as_set());
87  origins.push_back(receivedFormula->formula());
88  addSubformulaToPassedFormula(formula, std::make_shared<std::vector<FormulaT>>(std::move(origins)));
89  }
90  }
91  ++receivedFormula;
92  }
93  generateVariableAssignments();
94  SMTRAT_LOG_DEBUG("smtrat.pfe", "Simplification: " << FormulaT(rReceivedFormula()));
95  SMTRAT_LOG_DEBUG("smtrat.pfe", "to " << FormulaT(rPassedFormula()));
96  Answer ans = runBackends();
97  if (ans == UNSAT) {
98  getInfeasibleSubsets();
99  }
100  return ans;
101  }

110  template<typename Settings>
111  FormulaT PFEModule<Settings>::removeFactors(const FormulaT& formula){
112  if(formula.getType() == carl::FormulaType::CONSTRAINT) {
113  const auto& factors = formula.constraint().factorization();
114  SMTRAT_LOG_TRACE("smtrat.pfe", "Factorization of " << formula << " = " << factors);
115  std::vector<Factorization::const_iterator> Pq;
116  std::vector<Factorization::const_iterator> Pr;
117  carl::Relation qrel = carl::Relation::GREATER;
118  for (auto it = factors.begin(); it != factors.end(); it++) {
119  auto i = carl::IntervalEvaluation::evaluate(it->first, completeBounds(it->first));
120  SMTRAT_LOG_TRACE("smtrat.pfe", "Considering factor " << it->first << " with bounds " << i);
121  if (i.isPositive()) {
122  qrel = combine(qrel, carl::Relation::GREATER, it->second);
123  Pq.push_back(it);
124  } else if (i.isSemiPositive()) {
125  qrel = combine(qrel, carl::Relation::GEQ, it->second);
126  Pq.push_back(it);
127  } else if (i.isNegative()) {
128  qrel = combine(qrel, carl::Relation::LESS, it->second);
129  Pq.push_back(it);
130  } else if (i.isSemiNegative()) {
131  qrel = combine(qrel, carl::Relation::LEQ, it->second);
132  Pq.push_back(it);
133  } else if (i.isZero()) {
134  qrel = combine(qrel, carl::Relation::EQ, it->second);
135  Pq.push_back(it);
136  } else {
137  Pr.push_back(it);
138  }
139  }
140  if (!Pq.empty()) {
141  carl::Relation rel = formula.constraint().relation();
142  assert(qrel != carl::Relation::NEQ);
143  switch (qrel) {
144  case carl::Relation::GREATER: return FormulaT(getPoly(Pr), rel);
145  case carl::Relation::GEQ:
146  switch (rel) {
147  case carl::Relation::EQ: return FormulaT(carl::FormulaType::OR, {FormulaT(getPoly(Pq), carl::Relation::EQ), FormulaT(getPoly(Pr), rel)});
148  case carl::Relation::NEQ: return FormulaT(carl::FormulaType::AND, {FormulaT(getPoly(Pq), carl::Relation::GREATER), FormulaT(getPoly(Pr), rel)});
149  case carl::Relation::GEQ: return FormulaT(carl::FormulaType::OR, {FormulaT(getPoly(Pq), carl::Relation::EQ), FormulaT(getPoly(Pr), rel)});
150  case carl::Relation::GREATER: return FormulaT(carl::FormulaType::AND, {FormulaT(getPoly(Pq), carl::Relation::GREATER), FormulaT(getPoly(Pr), rel)});
151  case carl::Relation::LEQ: return FormulaT(carl::FormulaType::OR, {FormulaT(getPoly(Pq), carl::Relation::EQ), FormulaT(getPoly(Pr), rel)});
152  case carl::Relation::LESS: return FormulaT(carl::FormulaType::AND, {FormulaT(getPoly(Pq), carl::Relation::GREATER), FormulaT(getPoly(Pr), rel)});
153  default: return formula;
154  }
155  case carl::Relation::EQ: return FormulaT(Poly(0), rel);
156  case carl::Relation::LEQ:
157  switch (rel) {
158  case carl::Relation::EQ: return FormulaT(carl::FormulaType::OR, {FormulaT(getPoly(Pq), carl::Relation::EQ), FormulaT(getPoly(Pr), rel)});
159  case carl::Relation::NEQ: return FormulaT(carl::FormulaType::AND, {FormulaT(getPoly(Pq), carl::Relation::LESS), FormulaT(getPoly(Pr), rel)});
160  case carl::Relation::GEQ: return FormulaT(carl::FormulaType::OR, {FormulaT(getPoly(Pq), carl::Relation::EQ), FormulaT(getPoly(Pr), carl::Relation::LEQ)});
161  case carl::Relation::GREATER: return FormulaT(carl::FormulaType::AND, {FormulaT(getPoly(Pq), carl::Relation::LESS), FormulaT(getPoly(Pr), carl::Relation::LESS)});
162  case carl::Relation::LEQ: return FormulaT(carl::FormulaType::OR, {FormulaT(getPoly(Pq), carl::Relation::EQ), FormulaT(getPoly(Pr), carl::Relation::GEQ)});
163  case carl::Relation::LESS: return FormulaT(carl::FormulaType::AND, {FormulaT(getPoly(Pq), carl::Relation::LESS), FormulaT(getPoly(Pr), carl::Relation::GREATER)});
164  default: return formula;
165  }
166  case carl::Relation::LESS:
167  switch (rel) {
168  case carl::Relation::EQ: return FormulaT(getPoly(Pr), rel);
169  case carl::Relation::NEQ: return FormulaT(getPoly(Pr), rel);
170  case carl::Relation::GEQ: return FormulaT(getPoly(Pr), carl::Relation::LEQ);
171  case carl::Relation::GREATER: return FormulaT(getPoly(Pr), carl::Relation::LESS);
172  case carl::Relation::LEQ: return FormulaT(getPoly(Pr), carl::Relation::GEQ);
173  case carl::Relation::LESS: return FormulaT(getPoly(Pr), carl::Relation::GREATER);
174  default: return formula;
175  }
176  default: return formula;
177  }
178  }
179  }
180  return formula;
181  }

183  template<typename Settings>
184  FormulaT PFEModule<Settings>::removeSquaresFromStrict(const FormulaT& formula) {
185  const auto& factors = formula.constraint().factorization();
186  std::vector<Factorization::const_iterator> Pq;
187  std::vector<Factorization::const_iterator> Pr;
188 
189  for (auto it = factors.begin(); it != factors.end(); ++it) {
190  if (it->second % 2 == 0) {
191  // This implies that this factor is (strictly) positive and essentially reduces to factor != 0
192  SMTRAT_LOG_DEBUG("smtrat.pfe", "Eliminating factors " << it->first << " ^ " << it->second);
193  Pq.push_back(it);
194  } else {
195  Pr.push_back(it);
196  }
197  }
198 
199  if (Pq.empty()) return formula;
200 
201  FormulasT res;
202  for (const auto& q: Pq) {
203  res.emplace_back(ConstraintT(q->first, carl::Relation::NEQ));
204  }
205  auto polyrest = std::accumulate(Pr.begin(), Pr.end(), Poly(1), [](const auto& a, const auto& b){ return a * carl::pow(b->first, b->second); });
206  res.emplace_back(ConstraintT(polyrest, formula.constraint().relation()));
207  return FormulaT(carl::FormulaType::AND, std::move(res));
208  }

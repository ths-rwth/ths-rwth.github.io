37  Answer GBPPModule<Settings>::checkCore()
38  {
39  mEqualities.clear();
40  mEqualityComplexity = 0;
41  mBasis.reset();
42 
43  // Extract top-level Equalities
44  for (const auto& f: rReceivedFormula()) {
45  if (f.formula().getType() == carl::FormulaType::CONSTRAINT) {
46  if (f.formula().constraint().relation() == carl::Relation::EQ) {
47  SMTRAT_LOG_DEBUG("smtrat.gbpp", "Found equality " << f.formula().constraint());
48  mEqualities.emplace(f.formula().constraint());
49  mEqualityComplexity += carl::complexity(f.formula().constraint().lhs());
50  }
51  }
52  }
53 
54  // Compute GBasis
55  for (const auto& eq: mEqualities) {
56  SMTRAT_LOG_DEBUG("smtrat.gbpp", "Adding to Gröbner Basis: " << gpoly(eq.constraint().lhs().normalize()));
57  mBasis.addPolynomial(gpoly(eq.constraint().lhs().normalize()));
58  }
59  mBasis.calculate();
60  SMTRAT_LOG_DEBUG("smtrat.gbpp", "Constructed Gröbner Basis:" << std::endl << mBasis.getIdeal());
61 
62  // Simplify all inequalities w.r.t. GBasis and forward to backend
63  carl::FormulaVisitor<FormulaT> visitor;
64  for (const auto& f: rReceivedFormula()) {
65  if (mEqualities.find(f.formula()) != mEqualities.end()) continue;
66  auto res = visitor.visitResult(f.formula(), simplifyInequalityFunction);
67 
68  if (res != f.formula()) {
69  SMTRAT_LOG_INFO("smtrat.gbpp", "Reduced " << f.formula() << " to " << res);
70  }
71  if (!res.isTrue()) {
72  addSubformulaToPassedFormula(res, f.formula());
73  }
74  }
75 
76  // Forward basis to backend
77  std::size_t basisComplexity = 0;
78  for (const auto& p: mBasis.getIdeal().getGenerators()) {
79  basisComplexity += carl::complexity(p);
80  }
81  if (basisComplexity >= mEqualityComplexity) {
82  for (const auto& f: mEqualities) {
83  addSubformulaToPassedFormula(f, f);
84  }
85  } else {
86  for (const auto& p: mBasis.getIdeal().getGenerators()) {
87  addSubformulaToPassedFormula(FormulaT(ConstraintT(Poly(p), carl::Relation::EQ)));
88  }
89  }
90 
91  Answer ans = runBackends();
92  if (ans == UNSAT) {
93  generateTrivialInfeasibleSubset();
94  }
95  return ans;
96  }

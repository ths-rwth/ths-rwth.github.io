118  {

134  }

140  template<class Settings>
141  std::pair<bool,bool> EQPreprocessingModule<Settings>::P_collect_facts()
142  {
143  bool stable = true;
144 
145  for(auto&& value : mOldToNew) {
146  if(value.second.getType() == carl::UEQ) {
147  std::pair<bool,bool> r = P_collect_fact(value.first, value.second, false);
148  if(!r.first) {
149  return std::make_pair(false,false);
150  }
151 
152  stable &= r.second;
153  } else if(value.second.getType() == carl::NOT && value.second.subformula().getType() == carl::UEQ) {
154  std::pair<bool,bool> r = P_collect_fact(value.first, value.second.subformula(), true);
155  if(!r.first) {
156  return std::make_pair(false,false);
157  }
158 
159  stable &= r.second;
160  } else if(value.second.getType() == carl::AND) {
161  for(const FormulaT& fact : value.second.subformulas()) {
162  if(fact.getType() == carl::UEQ) {
163  std::pair<bool,bool> r = P_collect_fact(value.first, fact, false);
164  if(!r.first) {
165  return std::make_pair(false,false);
166  }
167 
168  stable &= r.second;
169  } else if(fact.getType() == carl::NOT && fact.subformula().getType() == carl::UEQ) {
170  std::pair<bool,bool> r = P_collect_fact(value.first, fact.subformula(), true);
171  if(!r.first) {
172  return std::make_pair(false,false);
173  }
174 
175  stable &= r.second;
176  }
177  }
178  }
179  }
180 
181  return std::make_pair(mEQHelper.check(), stable);
182  }

292  void EQPreprocessingModule<Settings>::P_update_model()
293  {
294  clearModel();
295 
296  if(solverState() == SAT) {
297  getBackendsModel();
298 
299  if(Settings::rewriteFunctionInstances) {
300  for(auto&& pair : mRewriter.get().UFItoVar()) {
301  const UFInstance& instance = pair.first;
302  const UVariable& helper = pair.second;
303 
304  carl::UFModel &ufModel = mModel.emplace(instance.uninterpretedFunction(), carl::UFModel(instance.uninterpretedFunction())).first->second.asUFModel();
305 
306  std::vector<carl::SortValue> args;
307  args.reserve(instance.args().size());
308  for(std::size_t i = 0, s = instance.args().size(); i < s; ++i) {
309  assert(instance.args()[i].isUVariable());
310  const UVariable& var = instance.args()[i].asUVariable();
311  auto model_iter = mModel.find(var.variable());
312 
313  if(model_iter == mModel.end()) {
314  // there is no model value for the variable, because it does not occur outside of function instances.
315  // this should really look different in the module (like not including the index at all), but we can just assume that all the values are distinct.
316  args.push_back(newSortValue(var.domain()));
317  } else {
318  // use the model value of the variable
319  args.push_back(model_iter->second.asSortValue());
320  }
321  }
322 
323  auto modelItr = mModel.find(helper.variable());
324  assert(modelItr != mModel.end());
325  ufModel.extend(std::move(args), modelItr->second.asSortValue());
326  mModel.erase(modelItr);
327  }
328  }
329  }

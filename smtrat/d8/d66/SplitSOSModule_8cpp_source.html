73  FormulaT SplitSOSModule<Settings>::splitSOS( const FormulaT& formula )
74  {
75  if( formula.getType() == carl::FormulaType::CONSTRAINT )
76  {
77  std::vector<std::pair<Rational,Poly>> sosDec;
78  bool lcoeffNeg = carl::isNegative( formula.constraint().lhs().lcoeff() );
79  if( lcoeffNeg ) {
80  sosDec = carl::sos_decomposition(-formula.constraint().lhs());
81  }
82  else
83  {
84  sosDec = carl::sos_decomposition(formula.constraint().lhs());
85  }
86  if( sosDec.size() <= 1 )
87  {
88  return formula;
89  }
90  carl::Relation rel = carl::Relation::EQ;
91  carl::FormulaType boolOp = carl::FormulaType::AND;
92  switch( formula.constraint().relation() )
93  {
94  case carl::Relation::EQ:
95  if( formula.constraint().lhs().hasConstantTerm() )
96  return FormulaT( carl::FormulaType::FALSE );
97  break;
98  case carl::Relation::NEQ:
99  if( formula.constraint().lhs().hasConstantTerm() )
100  return FormulaT( carl::FormulaType::TRUE );
101  rel = carl::Relation::NEQ;
102  boolOp = carl::FormulaType::OR;
103  break;
104  case carl::Relation::LEQ:
105  if( lcoeffNeg )
106  return FormulaT( carl::FormulaType::TRUE );
107  else if( formula.constraint().lhs().hasConstantTerm() )
108  return FormulaT( carl::FormulaType::FALSE );
109  break;
110  case carl::Relation::LESS:
111  if( lcoeffNeg )
112  {
113  if( formula.constraint().lhs().hasConstantTerm() )
114  return FormulaT( carl::FormulaType::TRUE );
115  rel = carl::Relation::NEQ;
116  boolOp = carl::FormulaType::OR;
117  }
118  else
119  return FormulaT(carl::FormulaType::FALSE);
120  break;
121  case carl::Relation::GEQ:
122  if( !lcoeffNeg )
123  return FormulaT( carl::FormulaType::TRUE );
124  else if( formula.constraint().lhs().hasConstantTerm() )
125  return FormulaT( carl::FormulaType::FALSE );
126  break;
127  default:
128  assert( formula.constraint().relation() == carl::Relation::GREATER );
129  if( lcoeffNeg )
130  return FormulaT( carl::FormulaType::FALSE );
131  else
132  {
133  if( formula.constraint().lhs().hasConstantTerm() )
134  return FormulaT( carl::FormulaType::TRUE );
135  rel = carl::Relation::NEQ;
136  boolOp = carl::FormulaType::OR;
137  }
138  }
139  FormulasT subformulas;
140  for( auto it = sosDec.begin(); it != sosDec.end(); ++it )
141  {
142  subformulas.emplace_back( it->second, rel );
143  }
144  return FormulaT( boolOp, subformulas );
145  }
146  return formula;
147  }
148 }

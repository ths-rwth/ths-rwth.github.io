135  bool UFCegarModule<Settings>::refine(const UFInstance& a, const UFInstance& b) noexcept
136  {
137  using carl::FormulaType;
138 
139  if (refined.count({a, b}))
140  return false;
141 
142  auto args = std::make_pair(a.args().begin(), b.args().begin());
143  auto end = a.args().end();
144 
145  auto record_instance = [&] (const auto& arg) {
146  if (arg.isUFInstance()) {
147  const auto& ufi = arg.asUFInstance();
148  if (!instances[ufi.uninterpretedFunction()].count(ufi)) {
149  pending.emplace(ufi);
150  }
151  }
152  };
153 
154  FormulasT eqs;
155  for ( ; args.first != end; ++args.first, ++args.second ) {
156  record_instance(*args.first);
157  record_instance(*args.second);
158  eqs.emplace_back(flatten(*args.first), flatten(*args.second), false);
159  }
160 
161  addSubformulaToPassedFormula(create_functional_contraint( std::move(eqs), a, b ));
162 
163  refined.emplace(a, b);
164  return true;

167  template<class Settings>
168  bool UFCegarModule<Settings>::refine() noexcept {
169  bool added_constraint = false;
170 
171  // generate functional consistency
172  for (const auto& [function, list] : instances) {
173  if (list.size() <= 1)
174  continue;
175 
176  for (auto i = list.begin(); i != std::prev(list.end()); ++i) {
177  for (auto j = std::next(list.begin()); j != list.end(); ++j) {
178  added_constraint |= refine(*i, *j);
179  }
180  }
181  }
182 
183  for (auto&& ufi : pending) {
184  instances[ufi.uninterpretedFunction()].insert(std::move(ufi));
185  }
186 
187  pending.clear();
188 
189  return added_constraint;
190  }

209  bool constrained = false;
210  for (const auto& [function, list] : instances) {
211  if (list.size() <= 1)
212  continue;
213 
214  for (auto i = list.begin(); i != std::prev(list.end()); ++i) {
215  for (auto j = std::next(list.begin()); j != list.end(); ++j) {

217  auto b = *j;

241  template<class Settings>
242  Answer UFCegarModule<Settings>::checkCore()
243  {
244  assert(flattened_function_calls(instances));
245 
246  Answer result;
247  if constexpr (Settings::cegar) {
248  result = runBackends();
249 
250  bool refinable = true;
251  while (result == Answer::SAT && refinable) {
252  if (checkModel()) {
253  return result;
254  }
255 
256  refinable = refine();
257  if (refinable) {
258  result = runBackends();
259  }
260  }
261  } else {
262  ackermanize();
263  result = runBackends();
264  }
265 
266  if (result == Answer::UNSAT) {
267  getInfeasibleSubsets();
268  }
269 
270  return result;
271  }

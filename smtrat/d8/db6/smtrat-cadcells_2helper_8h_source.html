7 /**
8  * Find the index of the highest variable (wrt. the ordering in 'order') that occurs with positive degree in 'poly'.
9  * We start counting at 1 and represent "no level/variable" as 0.
10  * Examples:
11  * - level(2) == 0 wrt. any variable order
12  * - level(0*x+2) == 0 wrt. any variable order
13  * - level(x+2) == 1 wrt. [x < y < z]
14  * - level(x+2) == 2 wrt. [y < x < z]
15  * - level(x+2) == 3 wrt. [y < z < x]
16  * - level(x*y+2) == 2 wrt. [x < y < z] because of y
17  * - level(x*y+2) == 2 wrt. [y < x < z] because of x
18  * - level(x*y+2) == 3 wrt. [x < z < y] because of y
19  * Preconditions:
20  * - 'poly.gatherVariables()' must be a subset of 'order'.
21  */
22 static size_t level_of(const VariableOrdering& order, const Poly& poly) {
23  // assert(isSubset(asVector(poly.gatherVariables()), order));
24  auto poly_variables = carl::variables(poly).as_set();
25  if (poly_variables.empty()) {
26  // polynomial is constant
27  return 0;
28  }
29  for (std::size_t level = 1; level <= order.size(); ++level) {
30  poly_variables.erase(order[level-1]);
31  if (poly_variables.empty()) return level;
32  }
33  assert(false && "Poly contains variable not found in order"); // precondition violated
34  return 0;
35 }

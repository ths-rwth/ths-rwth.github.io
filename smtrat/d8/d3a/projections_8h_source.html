110  PolyRef res(PolyRef p, PolyRef q) {
111  assert(p.level == q.level && p.id != q.id);
112 
113  if (p.id > q.id) return res(q,p);

169  bool is_zero(const Assignment& sample, PolyRef p) {
170  auto restricted_sample = restrict_assignment(sample ,p);
171  assert(p.level == level_of(restricted_sample));
172  if (restricted_sample.empty()) return is_zero(p);
173  if (cache(restricted_sample).is_zero.find(p) == cache(restricted_sample).is_zero.end()) {
174  auto mv = carl::evaluate(ConstraintT(m_pool(p), carl::Relation::EQ), restricted_sample);

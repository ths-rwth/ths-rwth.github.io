26  template<class Settings>
27  bool CSplitModule<Settings>::addCore(ModuleInput::const_iterator _subformula)
28  {
29  addReceivedSubformulaToPassedFormula(_subformula);
30  const FormulaT& formula{_subformula->formula()};
31  if (formula.getType() == carl::FormulaType::FALSE)
32  mInfeasibleSubsets.push_back({formula});
33  else if (formula.isBound())
34  {
35  /// Update the variable domain with the asserted bound
36  mVariableBounds.addBound(formula, formula);
37  const carl::Variable& variable{*formula.variables().begin()};
38  auto expansionIter{mExpansions.firstFind(variable)};
39  if (expansionIter == mExpansions.end())
40  expansionIter = mExpansions.emplace(variable);
41  expansionIter->mChangedBounds = true;
42  if (mVariableBounds.isConflicting())
43  mInfeasibleSubsets.emplace_back(mVariableBounds.getConflict());
44  }
45  else if (formula.getType() == carl::FormulaType::CONSTRAINT)
46  {
47  /// Normalize the left hand side of the constraint and turn the relation accordingly
48  const ConstraintT& constraint{formula.constraint()};
49  const Poly normalization{constraint.lhs().normalize()};
50  carl::Relation relation{constraint.relation()};
51  if (carl::isNegative(constraint.lhs().lcoeff()))
52  relation = carl::turn_around(relation);
53 
54  /// Purifiy and discretize the normalized left hand side to construct the linearization
55  auto linearizationIter{mLinearizations.firstFind(normalization)};
56  if (linearizationIter == mLinearizations.end())
57  {
58  Poly discretization;
59  std::vector<Purification *> purifications;
60  bool hasRealVariables{false};
61  for (TermT term : normalization)
62  {
63  if (!term.isConstant())
64  {
65  size_t realVariables{0};
66  for (const auto& exponent : term.monomial()->exponents())
67  if (exponent.first.type() == carl::VariableType::VT_REAL)
68  realVariables += exponent.second;
69  if (realVariables)
70  {
71  term.coeff() /= carl::pow(Rational(Settings::discrDenom), realVariables);
72  hasRealVariables = true;
73  }
74 
75  if (!term.isLinear())
76  {
77  Purification& purification{mPurifications[term.monomial()]};
78  purifications.emplace_back(&purification);
79  term = term.coeff()*purification.mSubstitutions[0];
80  }
81  else if (realVariables)
82  {
83  const carl::Variable variable{term.getSingleVariable()};
84  auto expansionIter{mExpansions.firstFind(variable)};
85  if (expansionIter == mExpansions.end())
86  expansionIter = mExpansions.emplace(variable);
87  term = term.coeff()*expansionIter->mQuotients[0];
88  }
89  }
90  discretization += term;
91  }
92  linearizationIter = mLinearizations.emplace(normalization, std::move(discretization.normalize()), std::move(purifications), hasRealVariables);
93  }
94  Linearization& linearization{*linearizationIter};
95  propagateFormula(FormulaT(linearization.mLinearization, relation), true);
96  if (linearization.mRelations.empty())
97  for (Purification *purification : linearization.mPurifications)
98  ++purification->mUsage;
99  linearization.mRelations.emplace(relation);
100 
101  /// Check if the asserted relation trivially conflicts with other asserted relations
102  switch (relation)
103  {
104  case carl::Relation::EQ:
105  if (linearization.mRelations.count(carl::Relation::NEQ))
106  mInfeasibleSubsets.push_back({
107  FormulaT(normalization, carl::Relation::EQ),
108  FormulaT(normalization, carl::Relation::NEQ)
109  });
110  if (linearization.mRelations.count(carl::Relation::LESS))
111  mInfeasibleSubsets.push_back({
112  FormulaT(normalization, carl::Relation::EQ),
113  FormulaT(normalization, carl::Relation::LESS)
114  });
115  if (linearization.mRelations.count(carl::Relation::GREATER))
116  mInfeasibleSubsets.push_back({
117  FormulaT(normalization, carl::Relation::EQ),
118  FormulaT(normalization, carl::Relation::GREATER)
119  });
120  break;
121  case carl::Relation::NEQ:
122  if (linearization.mRelations.count(carl::Relation::EQ))
123  mInfeasibleSubsets.push_back({
124  FormulaT(normalization, carl::Relation::NEQ),
125  FormulaT(normalization, carl::Relation::EQ)
126  });
127  break;
128  case carl::Relation::LESS:
129  if (linearization.mRelations.count(carl::Relation::EQ))
130  mInfeasibleSubsets.push_back({
131  FormulaT(normalization, carl::Relation::LESS),
132  FormulaT(normalization, carl::Relation::EQ)
133  });
134  if (linearization.mRelations.count(carl::Relation::GEQ))
135  mInfeasibleSubsets.push_back({
136  FormulaT(normalization, carl::Relation::LESS),
137  FormulaT(normalization, carl::Relation::GEQ)
138  });
139  case carl::Relation::LEQ:
140  if (linearization.mRelations.count(carl::Relation::GREATER))
141  mInfeasibleSubsets.push_back({
142  FormulaT(normalization, relation),
143  FormulaT(normalization, carl::Relation::GREATER)
144  });
145  break;
146  case carl::Relation::GREATER:
147  if (linearization.mRelations.count(carl::Relation::EQ))
148  mInfeasibleSubsets.push_back({
149  FormulaT(normalization, carl::Relation::GREATER),
150  FormulaT(normalization, carl::Relation::EQ)
151  });
152  if (linearization.mRelations.count(carl::Relation::LEQ))
153  mInfeasibleSubsets.push_back({
154  FormulaT(normalization, carl::Relation::GREATER),
155  FormulaT(normalization, carl::Relation::LEQ)
156  });
157  case carl::Relation::GEQ:
158  if (linearization.mRelations.count(carl::Relation::LESS))
159  mInfeasibleSubsets.push_back({
160  FormulaT(normalization, relation),
161  FormulaT(normalization, carl::Relation::LESS)
162  });
163  break;
164  default:
165  assert(false);
166  }
167  }
168  return mInfeasibleSubsets.empty();
169  }

171  template<class Settings>
172  void CSplitModule<Settings>::removeCore(ModuleInput::const_iterator _subformula)
173  {
174  const FormulaT& formula{_subformula->formula()};
175  if (formula.isBound())
176  {
177  /// Update the variable domain with the removed bound
178  mVariableBounds.removeBound(formula, formula);
179  mExpansions.firstAt(*formula.variables().begin()).mChangedBounds = true;
180  }
181  else if (formula.getType() == carl::FormulaType::CONSTRAINT)
182  {
183  /// Normalize the left hand side of the constraint and turn the relation accordingly
184  const ConstraintT& constraint{formula.constraint()};
185  const Poly normalization{constraint.lhs().normalize()};
186  carl::Relation relation{constraint.relation()};
187  if (carl::isNegative(constraint.lhs().lcoeff()))
188  relation = carl::turn_around(relation);
189 
190  /// Retrieve the normalized constraint and mark the separator object as changed
191  Linearization& linearization{mLinearizations.firstAt(normalization)};
192  propagateFormula(FormulaT(linearization.mLinearization, relation), false);
193  linearization.mRelations.erase(relation);
194  if (linearization.mRelations.empty())
195  for (Purification *purification : linearization.mPurifications)
196  ++purification->mUsage;
197  }

226  template<class Settings>
227  Answer CSplitModule<Settings>::checkCore()
228  {
229  /// Report unsatisfiability if the already found conflicts are still unresolved
230  if (!mInfeasibleSubsets.empty())
231  return Answer::UNSAT;
232 
233  /// Apply the method only if the asserted formula is not trivially undecidable
234  if (rReceivedFormula().isConstraintConjunction())
235  {
236  mLRAModule.push();
237  if (resetExpansions())
238  {
239  mCheckedWithBackends = false;
240  for (size_t i = 1; i <= Settings::maxIter; ++i)
241  if (mLRAModule.check(true) == Answer::SAT)
242  {
243  mLRAModel = mLRAModule.model();
244  mLRAModule.pop();
245  return Answer::SAT;
246  }
247  else
248  {
249  FormulaSetT conflict{mLRAModule.infeasibleSubsets()[0]};
250  if (bloatDomains(conflict))
251  {
252  mLRAModule.pop();
253  return analyzeConflict(conflict);
254  }
255  }
256  }
257  mLRAModule.pop();
258  }
259 
260  /// Check the asserted formula with the backends
261  mCheckedWithBackends = true;
262  Answer answer{runBackends()};
263  if (answer == Answer::UNSAT)
264  getInfeasibleSubsets();
265 
266  return answer;
267  }

269  template<class Settings>
270  bool CSplitModule<Settings>::resetExpansions()
271  {
272  /// Update the variable domains and watch out for discretization conflicts
273  for (Expansion& expansion : mExpansions)
274  {
275  RationalInterval& maximalDomain{expansion.mMaximalDomain};
276  if (expansion.mChangedBounds)
277  {
278  maximalDomain = mVariableBounds.getInterval(expansion.mRationalization);
279  if (expansion.mRationalization.type() == carl::VariableType::VT_REAL)
280  maximalDomain *= Rational(Settings::discrDenom);
281  maximalDomain.integralPart_assign();
282  if (expansion.mMaximalDomain.isUnbounded())
283  expansion.mMaximalDomainSize = DomainSize::UNBOUNDED;
284  else if (expansion.mMaximalDomain.diameter() > Settings::maxDomainSize)
285  expansion.mMaximalDomainSize = DomainSize::LARGE;
286  else
287  expansion.mMaximalDomainSize = DomainSize::SMALL;
288  expansion.mChangedBounds = false;
289  }
290  if (maximalDomain.isEmpty())
291  return false;
292  expansion.mActiveDomain = RationalInterval::emptyInterval();
293  expansion.mPurifications.clear();
294  }
295 
296  /// Activate all used purifications bottom-up
297  for (auto purificationIter = mPurifications.begin(); purificationIter != mPurifications.end(); ++purificationIter)
298  {
299  Purification& purification{purificationIter->second};
300  if (purification.mUsage)
301  {
302  carl::Monomial::Arg monomial{purificationIter->first};
303 
304  /// Find set of variables with maximal domain size
305  carl::Variables maxVariables;
306  DomainSize maxDomainSize{DomainSize::SMALL};
307  for (const auto& exponent : monomial->exponents())
308  {
309  const carl::Variable& variable{exponent.first};
310  auto expansionIter{mExpansions.firstFind(variable)};
311  if (expansionIter == mExpansions.end())
312  expansionIter = mExpansions.emplace(variable);
313  Expansion& expansion{*expansionIter};
314 
315  if (maxDomainSize <= expansion.mMaximalDomainSize)
316  {
317  if (maxDomainSize < expansion.mMaximalDomainSize)
318  {
319  maxVariables.clear();
320  maxDomainSize = expansion.mMaximalDomainSize;
321  }
322  maxVariables.emplace(variable);
323  }
324  }
325 
326  /// Find a locally optimal reduction for the monomial
327  const auto isReducible = [&](const auto& purificationsEntry) {
328  return purificationsEntry.second.mActive
329  && monomial->divisible(purificationsEntry.first)
330  && std::any_of(
331  maxVariables.begin(),
332  maxVariables.end(),
333  [&](const carl::Variable& variable) {
334  return purificationsEntry.first->has(variable);
335  }
336  );
337  };
338  auto reductionIter{std::find_if(std::make_reverse_iterator(purificationIter), mPurifications.rend(), isReducible)};
339 
340  /// Activate the sequence of reductions top-down
341  carl::Monomial::Arg guidance;
342  if (reductionIter == mPurifications.rend())
343  monomial->divide(*maxVariables.begin(), guidance);
344  else
345  monomial->divide(reductionIter->first, guidance);
346  auto hintIter{purificationIter};
347  for (const auto& exponentPair : guidance->exponents())
348  {
349  const carl::Variable& variable{exponentPair.first};
350  Expansion& expansion{mExpansions.firstAt(variable)};
351  for (carl::exponent exponent = 1; exponent <= exponentPair.second; ++exponent)
352  {
353  hintIter->second.mActive = true;
354  expansion.mPurifications.emplace_back(&hintIter->second);
355  monomial->divide(variable, monomial);
356  if (monomial->isAtMostLinear())
357  hintIter->second.mReduction = mExpansions.firstAt(monomial->getSingleVariable()).mQuotients[0];
358  else
359  {
360  auto temp{mPurifications.emplace_hint(hintIter, std::piecewise_construct, std::make_tuple(monomial), std::make_tuple())};
361  hintIter->second.mReduction = temp->second.mSubstitutions[0];
362  hintIter = temp;
363  }
364  }
365  }
366  }
367  else
368  purification.mActive = false;
369  }
370 
371  /// Activate expansions that are used for case splits and deactivate them otherwise
372  for (Expansion& expansion : mExpansions)
373  {
374  /// Calculate the center point where the initial domain is located
375  expansion.mNucleus = 0;
376  if (expansion.mMaximalDomain.lowerBoundType() != carl::BoundType::INFTY
377  && expansion.mNucleus < expansion.mMaximalDomain.lower())
378  expansion.mNucleus = expansion.mMaximalDomain.lower();
379  else if (expansion.mMaximalDomain.upperBoundType() != carl::BoundType::INFTY
380  && expansion.mNucleus > expansion.mMaximalDomain.upper())
381  expansion.mNucleus = expansion.mMaximalDomain.upper();
382 
383  /// Calculate and activate the corresponding domain
384  RationalInterval domain(0, 1);
385  domain.mul_assign(carl::Interval<Rational>(Rational(Settings::initialRadius)));
386  domain.add_assign(carl::Interval<Rational>(expansion.mNucleus));
387  domain = carl::set_intersection(domain, expansion.mMaximalDomain);
388  changeActiveDomain(expansion, std::move(domain));
389  }
390 
391  return true;
392  }

394  template<class Settings>
395  bool CSplitModule<Settings>::bloatDomains(const FormulaSetT& LRAConflict)
396  {
397  /// Data structure for potential bloating candidates
398  struct Candidate
399  {
400  Expansion& mExpansion;
401  const Rational mDirection;
402  const Rational mRadius;
403 
404  Candidate(Expansion& expansion, Rational&& direction, Rational&& radius)
405  : mExpansion(expansion)
406  , mDirection(std::move(direction))
407  , mRadius(std::move(radius))
408  {}
409 
410  bool operator<(const Candidate& rhs) const
411  {
412  if (carl::isOne(mDirection*rhs.mDirection))
413  return mRadius < rhs.mRadius;
414  else if (carl::isOne(mDirection))
415  return mRadius < Rational(Settings::thresholdRadius);
416  else
417  return rhs.mRadius >= Rational(Settings::thresholdRadius);
418  }
419  };
420  std::set<Candidate> candidates;
421 
422  /// Scan the infeasible subset of the LRA solver for potential candidates
423  for (const FormulaT& formula : LRAConflict)
424  if (formula.isBound())
425  {
426  const ConstraintT& constraint{formula.constraint()};
427  const carl::Variable variable{constraint.variables().as_vector().front()};
428  auto expansionIter{mExpansions.secondFind(variable)};
429  if (expansionIter != mExpansions.end())
430  {
431  Expansion& expansion{*expansionIter};
432  Rational direction;
433  if (constraint.isLowerBound()
434  && (expansion.mMaximalDomain.lowerBoundType() == carl::BoundType::INFTY
435  || expansion.mMaximalDomain.lower() < expansion.mActiveDomain.lower()))
436  direction = -1;
437  else if (constraint.isUpperBound()
438  && (expansion.mMaximalDomain.upperBoundType() == carl::BoundType::INFTY
439  || expansion.mMaximalDomain.upper() > expansion.mActiveDomain.upper()))
440  direction = 1;
441  if (!carl::isZero(direction))
442  {
443  Rational radius{(direction*(expansion.mActiveDomain-expansion.mNucleus)).upper()};
444  if (radius <= Settings::maximalRadius)
445  {
446  candidates.emplace(expansion, std::move(direction), std::move(radius));
447  if (candidates.size() > Settings::maxBloatedDomains)
448  candidates.erase(std::prev(candidates.end()));
449  }
450  }
451  }
452  }
453 
454  /// Change the active domain of the candidates with highest priority
455  for (const Candidate& candidate : candidates)
456  {
457  RationalInterval domain;
458  if (candidate.mRadius <= Settings::thresholdRadius)
459  domain = RationalInterval(0, 1);
460  else if (candidate.mExpansion.mPurifications.empty())
461  domain = RationalInterval(0, carl::BoundType::WEAK, 0, carl::BoundType::INFTY);
462  else
463  domain = RationalInterval(0, candidate.mRadius);
464  domain.mul_assign(carl::Interval<Rational>(candidate.mDirection));
465  domain.add_assign(candidate.mExpansion.mActiveDomain);
466  domain = carl::set_intersection(domain, candidate.mExpansion.mMaximalDomain);
467  changeActiveDomain(candidate.mExpansion, std::move(domain));
468  }
469 
470  /// Report if any variable domain was bloated
471  return candidates.empty();
472  }

474  template<class Settings>
475  Answer CSplitModule<Settings>::analyzeConflict(const FormulaSetT& LRAConflict)
476  {
477  FormulaSetT infeasibleSubset;
478  for (const FormulaT& formula : LRAConflict)
479  {
480  if (formula.isBound())
481  {
482  auto expansionIter{mExpansions.secondFind(*formula.variables().begin())};
483  if (expansionIter != mExpansions.end())
484  {
485  const Expansion& expansion{*expansionIter};
486  if (expansion.mRationalization.type() == carl::VariableType::VT_REAL
487  || expansion.mMaximalDomain != expansion.mActiveDomain)
488  return Answer::UNKNOWN;
489  else
490  {
491  FormulaSetT boundOrigins{mVariableBounds.getOriginSetOfBounds(expansion.mRationalization)};
492  infeasibleSubset.insert(boundOrigins.begin(), boundOrigins.end());
493  }
494  }
495  }
496  else if (formula.getType() == carl::FormulaType::CONSTRAINT)
497  {
498  const ConstraintT& constraint{formula.constraint()};
499  auto linearizationIter{mLinearizations.secondFind(constraint.lhs().normalize())};
500  if (linearizationIter != mLinearizations.end())
501  {
502  const Linearization& linearization{*linearizationIter};
503  if (linearization.mHasRealVariables)
504  return Answer::UNKNOWN;
505  else
506  {
507  carl::Relation relation{constraint.relation()};
508  if (carl::isNegative(constraint.lhs().lcoeff()))
509  relation = carl::turn_around(relation);
510  infeasibleSubset.emplace(linearization.mNormalization, relation);
511  }
512  }
513  }
514  }
515  mInfeasibleSubsets.emplace_back(std::move(infeasibleSubset));
516  return Answer::UNSAT;
517  }

519  template<class Settings>
520  void CSplitModule<Settings>::changeActiveDomain(Expansion& expansion, RationalInterval&& domain)
521  {
522  RationalInterval activeDomain{std::move(expansion.mActiveDomain)};
523  expansion.mActiveDomain = domain;
524 
525  /// Update the variable bounds
526  if (!activeDomain.isEmpty())
527  {
528  if (activeDomain.lowerBoundType() != carl::BoundType::INFTY
529  && (domain.lowerBoundType() == carl::BoundType::INFTY
530  || domain.lower() != activeDomain.lower()
531  || domain.isEmpty()))
532  propagateFormula(FormulaT(expansion.mQuotients[0]-Poly(activeDomain.lower()), carl::Relation::GEQ), false);
533  if (activeDomain.upperBoundType() != carl::BoundType::INFTY
534  && (domain.upperBoundType() == carl::BoundType::INFTY
535  || domain.upper() != activeDomain.upper()
536  || domain.isEmpty()))
537  propagateFormula(FormulaT(expansion.mQuotients[0]-Poly(activeDomain.upper()), carl::Relation::LEQ), false);
538  }
539  if (!domain.isEmpty())
540  {
541  if (domain.lowerBoundType() != carl::BoundType::INFTY
542  && (activeDomain.lowerBoundType() == carl::BoundType::INFTY
543  || activeDomain.lower() != domain.lower()
544  || activeDomain.isEmpty()))
545  propagateFormula(FormulaT(expansion.mQuotients[0]-Poly(domain.lower()), carl::Relation::GEQ), true);
546  if (domain.upperBoundType() != carl::BoundType::INFTY
547  && (activeDomain.upperBoundType() == carl::BoundType::INFTY
548  || activeDomain.upper() != domain.upper()
549  || activeDomain.isEmpty()))
550  propagateFormula(FormulaT(expansion.mQuotients[0]-Poly(domain.upper()), carl::Relation::LEQ), true);
551  }
552 
553  /// Check if the digits of the expansion need to be encoded
554  if (expansion.mPurifications.empty())
555  {
556  activeDomain = RationalInterval::emptyInterval();
557  domain = RationalInterval::emptyInterval();
558  }
559 
560  /// Update the case splits of the corresponding digits
561  for (size_t i = 0; activeDomain != domain; ++i)
562  {
563  if (domain.diameter() <= Settings::maxDomainSize)
564  {
565  /// Update the currently active linear encoding
566  Rational lower{activeDomain.isEmpty() ? domain.lower() : activeDomain.lower()};
567  Rational upper{activeDomain.isEmpty() ? domain.lower() : activeDomain.upper()+1};
568  for (const Purification *purification : expansion.mPurifications)
569  {
570  for (Rational alpha = domain.lower(); alpha < lower; ++alpha)
571  propagateFormula(
572  FormulaT(
573  carl::FormulaType::IMPLIES,
574  FormulaT(Poly(expansion.mQuotients[i])-Poly(alpha), carl::Relation::EQ),
575  FormulaT(Poly(purification->mSubstitutions[i])-Poly(alpha)*Poly(purification->mReduction), carl::Relation::EQ)
576  ),
577  true
578  );
579  for (Rational alpha = upper; alpha <= domain.upper(); ++alpha)
580  propagateFormula(
581  FormulaT(
582  carl::FormulaType::IMPLIES,
583  FormulaT(Poly(expansion.mQuotients[i])-Poly(alpha), carl::Relation::EQ),
584  FormulaT(Poly(purification->mSubstitutions[i])-Poly(alpha)*Poly(purification->mReduction), carl::Relation::EQ)
585  ),
586  true
587  );
588  }
589  }
590  else if (activeDomain.diameter() <= Settings::maxDomainSize)
591  {
592  /// Switch from the linear to a logarithmic encoding
593  if (expansion.mQuotients.size() <= i+1)
594  {
595  expansion.mQuotients.emplace_back(carl::freshIntegerVariable());
596  expansion.mRemainders.emplace_back(carl::freshIntegerVariable());
597  }
598  for (Purification *purification : expansion.mPurifications)
599  {
600  if (purification->mSubstitutions.size() <= i+1)
601  purification->mSubstitutions.emplace_back(carl::freshIntegerVariable());
602  for (Rational alpha = activeDomain.lower(); alpha <= activeDomain.upper(); ++alpha)
603  propagateFormula(
604  FormulaT(
605  carl::FormulaType::IMPLIES,
606  FormulaT(Poly(expansion.mQuotients[i])-Poly(alpha), carl::Relation::EQ),
607  FormulaT(Poly(purification->mSubstitutions[i])-Poly(alpha)*Poly(purification->mReduction), carl::Relation::EQ)
608  ),
609  false
610  );
611  for (Rational alpha = 0; alpha < Settings::expansionBase; ++alpha)
612  propagateFormula(
613  FormulaT(

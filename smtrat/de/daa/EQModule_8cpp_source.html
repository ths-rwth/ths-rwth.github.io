39  bool EQModule<Settings>::informCore(const FormulaT& _constraint)
40  {
41  if(_constraint.getType() == carl::UEQ) {
42  const UEquality &ueq = _constraint.uequality();
43  const term_type &lhs = ueq.lhs();
44  const term_type &rhs = ueq.rhs();
45 
46  if(ueq.negated()) {
47  if(lhs == rhs) {
48  return false;
49  }
50  }
51 
52  if(Settings::deduceUnassignedLiterals) {
53  // add the terms to our map if not present
54  g_iterator itrLhs; bool insertedLhs;
55  g_iterator itrRhs; bool insertedRhs;
56  std::tie(itrLhs, insertedLhs) = mEqualityGraph.emplace(std::piecewise_construct, std::forward_as_tuple(lhs), std::forward_as_tuple(*this));
57  std::tie(itrRhs, insertedRhs) = mEqualityGraph.emplace(std::piecewise_construct, std::forward_as_tuple(rhs), std::forward_as_tuple(*this));
58 
59  if(insertedLhs) {
60  if(ueq.lhs().isUFInstance()) {
61  P_add_function_instance(ueq.lhs().asUFInstance(), itrLhs);
62  } else {
63  P_add_variable(ueq.lhs().asUVariable(), itrLhs);
64  }
65  }
66 
67  if(insertedRhs) {
68  if(ueq.rhs().isUFInstance()) {
69  P_add_function_instance(ueq.rhs().asUFInstance(), itrRhs);
70  } else {
71  P_add_variable(ueq.rhs().asUVariable(), itrRhs);
72  }
73  }
74 
75  // add term to normalization map; make sure we do not produce literals that did not occur in the input
76  P_add_not_asserted_normalized(_constraint, ueq);
77  }
78  }
79 
80  return true;
81  }

175  template<class Settings>
176  bool EQModule<Settings>::addCore(ModuleInput::const_iterator _subformula)
177  {
178  if(Settings::printFormulas) {
179  std::cout << "assertSubformula: " << _subformula->formula() << std::endl;
180  }
181 
182  if(_subformula->formula().getType() != carl::UEQ) {
183  ++mCountNonUEQFormulas;
184  return true;
185  }
186 
187  const FormulaT& formula = _subformula->formula();
188  const UEquality &ueq = formula.uequality();
189  const term_type &lhs = ueq.lhs();
190  const term_type &rhs = ueq.rhs();
191 
192  // in any case, add the terms to our map if not present (even for an inequality, otherwise we have no iterators)
193  g_iterator itrLhs;
194  g_iterator itrRhs;
195  itrLhs = mEqualityGraph.find(lhs);
196  itrRhs = mEqualityGraph.find(rhs);
197 
198  if(itrLhs == mEqualityGraph.end()) {
199  itrLhs = mEqualityGraph.emplace(std::piecewise_construct, std::forward_as_tuple(lhs), std::forward_as_tuple(*this)).first;
200  if(ueq.lhs().isUFInstance()) {
201  P_add_function_instance(ueq.lhs().asUFInstance(), itrLhs);
202  } else {
203  P_add_variable(ueq.lhs().asUVariable(), itrLhs);
204  }
205  }
206 
207  if(itrRhs == mEqualityGraph.end()) {
208  itrRhs = mEqualityGraph.emplace(std::piecewise_construct, std::forward_as_tuple(rhs), std::forward_as_tuple(*this)).first;
209  if(ueq.rhs().isUFInstance()) {
210  P_add_function_instance(ueq.rhs().asUFInstance(), itrRhs);
211  } else {
212  P_add_variable(ueq.rhs().asUVariable(), itrRhs);
213  }
214  }
215 
216  if(Settings::deduceUnassignedLiterals) {
217  mAllNotAssertedEqualities.erase(not_asserted_equality(mEqualityNormalization.find(formula)->second));
218  }
219 
220  bool found_inconsistency = false;
221 
222  if(ueq.negated()) {
223  std::size_t indexLhs = mUnionFind.find(itrLhs->second.mUFIndex);
224  std::size_t indexRhs = mUnionFind.find(itrRhs->second.mUFIndex);
225 
226  std::size_t indexCLhs = mComponentUnionFind.find(indexLhs);
227  std::size_t indexCRhs = mComponentUnionFind.find(indexRhs);
228 
229  if(indexCLhs == indexCRhs) {
230  // construct edges for inequality
231  ineq_edge_info *lhs_edge = &mIneqEdgeAlloc.emplace(
232  mUnionFind[indexCLhs], mUnionFind[indexCRhs], itrLhs, itrRhs, formula
233  );
234  ineq_edge_info *rhs_edge = &mIneqEdgeAlloc.emplace(
235  mUnionFind[indexCRhs], mUnionFind[indexCLhs], itrRhs, itrLhs, formula
236  );
237 
238  lhs_edge->mOther = rhs_edge;
239  rhs_edge->mOther = lhs_edge;
240 
241  // add inequality to inconsistencies since lhs and rhs are in same component
242  mPossibleInconsistencies.push_back(lhs_edge);
243  found_inconsistency = true;
244 
245  if(Settings::printFormulas) {
246  std::cout << "Found inconsistency: " << lhs_edge->mFormula << std::endl;
247  }
248  } else {
249  P_add_ineq_edge(indexCLhs, indexCRhs, itrLhs, itrRhs, formula);
250  }
251  } else {
252  P_add_explicit_edge(itrLhs, itrRhs, formula);
253 
254  std::size_t indexLhs = mUnionFind.find(itrLhs->second.mUFIndex);
255  std::size_t indexRhs = mUnionFind.find(itrRhs->second.mUFIndex);
256 
257  std::size_t indexCLhs = mComponentUnionFind.find(indexLhs);
258  std::size_t indexCRhs = mComponentUnionFind.find(indexRhs);
259 
260  // we only have to do something if we actually merge two components
261  if(indexLhs != indexRhs) {
262  std::size_t indexNew = mUnionFind.fast_union(indexLhs, indexRhs);
263  std::size_t indexCNew = mComponentUnionFind.find(indexNew);
264  std::size_t indexOld = (indexNew == indexLhs) ? indexRhs : indexLhs;
265  std::size_t indexCOld = (indexCNew == indexCLhs) ? indexCRhs : indexCLhs;
266 
267  if(indexCLhs == indexCRhs) {
268  P_bfs_search_implicit(mUnionFind[indexLhs], mEqualityGraph.end());
269  assert(mUnionFind[indexRhs]->second.mVisited);
270  assert(mUnionFind[indexCLhs]->second.mVisited);
271 
272  // in case the two components are in the same class we have to delete some edge on the path from CLhs to CRhs to not introduce a cycle
273  // to make sure we do not delete an edge that is necessary to prove another implicit edge, we remove the newest one, according to its mID
274  g_iterator cur = mUnionFind[indexRhs];
275  implicit_edge_info* maxEdge = cur->second.mImplicitPred;
276 
277  while (true) {
278  if (cur->second.mImplicitPred->mPred == mUnionFind[indexLhs]) break;
279  if (cur->second.mImplicitPred->mID > maxEdge->mID) {
280  maxEdge = cur->second.mImplicitPred;
281  }
282  cur = cur->second.mImplicitPred->mPred;
283  }
284  P_destroy_implicit_edge(maxEdge);
285 
286  if(indexCLhs == indexOld) {
287  // if old index was component component representative we have to update mComponentUnionFind, hash buckets and and inequality edges
288  for(g_iterator entry : mEqualityComponent){
289  mComponentUnionFind.set_class(entry->second.mUFIndex, indexNew);
290  }
291  mComponentUnionFind.set_class(indexOld, indexOld);
292 
293  P_merge_buckets(indexCLhs, indexNew);
294  P_cc_union_for_ineq(indexCLhs, indexNew);
295  } else {
296  // indexOld might appear on some path in the component union find structure which can cause problems
297  for(g_iterator entry : mEqualityComponent){
298  mComponentUnionFind.set_class(entry->second.mUFIndex, indexCLhs);
299  }
300  mComponentUnionFind.set_class(indexOld, indexOld);
301  }
302 
303  P_clear_bfs_markings();
304 
305  g_iterator nodeOld = mUnionFind[indexOld];
306  g_iterator nodeNew = mUnionFind[indexNew];
307 
308  // implicit neighbors of the old component representative have to be moved to the new representative
309  for(std::size_t i = nodeOld->second.mImplicit.size() - 1; i < nodeOld->second.mImplicit.size(); i--) {
310  implicit_edge_info *edge = nodeOld->second.mImplicit[i];
311  nodeOld->second.mImplicit.remove_back();
312  nodeNew->second.mImplicit.add(edge);
313  edge->mPred = nodeNew;
314  edge->mOther->mSucc = nodeNew;
315  }
316  } else {
317  // union two different component components and check for inconsistencies
318  found_inconsistency = P_cc_union_for_ineq(indexCOld, indexCNew);
319 
320  P_bfs_search_implicit(mUnionFind[indexOld], mEqualityGraph.end());
321  assert(mUnionFind[indexCOld]->second.mVisited);
322 
323  // update component union find; we can not just do a union here since indexOld might appear on some component union find path
324  for(g_iterator entry : mEqualityComponent){
325  mComponentUnionFind.set_class(entry->second.mUFIndex, indexCNew);
326  }
327  mComponentUnionFind.set_class(indexOld, indexOld);
328 
329  P_clear_bfs_markings();
330 
331  g_iterator nodeOld = mUnionFind[indexOld];
332  g_iterator nodeNew = mUnionFind[indexNew];
333 
334  // implicit neighbors of the old component representative have to be moved to the new representative
335  for(std::size_t i = nodeOld->second.mImplicit.size() - 1; i < nodeOld->second.mImplicit.size(); i--) {
336  implicit_edge_info *edge = nodeOld->second.mImplicit[i];
337  nodeOld->second.mImplicit.remove_back();
338  edge->mPred = nodeNew;
339  edge->mOther->mSucc = nodeNew;
340  nodeNew->second.mImplicit.add(edge);
341  }
342 
343  P_merge_buckets(indexCOld, indexCNew);
344  }
345  }
346  }
347 
348  if(Settings::printBucketSets) {
349  P_print_bucket_sets();
350  }
351 
352  // construct infeasible subset in case inconsistency is found
353  if(found_inconsistency) {
354  ineq_edge_info* ineq_edge = mPossibleInconsistencies.back();
355  P_construct_infeasible_subset(ineq_edge->mRealPred, ineq_edge->mRealSucc, ineq_edge->mFormula);
356  return false;
357  }
358 
359  return true;
360  }

364  {

366  std::cout << "removeSubformula: " << _subformula->formula() << std::endl;
367  }

372  }

375  const UEquality &ueq = formula.uequality();
376  const term_type &lhs = ueq.lhs();
377  const term_type &rhs = ueq.rhs();
378 
379  g_iterator itrLhs = mEqualityGraph.find(lhs);
380  g_iterator itrRhs = mEqualityGraph.find(rhs);
381 
382  assert(itrLhs != mEqualityGraph.end());
383  assert(itrRhs != mEqualityGraph.end());
384 
385  if(Settings::deduceUnassignedLiterals) {
386  mAllNotAssertedEqualities.emplace(mEqualityNormalization.find(formula)->second, itrLhs, itrRhs);
387  }
388 
389  if(ueq.negated()) {
390  // we only have to remove the inequality edge in case of an inequality
391  P_remove_ineq_edge(itrLhs, itrRhs, formula);
392  } else {
393  P_remove_edge(itrLhs, itrRhs, formula);
394 
395  // we only have to do something if the component splits
396  if(!P_bfs_search_explicit(itrLhs, itrRhs)) {
397  // first we have to split up the lower component and adjust the implicit neighbors
398 
399  std::size_t oldcentry = mUnionFind.find(itrLhs->second.mUFIndex);
400  g_iterator oldcnode = mUnionFind[oldcentry];
401  g_iterator newcnode;
402 
403  // if old component entry is on left side, keep it on left side, otherwise keep it on right side
404  if(oldcnode->second.mVisited) {
405  std::size_t split_pos = mEqualityComponent.size();
406  if(P_bfs_search_explicit(itrRhs, itrLhs)) {
407  assert(false);
408  }
409 
410  // inform hash buckets about changing indizes
411  for(g_iterator entry : mEqualityComponent) {
412  if(entry->second.mArgs != nullptr) {
413  hash_bucket_type* bucket = bucketsof(entry)->find(entry);
414  if(!bucket->mEntries.trivial()) {
415  bucketsof(entry)->mChanged.add_front(bucket);
416  }
417  }
418  }
419 
420  // split classes in union find
421  for(std::size_t i = 0; i < split_pos; i++){
422  g_iterator entry = mEqualityComponent[i];
423  entry->second.mVisited = false;
424  mUnionFind.set_class(entry->second.mUFIndex, oldcentry);
425  }
426 
427  newcnode = itrRhs;
428  for(std::size_t i = split_pos; i < mEqualityComponent.size(); i++) {
429  g_iterator entry = mEqualityComponent[i];
430  mUnionFind.set_class(entry->second.mUFIndex, newcnode->second.mUFIndex);
431  }
432 
433  // split implicit neighbors of the old representative
434  for(std::size_t i = oldcnode->second.mImplicit.size() - 1; i < oldcnode->second.mImplicit.size(); i--) {
435  implicit_edge_info *edge = oldcnode->second.mImplicit[i];
436  if(edge->mRealPred->second.mVisited) {
437  oldcnode->second.mImplicit.remove(edge);
438  edge->mPred = newcnode;
439  edge->mOther->mSucc = newcnode;
440  newcnode->second.mImplicit.add(edge);
441  }
442  }
443  } else {
444  std::size_t split_pos = mEqualityComponent.size();
445  if(P_bfs_search_explicit(itrRhs, itrLhs)) {
446  assert(false);
447  }
448 
449  // inform hash buckets about changing indizes
450  for(g_iterator entry : mEqualityComponent) {
451  if(entry->second.mArgs != nullptr) {
452  hash_bucket_type* bucket = bucketsof(entry)->find(entry);
453  if(!bucket->mEntries.trivial()) {
454  bucketsof(entry)->mChanged.add_front(bucket);
455  }
456  }
457  }
458 
459  // split classes in union find
460  newcnode = itrLhs;
461  for(std::size_t i = 0; i < split_pos; i++){
462  g_iterator entry = mEqualityComponent[i];
463  mUnionFind.set_class(entry->second.mUFIndex, newcnode->second.mUFIndex);
464  }
465 
466  for(std::size_t i = split_pos; i < mEqualityComponent.size(); i++) {
467  g_iterator entry = mEqualityComponent[i];
468  entry->second.mVisited = false;
469  mUnionFind.set_class(entry->second.mUFIndex, oldcentry);
470  }
471 
472  // split implicit neighbors of the old representative
473  for(std::size_t i = oldcnode->second.mImplicit.size() - 1; i < oldcnode->second.mImplicit.size(); i--) {
474  implicit_edge_info *edge = oldcnode->second.mImplicit[i];
475  if(edge->mRealPred->second.mVisited) {
476  oldcnode->second.mImplicit.remove(edge);
477  edge->mPred = newcnode;
478  edge->mOther->mSucc = newcnode;
479  newcnode->second.mImplicit.add(edge);
480  }
481  }
482  }
483 
484  P_clear_bfs_markings();
485 
486  // split the upper component and update component union find
487 
488  std::size_t oldccentry = mComponentUnionFind.find(oldcentry);
489 
490  if(P_bfs_search_implicit(oldcnode, newcnode)) {
491  assert(false);
492  }
493  g_iterator oldccnode = mUnionFind[oldccentry];
494  g_iterator newccnode;
495 
496  // all inequality neighbors have to checked; instead we remove them all and add them to possible inconsistencies so they are reinserted at a later point all together
497  P_remove_ineq_edges_of_vertex(oldccnode);
498 
499  // if old component entry is on left side, keep it on left side, otherwise keep it on right side
500  if(oldccnode->second.mVisited) {
501  for(g_iterator entry : mEqualityComponent) {
502  mComponentUnionFind.set_class(entry->second.mUFIndex, oldccentry);
503  }
504  P_clear_bfs_markings();
505

507  assert(false);
508  }
509

511  for(g_iterator entry : mEqualityComponent) {
512  mComponentUnionFind.set_class(entry->second.mUFIndex, newccnode->second.mUFIndex);
513  }
514 
515  } else {
516  newccnode = oldcnode;
517  for(g_iterator entry : mEqualityComponent) {
518  mComponentUnionFind.set_class(entry->second.mUFIndex, newccnode->second.mUFIndex);
519  }

524  }
525

529  }

866  ++j;
867 
868  for(auto k = last; k > j; --k) {

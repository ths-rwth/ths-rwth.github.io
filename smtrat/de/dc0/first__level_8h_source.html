107  void insert_root(const RAN& ran, const indexed_root&& root) {
108  assert(m_samples.empty());
109  auto it = m_root_map.find(ran);
110  if (it == m_root_map.end()) {
111  m_root_map.emplace(ran, std::vector<indexed_root>({root}));
112  m_roots.emplace_back(ran);
113  } else {
114  it->second.emplace_back(std::move(root));

118  void process() {
119  assert(m_samples.empty());
120  std::sort(m_roots.begin(), m_roots.end());
121  m_samples.reserve(2 * m_roots.size() + 1);
122  for (std::size_t n = 0; n < m_roots.size(); n++) {
123  if (n == 0) m_samples.emplace_back(carl::sample_below(m_roots.front()), std::nullopt, &m_roots.front());
124  else m_samples.emplace_back(carl::sample_between(m_roots[n-1], m_roots[n]), &m_roots[n-1], &m_roots[n]);
125  m_samples.emplace_back(m_roots[n], &m_roots[n], &m_roots[n]);
126  }
127  if (m_roots.empty()) m_samples.emplace_back(RAN(0), std::nullopt, std::nullopt);
128  else m_samples.emplace_back(carl::sample_above(m_roots.back()), &m_roots.back(), std::nullopt);
129  }

191 bool compute_unsat_intervals(const VariableComparisonT& constr, const Model& model, carl::Variable variable, std::vector<cell_at_level>& results) {
192  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Compute unsat intervals of " << constr);
193 
194  if (constr.var() != variable) {
195  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Does not have variable");
196  return false;
197  }
198  carl::carlVariables vars;
199  constr.gatherVariables(vars);
200  if (!var_subset(vars, model, variable)) {
201  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Contains unassigned variables");
202  return false;
203  }
204  if (carl::ran::interval::vanishes(carl::to_univariate_polynomial(constr.definingPolynomial(), variable), as_ran_map(model))) {
205  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Vanishes");
206  return false;
207  }
208  auto eval = carl::model::evaluate(FormulaT(constr), model);
209  if (eval.isBool()) {
210  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Already evaluates to a constant"); // TODO is this case possible?
211  return false;
212  }
213 
214  assert(carl::irreducibleFactors(constr.definingPolynomial(), false).size() == 1);
215 
216  RAN ran;
217  indexed_root ir;
218  if (std::holds_alternative<RAN>(constr.value())) {
219  ran = std::get<RAN>(constr.value());
220  size_t index = 1; // TODO determine index, not relevant for covering
221  ir = indexed_root({ constr.definingPolynomial(), index });
222  } else {
223  assert(std::holds_alternative<MultivariateRootT>(constr.value()));
224  ir = indexed_root({ constr.definingPolynomial(), std::get<MultivariateRootT>(constr.value()).k() });
225 
226  auto res = std::get<MultivariateRootT>(constr.value()).evaluate(as_ran_map(model));
227  assert(res);
228  ran = *res;
229  }
230 
231  auto relation = constr.negated() ? carl::inverse(constr.relation()) : constr.relation();
232  bool point = relation == carl::Relation::GREATER || relation == carl::Relation::LESS || relation == carl::Relation::NEQ;
233  bool below = relation == carl::Relation::GREATER || relation == carl::Relation::GEQ || relation == carl::Relation::EQ;
234  bool above = relation == carl::Relation::LESS || relation == carl::Relation::LEQ || relation == carl::Relation::EQ;
235 
236  if (point) {
237  section sec;
238  sec.current_level.push_back(ir.poly);
239  sec.point = ran;
240  sec.point_defining_poly = ir;
241  results.push_back(sec);
242  }
243  if (below) {
244  sector sec;
245  sec.current_level.push_back(ir.poly);
246  sec.right = ran;
247  sec.right_defining_poly = ir;
248  results.push_back(sec);
249  }
250  if (above) {
251  sector sec;
252  sec.current_level.push_back(ir.poly);
253  sec.left = ran;
254  sec.left_defining_poly = ir;
255  results.push_back(sec);
256  }
257  return true;
258 }

260 bool compute_unsat_intervals(const ConstraintT& constr, const Model& model, carl::Variable variable, std::vector<cell_at_level>& results) {
261  assert(!carl::is_zero(constr.lhs()));
262  assert(!carl::is_constant(constr.lhs()));
263 
264  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Compute unsat intervals of " << constr);
265 
266  if (!constr.lhs().has(variable)) {
267  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Does not have variable");
268  return false;
269  }
270  if (!var_subset(constr.variables(), model, variable)) {
271  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Contains unassigned variables");
272  return false;
273  }
274  if (carl::ran::interval::vanishes(carl::to_univariate_polynomial(constr.lhs(), variable), as_ran_map(model))) {
275  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Vanishes");
276  return false;
277  }
278  // this is stronger than in CAC / OneCell: (not only nullification)
279  auto eval = carl::model::evaluate(constr, model);
280  if (eval.isBool()) {
281  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Already evaluates to a constant");
282  if (!eval.asBool()) {
283  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Already evaluates to false");
284  sector sec;
285  auto factors = carl::irreducibleFactors(constr.lhs(), false);
286  for (const auto& f : factors) {
287  if (!f.has(variable)) {
288  sec.lower_level.push_back(f);
289  } else {
290  sec.current_level.push_back(f);
291  }
292  }
293  results.push_back(sec);
294  }
295  return true;
296  }
297 
298  std::vector<Poly> lower_level;
299  std::vector<Poly> current_level;
300 
301  root_indexer ri;
302  auto factors = carl::irreducibleFactors(constr.lhs(), false);
303  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Computed factors " << factors);
304 
305  for (const auto& f : factors) {
306  if (!f.has(variable)) {
307  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Factor " << f << " is of lower level");
308  lower_level.push_back(f);
309  } else {
310  current_level.push_back(f);
311  auto f_roots = carl::model::real_roots(f, variable, model);
312  assert(f_roots.is_univariate());
313  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Factor " << f << " has roots " << f_roots.roots());
314  for (size_t index = 0; index < f_roots.roots().size(); index++) {
315  ri.insert_root(f_roots.roots()[index], indexed_root({f, index}));
316  }
317  }
318  }
319 
320  ri.process();
321  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Got " << ri);
322  for (const auto& s : ri.samples()) {
323  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Plugging in sample " << s);
324  if (s.is_point()) {
325  if (!carl::evaluate(carl::Sign::ZERO, constr.relation())) {
326  section sec;
327  sec.lower_level = lower_level;
328  sec.current_level = current_level;
329  {
330  sec.point = s.val;
331  const auto& indexed_roots = ri.indexed_roots_at(sec.point);
332  sec.point_defining_poly = indexed_roots.front();
333  }
334  for (const auto& os : ri.roots()) {
335  const auto& indexed_roots = ri.indexed_roots_at(os);
336  for (auto o = indexed_roots.begin()+1; o != indexed_roots.end(); o++) {
337  if (o->poly != sec.point_defining_poly.poly) {
338  sec.others.insert(o->poly);
339  }
340  }
341  }
342  results.push_back(sec);
343  }
344  } else {
345  Model extended_model = model;
346  extended_model.assign(variable, s.val);
347  auto res = carl::model::evaluate(constr, extended_model);
348  assert(res.isBool());
349  if (!res.asBool()) {
350  sector sec;
351  sec.lower_level = lower_level;
352  sec.current_level = current_level;
353  if (s.left) {
354  sec.left = **s.left;
355  const auto& indexed_roots = ri.indexed_roots_at(*sec.left);
356  sec.left_defining_poly = indexed_roots.front();
357  }
358  if (s.right) {
359  sec.right = **s.right;
360  const auto& indexed_roots = ri.indexed_roots_at(*sec.right);
361  sec.right_defining_poly = indexed_roots.front();
362  }
363  bool left_of_sector = (bool) s.left;
364  for (const auto& os : ri.roots()) {
365  const auto& indexed_roots = ri.indexed_roots_at(os);
366  for (auto o = indexed_roots.begin(); o != indexed_roots.end(); o++) {
367  if (left_of_sector) {
368  assert(s.left);
369  if (o->poly != sec.left_defining_poly->poly) {
370  sec.others_left.insert(o->poly);
371  }
372  } else {
373  assert(s.right);
374  if (o->poly != sec.right_defining_poly->poly) {
375  sec.others_right.insert(o->poly);
376  }
377  }
378  }
379  if (os == sec.left) {
380  left_of_sector = false;
381  }
382  }
383  results.push_back(sec);
384  }
385  }
386  }
387  return true;
388 }

404 void insert_projection_result(const Poly& respoly, InvarianceType type, projection_results& results) {
405  assert(!carl::is_zero(respoly));
406  if (carl::is_constant(respoly)) return;
407  for (const auto& f : carl::irreducibleFactors(respoly, false)) {
408  if (!cad::projection::doesNotVanish(f)) {
409  auto entry = results.find(f);
410  if (entry != results.end()) {
411  if (entry->second == InvarianceType::SIGN_INV && type == InvarianceType::ORD_INV) {
412  entry->second = InvarianceType::ORD_INV;
413  }
414  } else {
415  results.emplace(f, type);
416  }
417  }
418  }

563  std::vector<const cell_at_level*> sorted_cells;
564  for (const auto& cell : cells) sorted_cells.emplace_back(&cell);
565  std::sort(sorted_cells.begin(), sorted_cells.end(), [](const cell_at_level* p_cell1, const cell_at_level* p_cell2) { // cell1 < cell2
566  const auto& cell1 = *p_cell1;
567  const auto& cell2 = *p_cell2;
568  bool cell1_has_left = !(std::holds_alternative<sector>(cell1) && !std::get<sector>(cell1).left);
569  bool cell1_has_right = !(std::holds_alternative<sector>(cell1) && !std::get<sector>(cell1).right);
570  bool cell2_has_left = !(std::holds_alternative<sector>(cell2) && !std::get<sector>(cell2).left);
571  bool cell2_has_right = !(std::holds_alternative<sector>(cell2) && !std::get<sector>(cell2).right);
572 
573  if (std::holds_alternative<section>(cell1) && std::holds_alternative<section>(cell2)) {
574  return std::get<section>(cell1).point < std::get<section>(cell2).point;
575  } else if (std::holds_alternative<sector>(cell1) && std::holds_alternative<section>(cell2)) {
576  if (!cell1_has_left) return true;
577  else {
578  return std::get<sector>(cell1).left < std::get<section>(cell2).point;
579  }
580  } else if (std::holds_alternative<section>(cell1) && std::holds_alternative<sector>(cell2)) {
581  if (!cell2_has_left) return false;
582  else {
583  return std::get<section>(cell1).point <= std::get<sector>(cell2).left;
584  }
585  } else if (std::holds_alternative<sector>(cell1) && std::holds_alternative<sector>(cell2)) {
586  if (!cell1_has_left && cell2_has_left) {
587  return true;
588  } else if (!cell2_has_left && cell1_has_left) {
589  return false;
590  } else if ((!cell1_has_left && !cell2_has_left) || (cell1_has_left && cell2_has_left && std::get<sector>(cell1).left == std::get<sector>(cell2).left) ) {
591  if (!cell1_has_right && cell2_has_right) {
592  return false;
593  } else if (!cell2_has_right && cell1_has_right) {
594  return true;
595  } else if (!cell2_has_right && !cell1_has_right) {
596  return false;
597  } else {
598  assert(cell2_has_right && cell1_has_right);
599  return std::get<sector>(cell1).right < std::get<sector>(cell2).right;
600  }
601  } else {
602  assert(cell1_has_left && cell2_has_left);
603  return std::get<sector>(cell1).left < std::get<sector>(cell2).left;
604  }
605  }
606  assert(false);
607  return false;
608  });
609 
610  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Sorted cells " << sorted_cells);
611 
612  auto iter = sorted_cells.begin();
613  assert(std::holds_alternative<sector>(**iter) && !std::get<sector>(**iter).left);
614  while (iter+1 != sorted_cells.end() && std::holds_alternative<sector>(**(iter+1)) && !std::get<sector>(**(iter+1)).left) {
615  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Section " << **iter << " is redundant");
616  iter++;
617  }
618  covering.cells.push_back(**iter);
619  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Chose first section " << **iter);
620  std::optional<RAN> current_right = std::get<sector>(**iter).right;
621  iter++;
622  while (current_right) {
623  assert(iter != sorted_cells.end());
624  if (std::holds_alternative<sector>(**iter)) {
625  std::optional<RAN> current_left = *std::get<sector>(**iter).left;
626  while (iter+1 != sorted_cells.end() && std::holds_alternative<sector>(**(iter+1)) && std::get<sector>(**(iter+1)).left == current_left) {
627  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Section " << **iter << " is redundant");
628  iter++;
629  }
630  if (!std::get<sector>(**(iter)).right || *std::get<sector>(**(iter)).right > *current_right) {
631  covering.cells.push_back(**iter);
632  current_right = std::get<sector>(**(iter)).right;
633  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Choose section " << **iter);
634  iter++;
635  } else {
636  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Section " << **iter << " is redundant");
637  iter++;
638  }
639  } else {
640  assert(std::holds_alternative<section>(**iter));
641  if (std::get<section>(**iter).point < *current_right) {
642  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Sector " << **iter << " is redundant");
643  iter++;
644  } else {
645  assert(std::get<section>(**iter).point == *current_right);
646  covering.cells.push_back(**iter);
647  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Chose sector " << **iter);
648  iter++;
649  while (std::holds_alternative<section>(**iter)) {
650  SMTRAT_LOG_TRACE("smtrat.mcsat.onecellcad.firstlevel", "Sector " << **iter << " is redundant");
651  iter++;
652  }
653  }
654  }
655  }
656 
657  return covering;
658 }

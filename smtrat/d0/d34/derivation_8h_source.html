29  * - a @ref BaseDerivation (a set of properties for each level)
30  * - a @ref DelineatedDerivation (a @ref BaseDerivation with an underlying @ref SampledDerivation on which the properties can be delineated)

339 /**
340  * Initializes a derivation according the the given assignment and level.
341  */
342 template<typename Properties>
343 DerivationRef<Properties> make_derivation(Projections& proj, const Assignment& assignment, size_t level) {
344  const auto& vars = proj.polys().var_order();
345 
346  auto zero_base = std::make_shared<BaseDerivation<Properties>>(proj, BaseDerivationRef<Properties>(nullptr), 0);
347  auto zero_delineated = std::make_shared<DelineatedDerivation<Properties>>(zero_base);
348  DerivationRef<Properties> current = std::make_shared<SampledDerivation<Properties>>(zero_delineated, RAN(0));
349 
350  for (size_t i = 1; i <= level; i++) {
351  auto base = std::make_shared<BaseDerivation<Properties>>(proj, current, i);
352  auto delineated = std::make_shared<DelineatedDerivation<Properties>>(base);
353  if (assignment.find(vars[i - 1]) != assignment.end()) {
354  current = std::make_shared<SampledDerivation<Properties>>(delineated, assignment.at(vars[i - 1]));
355  } else {
356  current = delineated;
357  }
358  }
359 
360  return current;
361 }

373 /**
374  * Merges the underlying derivations of a set of sampled derivations. After the operation, all sampled derivations point to the same underlying derivation.
375  */
376 template<typename Properties>
377 void merge_underlying(std::vector<SampledDerivationRef<Properties>>& derivations) {
378  std::set<DerivationRef<Properties>> underlying;
379  for (auto& deriv : derivations) {
380  underlying.insert(deriv->underlying());
381  }
382  assert(!underlying.empty());
383  auto first_underlying = *underlying.begin();
384  for (auto iter = std::next(underlying.begin()); iter != underlying.end(); iter++) {
385  first_underlying.base().merge_with(iter->base());
386  }
387  for (auto& deriv : derivations) {
388  deriv->base()->m_underlying = first_underlying;
389  }
390 }

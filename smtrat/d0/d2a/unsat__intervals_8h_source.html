163 /**
164  * Returns the unsat intervals of the given atom w.r.t. an underlying sample.
165  *
166  * @param c A constraint or a variable comparison.
167  * @param sample A sample point such that the highest variable in @ref c w.r.t. the variable odering in @ref proj is the only unassigned variable.
168  * @return A list of sampled derivations with the same delineated derivations. The samples for the unassigned variables are sampled from the respective interval.
169  */
170 template <cadcells::operators::op op>
171 std::vector<datastructures::SampledDerivationRef<typename operators::PropertiesSet<op>::type>> get_unsat_intervals(const FormulaT& c, datastructures::Projections& proj, const Assignment& sample) {
172  SMTRAT_LOG_FUNC("smtrat.cadcells.algorithms.onecell", c << ", " << sample);
173  if (c.getType() == carl::FormulaType::CONSTRAINT) {
174  return get_unsat_intervals<op>(c.constraint(), proj, sample);
175  } else if (c.getType() == carl::FormulaType::VARCOMPARE) {
176  return get_unsat_intervals<op>(c.variableComparison(), proj, sample);
177  } else {
178  assert(false);

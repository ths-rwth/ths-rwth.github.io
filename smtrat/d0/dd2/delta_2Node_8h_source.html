18 /**
19  * This class represents a node in a SMTLIB file.
20  *
21  * Every syntactical constructor is represented by a node.
22  * A node may have a name and may have several children.
23  * In a file, this corresponds to `(<name> <child> <child>...)`.
24  *
25  * Note that these nodes are constructed by a parser that has no semantic knowledge of the smtlib format.
26  * Hence, the data representable by these nodes is only a very rough overapproximation of the actual smtlib format.
27  */

107  * If the node `from` is encountered, replace it with `to`. If `to` is a nullptr, remove it instead.
108  * @param from Node to replace.

128  * @param from Node name to replace.
129  * @param to Node to replace with.
130  * @return Cloned node.

132  Node clone(const std::string& from, const Node* to) const {
133  if (children.size() == 0 && name == from) return *to;

135  for (const auto& c: children) {
136  if (c.children.size() == 0 && c.name == from) {
137  if (to != nullptr) newChildren.push_back(*to);

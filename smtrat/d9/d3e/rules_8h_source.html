45 template<typename P>
46 bool poly_non_null(datastructures::SampledDerivation<P>& deriv, datastructures::PolyRef poly) {
47  SMTRAT_LOG_TRACE("smtrat.cadcells.operators.rules", "non_null(" << poly << ")");
48  if(deriv.proj().is_nullified(deriv.sample(), poly)) {
49  SMTRAT_LOG_TRACE("smtrat.cadcells.operators.rules", "non_null(" << poly << ") <= false");
50  return false;
51  } else if (deriv.proj().has_const_coeff(poly)) {
52  SMTRAT_LOG_TRACE("smtrat.cadcells.operators.rules", "non_null(" << poly << ") <= " << poly << " has const coeff");
53  } else if (!deriv.proj().is_ldcf_zero(deriv.sample(), poly) && deriv.contains(properties::poly_sgn_inv{ deriv.proj().ldcf(poly) } )) {
54  SMTRAT_LOG_TRACE("smtrat.cadcells.operators.rules", "non_null(" << poly << ") <= ldcf(" << poly << ")(" << deriv.sample() << ")!=0 && sgn_inv(ldcf(" << poly << "))");
55  } else if (
56  deriv.proj().know_disc(poly) && !deriv.proj().is_disc_zero(deriv.sample(), poly) &&
57  (deriv.contains(properties::poly_sgn_inv{ deriv.proj().disc(poly) }) || deriv.contains(properties::poly_ord_inv{ deriv.proj().disc(poly) }))
58  ) {
59  SMTRAT_LOG_TRACE("smtrat.cadcells.operators.rules", "non_null(" << poly << ") <= disc(" << poly << ")(" << deriv.sample() << ")!=0 && sgn_inv(disc(" << poly << "))");
60  } else {
61  auto coeff = deriv.proj().simplest_nonzero_coeff(deriv.sample(), poly, [&](const Poly& a, const Poly& b) {
62  if (deriv.proj().known(a) && deriv.contains( properties::poly_sgn_inv { deriv.proj().polys()(a)} )) return true;
63  if (deriv.proj().known(b) && deriv.contains( properties::poly_sgn_inv { deriv.proj().polys()(b)} )) return true;
64  return helper::level_of(deriv.polys().var_order(),a) < helper::level_of(deriv.polys().var_order(),b) || (helper::level_of(deriv.polys().var_order(),a) == helper::level_of(deriv.polys().var_order(),b) && a.degree(helper::main_var(deriv.polys().var_order(), a)) < b.degree(helper::main_var(deriv.polys().var_order(), b)));
65  });
66  SMTRAT_LOG_TRACE("smtrat.cadcells.operators.rules", "non_null(" << poly << ") <= sgn_inv(" << coeff << ") && " << coeff << " is coeff of " << poly << "");
67  deriv.insert(properties::poly_sgn_inv{ coeff });
68  }
69  return true;
70 }

116  if (deriv.proj().is_const(poly)) {
117  SMTRAT_LOG_TRACE("smtrat.cadcells.operators.rules", "-> sgn_inv(" << poly << ") <= " << poly << " const");
118  } else if (deriv.contains(properties::poly_ord_inv{ poly })) {

124  SMTRAT_LOG_TRACE("smtrat.cadcells.operators.rules", "-> sgn_inv(" << poly << ") <= sgn_inv(" << factor << ") && "<< factor <<"("<< deriv.underlying_sample() <<")=0");

226 template<typename P>
227 void poly_irreducible_sgn_inv(datastructures::SampledDerivation<P>& deriv, const datastructures::CellDescription& cell, const datastructures::IndexedRootOrdering& ordering, datastructures::PolyRef poly) {
228  SMTRAT_LOG_TRACE("smtrat.cadcells.operators.rules", "sgn_inv(" << poly << "), " << poly << " irreducible");
229  assert(deriv.contains(properties::poly_pdel{ poly }));
230  deriv.insert(properties::cell_connected{ poly.level-1 });
231  if (cell.is_section() && deriv.proj().is_zero(deriv.sample(), poly)) {
232  // ok by indexed root ordering!
233  } else {
234  assert(!deriv.proj().is_zero(deriv.sample(), poly));
235  if (cell.is_sector() && (deriv.cell().lower_unbounded() || deriv.cell().upper_unbounded())) {
236  deriv.insert(properties::poly_sgn_inv{ deriv.proj().ldcf(poly) });
237  } else {
238  auto lower_root = ordering.leq_transitive(poly, *cell.lower_defining());
239  auto upper_root = ordering.leq_transitive(*cell.upper_defining(), poly);
240 
241  // compute set of polynomials that have a resultant with the current polynomials (according to the given ordering)
242  boost::container::flat_set<datastructures::PolyRef> res_polys;
243  for (const auto& rel : ordering.data()) {
244  if (rel.first.poly == poly) {
245  res_polys.insert(rel.second.poly);
246  } else if (rel.second.poly == poly) {
247  res_polys.insert(rel.first.poly);
248  }
249  }
250 
251  // compute protection for lower bound
252  auto protect_lower = [&]() {
253  if (poly == cell.lower_defining()->poly) return true;
254 
255  if (upper_root) {
256  // check if a res_poly has a well defined root below
257  for (const auto& res_poly : res_polys) {
258  if (res_poly == cell.lower_defining()->poly) {
259  return true;
260  } else {
261  auto res_lower_root = ordering.leq_transitive(res_poly, *cell.lower_defining());
262  if (res_lower_root) {
263  if (deriv.contains(properties::root_well_def{*res_lower_root}) || is_trivial_root_well_def(deriv.underlying().sampled(), *res_lower_root)) {
264  return true;
265  }
266  }
267  }
268  }
269  }
270 
271  if (lower_root) {
272  deriv.insert(properties::root_well_def{*lower_root});
273  return true;
274  }
275 
276  return false;
277  };
278 
279  // compute protection for upper bound
280  auto protect_upper = [&]() {
281  if (poly == cell.upper_defining()->poly) return true;
282 
283  if (lower_root) {
284  // check if a res_poly has a well defined root above
285  for (const auto& res_poly : res_polys) {
286  if (res_poly == cell.upper_defining()->poly) {
287  return true;
288  } else {
289  auto res_upper_root = ordering.leq_transitive(*cell.upper_defining(), res_poly);
290  if (res_upper_root) {
291  if (deriv.contains(properties::root_well_def{*res_upper_root}) || is_trivial_root_well_def(deriv.underlying().sampled(), *res_upper_root)) {
292  return true;
293  }
294  }
295  }
296  }
297  }
298 
299  if (upper_root) {
300  deriv.insert(properties::root_well_def{*upper_root});
301  return true;
302  }
303 
304  return false;
305  };
306 
307  if (!protect_lower() || !protect_upper()) {
308  deriv.insert(properties::poly_sgn_inv{ deriv.proj().ldcf(poly) });
309  }
310  }
311  }
312 }

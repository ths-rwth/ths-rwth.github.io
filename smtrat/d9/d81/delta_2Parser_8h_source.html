93 /**
94  * This class parses a whole smtlib file into a hierarchy of Node objects.
95  */
96 class Parser {
97  /// Parses symbols.
98  SymbolParser symbol;
99  /// Parses a Node that consists of a symbol.
100  qi::rule<Iterator, std::tuple<std::string, bool>(), Skipper> symbol_node;
101  /// Parses a Node that is empty.
102  qi::rule<Iterator, std::tuple<std::vector<Node>, bool>(), Skipper> empty_node;
103  /// Parses a Node that consists of a symbol and further children.
104  qi::rule<Iterator, std::tuple<std::string, std::vector<Node>, bool>(), Skipper> full_node;
105  /// Parses any Node.
106  qi::rule<Iterator, Node(), Skipper> node;
107  /// Parses a sequence of nodes.
108  qi::rule<Iterator, std::tuple<std::vector<Node>, bool>(), Skipper> nodelist;
109  /// Parses a smtlib file.
110  qi::rule<Iterator, Node(), Skipper> main;
111  // Error handler.
112  boost::phoenix::function<ErrorHandler> errorHandler;
113 
114 public:
115  /**
116  * Constructs the parsing rules.
117  */
118  Parser() {
119  symbol_node = symbol >> qi::attr(false);
120  full_node = qi::lit("(") >> symbol >> *node >> qi::attr(true) >> qi::lit(")");
121  empty_node = qi::lit("(") >> *node >> qi::attr(true) >> qi::lit(")");
122  node = symbol_node | full_node | empty_node;
123  nodelist = *node >> qi::attr(false);
124  main = qi::eps > nodelist > qi::eoi;
125  qi::on_error<qi::fail>(main, errorHandler(qi::_1, qi::_2, qi::_3));
126  }

128  /**
129  * Parses a file into a node.
130  * @param filename Filename of the input file.
131  * @return Node object produced by the parser.
132  */
133  bool parseFile(const std::string& filename, Node& node) {
134  std::ifstream in(filename);
135  in.unsetf(std::ios::skipws);
136  BaseIteratorType basebegin(in);

138  Iterator end;
139  return qi::phrase_parse(begin, end, main, Skipper(), node);
140  }

146  */

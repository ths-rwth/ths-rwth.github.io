136 inline void compute_barriers(datastructures::SampledDerivationRef<T>& der, datastructures::CellRepresentation<T>& response, bool section) {
137  while(section) {
138  auto old_size = response.equational.size();
139 
140  auto it = der->cell().lower();
141  while(true) {
142  for (auto ir = it->second.begin(); ir != it->second.end(); ir++) {
143  if (ir->poly == response.description.section_defining().poly) continue;
144  if (response.equational.contains(ir->poly)) continue;
145  if (!util::compare_simplest(der->proj(),ir->poly,response.description.section_defining().poly)) {
146  response.equational.insert(ir->poly);
147  }
148  }
149  if (it != der->delin().roots().begin()) it--;
150  else break;
151  }
152 
153  it = der->cell().upper();
154  while(it != der->delin().roots().end()) {
155  for (auto ir = it->second.begin(); ir != it->second.end(); ir++) {
156  if (ir->poly == response.description.section_defining().poly) continue;
157  if (response.equational.contains(ir->poly)) continue;
158  if (!util::compare_simplest(der->proj(),ir->poly,response.description.section_defining().poly)) {
159  response.equational.insert(ir->poly);
160  }
161  }
162  it++;
163  }
164 
165  if (old_size == response.equational.size()) {
166  break;
167  }
168  }
169 
170  if (!der->cell().lower_unbounded()) {
171  boost::container::flat_set<datastructures::PolyRef> ignoring;
172  auto it = der->cell().lower();
173  auto barrier = *response.description.lower_defining();
174  while(true) {
175  auto old_barrier = barrier;
176  for (auto ir = it->second.begin(); ir != it->second.end(); ir++) {
177  if (ignoring.contains(ir->poly)) continue;
178  if (section && response.equational.contains(ir->poly)) continue;
179  if (util::compare_simplest(der->proj(),ir->poly,barrier.poly)) {
180  barrier = *ir;
181  }
182  }
183  assert(it != der->cell().lower() || barrier == *response.description.lower_defining());
184  if (barrier != old_barrier) {
185  response.ordering.add_leq(barrier, old_barrier);
186  }
187  for (const auto& ir : it->second) {
188  if (ignoring.contains(ir.poly)) continue;
189  if (section && response.equational.contains(ir.poly)) continue;
190  if (ir != barrier) {
191  response.ordering.add_leq(ir, barrier);
192  }
193  ignoring.insert(ir.poly);
194  }
195  if (it != der->delin().roots().begin()) it--;
196  else break;
197  }
198  }
199  if (!der->cell().upper_unbounded()) {
200  boost::container::flat_set<datastructures::PolyRef> ignoring;
201  auto it = der->cell().upper();
202  auto barrier = *response.description.upper_defining();
203  while(it != der->delin().roots().end()) {
204  auto old_barrier = barrier;
205  for (auto ir = it->second.begin(); ir != it->second.end(); ir++) {
206  if (ignoring.contains(ir->poly)) continue;
207  if (section && response.equational.contains(ir->poly)) continue;
208  if (util::compare_simplest(der->proj(),ir->poly,barrier.poly)) {
209  barrier = *ir;
210  }
211  }
212  assert(it != der->cell().upper() || barrier == *response.description.upper_defining());
213  if (barrier != old_barrier) {
214  response.ordering.add_leq(old_barrier, barrier);
215  }
216  for (const auto& ir : it->second) {
217  if (ignoring.contains(ir.poly)) continue;
218  if (section && response.equational.contains(ir.poly)) continue;
219  if (ir != barrier) {
220  response.ordering.add_leq(barrier, ir);
221  }
222  ignoring.insert(ir.poly);
223  }
224  it++;
225  }
226  }

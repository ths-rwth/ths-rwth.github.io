49  carl::Relation combine(carl::Relation lhs, carl::Relation rhs, std::size_t exponent) {
50  assert(lhs != carl::Relation::NEQ);
51  assert(rhs != carl::Relation::NEQ);
52  if (rhs == carl::Relation::EQ) return carl::Relation::EQ;
53  if (exponent % 2 == 0) {
54  if (rhs == carl::Relation::LEQ) rhs = carl::Relation::GEQ;
55  else if (rhs == carl::Relation::LESS) rhs = carl::Relation::GREATER;
56  }
57  switch (lhs) {
58  case carl::Relation::GREATER: return rhs;
59  case carl::Relation::GEQ:
60  switch (rhs) {
61  case carl::Relation::GREATER:
62  case carl::Relation::GEQ: return carl::Relation::GEQ;
63  case carl::Relation::LEQ:
64  case carl::Relation::LESS: return carl::Relation::LEQ;
65  default: return carl::Relation::NEQ;
66  }
67  case carl::Relation::EQ: return carl::Relation::EQ;
68  case carl::Relation::LEQ:
69  switch (rhs) {
70  case carl::Relation::GREATER:
71  case carl::Relation::GEQ: return carl::Relation::LEQ;
72  case carl::Relation::LEQ:
73  case carl::Relation::LESS: return carl::Relation::GEQ;
74  default: return carl::Relation::NEQ;
75  }
76  case carl::Relation::LESS:
77  switch (rhs) {
78  case carl::Relation::GREATER: return carl::Relation::LESS;
79  case carl::Relation::GEQ: return carl::Relation::LEQ;
80  case carl::Relation::LEQ: return carl::Relation::GEQ;
81  case carl::Relation::LESS: return carl::Relation::GREATER;
82  default: return carl::Relation::NEQ;
83  }
84  case carl::Relation::NEQ: return carl::Relation::NEQ;
85  }
86  return carl::Relation::NEQ;
87  }

104  EvalRationalIntervalMap completeBounds(const Poly& p) const {
105  auto res = varbounds.getEvalIntervalMap();
106  for (auto var: carl::variables(p)) {
107  if (res.find(var) == res.end()) {

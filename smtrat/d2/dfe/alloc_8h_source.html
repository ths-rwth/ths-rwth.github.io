94  /**
95  * First, try to return the first entry on the freelist.
96  */
97  if(mFree != nullptr) {
98  void* result = mFree;
99  mFree = *static_cast<void**>(mFree);
100  return result;
101  }

104  * If that is empty, try forward-allocation
105  * (allocating the next element on the current page
106  * that has never been allocated before.)
107  */
108  if(mCurrentPageFreeFrom != (first_page_size << mCurrentPage)) {
109  return static_cast<void*>(
110  static_cast<char*>(mPages[mCurrentPage]) +
111  (mCurrentPageFreeFrom++ << mSizeShift)
112  );
113  }

117  */
118  ++mCurrentPage;

120  mCurrentPageFreeFrom = 1;

126  * This page has twice the space of the previous page.
127  */

129  if(newPage == nullptr) {

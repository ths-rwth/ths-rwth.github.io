20 void extend_valuation(Node& f, const cadcells::Assignment& ass, bool evaluate_all) {
21  if (f.c().valuation == Valuation::TRUE || f.c().valuation == Valuation::FALSE) return;
22  return std::visit(overloaded{
23  [&](TRUE&) {
24  f.c().valuation = Valuation::TRUE;
25  },
26  [&](FALSE&) {
27  f.c().valuation = Valuation::FALSE;
28  },
29  [&](NOT& c) {
30  extend_valuation(c.subformula, ass, evaluate_all);
31  if (c.subformula.c().valuation == Valuation::TRUE) {
32  f.c().valuation = Valuation::FALSE;
33  } else if (c.subformula.c().valuation == Valuation::FALSE) {
34  f.c().valuation = Valuation::TRUE;
35  } else {
36  assert(c.subformula.c().valuation == Valuation::MULTIVARIATE);
37  f.c().valuation = Valuation::MULTIVARIATE;
38  }
39  },
40  [&](AND& c) {
41  f.c().valuation = Valuation::TRUE;
42  for (auto& sf : c.subformulas) {
43  extend_valuation(sf, ass, evaluate_all);
44  if (sf.c().valuation == Valuation::FALSE) {
45  f.c().valuation = Valuation::FALSE;
46  if (!evaluate_all)
47  break;
48  } else if (sf.c().valuation == Valuation::MULTIVARIATE) {
49  if (!evaluate_all || f.c().valuation != Valuation::FALSE)
50  f.c().valuation = Valuation::MULTIVARIATE;
51  }
52  }
53  },
54  [&](OR& c) {
55  f.c().valuation = Valuation::FALSE;
56  for (auto& sf : c.subformulas) {
57  extend_valuation(sf, ass, evaluate_all);
58  if (sf.c().valuation == Valuation::TRUE) {
59  f.c().valuation = Valuation::TRUE;
60  if (!evaluate_all)
61  break;
62  } else if (sf.c().valuation == Valuation::MULTIVARIATE) {
63  if (!evaluate_all || f.c().valuation != Valuation::TRUE)
64  f.c().valuation = Valuation::MULTIVARIATE;
65  }
66  }
67  },
68  [&](IFF& c) {
69  f.c().valuation = Valuation::UNKNOWN;
70  Valuation reference = Valuation::UNKNOWN;
71  for (auto& sf : c.subformulas) {
72  extend_valuation(sf, ass, evaluate_all);
73  if (sf.c().valuation != Valuation::MULTIVARIATE) {
74  if (reference == Valuation::UNKNOWN) {
75  reference = sf.c().valuation;
76  } else if (reference == sf.c().valuation) {
77  f.c().valuation = Valuation::TRUE;
78  } else {
79  assert(reference != sf.c().valuation);
80  f.c().valuation = Valuation::FALSE;
81  if (!evaluate_all)
82  break;
83  }
84  } else {
85  if (!evaluate_all || f.c().valuation != Valuation::FALSE)
86  f.c().valuation = Valuation::MULTIVARIATE;
87  }
88  }
89  },
90  [&](XOR& c) {
91  f.c().valuation = Valuation::FALSE;
92  for (auto& sf : c.subformulas) {
93  extend_valuation(sf, ass, evaluate_all);
94  if (sf.c().valuation == Valuation::MULTIVARIATE) {
95  f.c().valuation = Valuation::MULTIVARIATE;
96  if (!evaluate_all) break;
97  } else if (sf.c().valuation == Valuation::TRUE) {
98  if (!evaluate_all || f.c().valuation != Valuation::MULTIVARIATE)
99  f.c().valuation = Valuation::FALSE;
100  } else {
101  assert(sf.c().valuation == Valuation::FALSE);
102  if (!evaluate_all || f.c().valuation != Valuation::MULTIVARIATE)
103  f.c().valuation = Valuation::TRUE;
104  }
105  }
106  },
107  [&](BOOL&) {
108  f.c().valuation = Valuation::MULTIVARIATE;
109  },
110  [&](CONSTRAINT& c) {

123 void revert_valuation(Node& f, std::size_t level) {
124  return std::visit(overloaded{
125  [&](TRUE&) {
126  // do nothing
127  },
128  [&](FALSE&) {
129  // do nothing

142  exists_false = true;
143  }

146  f.c().valuation = Valuation::MULTIVARIATE;
147  }
148  },

153  if (sf.c().valuation == Valuation::TRUE) {
154  exists_true = true;
155  }
156  }
157  if (!exists_true) {
158  f.c().valuation = Valuation::MULTIVARIATE;
159  }
160  },
161  [&](IFF& c) {
162  bool exists_false = false;
163  bool exists_true = false;

165  for (auto& sf : c.subformulas) {
166  revert_valuation(sf, level);
167  if (sf.c().valuation == Valuation::TRUE) {
168  exists_true = true;
169  } else if (sf.c().valuation == Valuation::FALSE) {

171  } else if (sf.c().valuation == Valuation::MULTIVARIATE) {
172  exists_multivariate = true;
173  }
174  }
175  if (!(exists_true && exists_false) && exists_multivariate) {
176  f.c().valuation = Valuation::MULTIVARIATE;
177  }
178  },
179  [&](XOR& c) {
180  for (auto& sf : c.subformulas) {
181  revert_valuation(sf, level);
182  if (sf.c().valuation == Valuation::MULTIVARIATE) {
183  f.c().valuation = Valuation::MULTIVARIATE;
184  }
185  }
186  },
187  [&](BOOL&) {
188  f.c().valuation = Valuation::MULTIVARIATE;
189  },

198 Node to_node(typename cadcells::Polynomial::ContextType c, const FormulaT& f, std::map<std::size_t,Node>& cache) {
199  {
200  auto cache_it = cache.find(f.id());
201  if (cache_it != cache.end()) return cache_it->second;
202  }
203  switch(f.type()) {
204  case carl::FormulaType::ITE: {
205  return to_node(c, FormulaT(carl::FormulaType::OR, FormulaT(carl::FormulaType::AND, f.condition(), f.first_case()), FormulaT(carl::FormulaType::AND, f.condition().negated(), f.second_case())), cache);
206  }
207  case carl::FormulaType::TRUE: {
208  return Node(TRUE{ });
209  }
210  case carl::FormulaType::FALSE: {
211  return Node(FALSE{ });
212  }
213  case carl::FormulaType::BOOL: {
214  return Node(BOOL{ f.boolean() });
215  }
216  case carl::FormulaType::NOT: {
217  return Node(NOT{ to_node(c,f.subformula(), cache) });
218  }
219  case carl::FormulaType::IMPLIES: {
220  return to_node(c, FormulaT(carl::FormulaType::OR, f.premise().negated(), f.conclusion()), cache);
221  }
222  case carl::FormulaType::AND: {
223  std::vector<Node> subformulas;
224  for (const auto& sf : f.subformulas()) {
225  subformulas.emplace_back(to_node(c,sf, cache));
226  }
227  return Node(AND{ std::move(subformulas) });
228  }
229  case carl::FormulaType::OR: {
230  std::vector<Node> subformulas;
231  for (const auto& sf : f.subformulas()) {
232  subformulas.emplace_back(to_node(c,sf, cache));
233  }
234  return Node(OR{ std::move(subformulas) });
235  }
236  case carl::FormulaType::XOR: {
237  std::vector<Node> subformulas;
238  for (const auto& sf : f.subformulas()) {
239  subformulas.emplace_back(to_node(c,sf, cache));
240  }
241  return Node(XOR{ std::move(subformulas) });
242  }
243  case carl::FormulaType::IFF: {
244  std::vector<Node> subformulas;
245  for (const auto& sf : f.subformulas()) {
246  subformulas.emplace_back(to_node(c,sf, cache));
247  }
248  return Node(IFF{ std::move(subformulas) });
249  }
250  case carl::FormulaType::CONSTRAINT: {
251  auto bc = carl::convert<cadcells::Polynomial>(c, f.constraint().constr());
252  return Node(CONSTRAINT{ bc });
253  }
254  default: {
255  assert(false);
256  return Node(FALSE{});
257  }
258  }

427 void compute_implicant(const node_ds::Node& f, boost::container::flat_set<cadcells::Constraint>& implicant) {
428  assert (f.c().valuation == Valuation::TRUE || f.c().valuation == Valuation::FALSE);
429  return std::visit(overloaded{
430  [&](const node_ds::TRUE&) {
431  // do nothing
432  },
433  [&](const node_ds::FALSE&) {
434  // do nothing
435  },
436  [&](const node_ds::NOT& c) {
437  boost::container::flat_set<cadcells::Constraint> sub_implicant;
438  compute_implicant(c.subformula, sub_implicant);
439  for (const auto& si : sub_implicant) {
440  implicant.insert(si.negation());
441  }
442  },
443  [&](const node_ds::AND& c) {
444  if (f.c().valuation == Valuation::FALSE) {
445  boost::container::flat_set<cadcells::Constraint> sub_implicant;
446  for (auto sf = c.subformulas.rbegin(); sf != c.subformulas.rend(); sf++) {
447  if (sf->c().valuation == Valuation::FALSE) {
448  sub_implicant.clear();
449  compute_implicant(*sf, sub_implicant);
450  if (std::includes(implicant.begin(), implicant.end(), sub_implicant.begin(), sub_implicant.end())) {
451  sub_implicant.clear();
452  break;
453  }
454  }
455  }
456  implicant.insert(sub_implicant.begin(), sub_implicant.end());
457  } else if (f.c().valuation == Valuation::TRUE) {
458  for (const auto& sf : c.subformulas) {
459  assert(sf.c().valuation == Valuation::TRUE);
460  compute_implicant(sf, implicant);
461  }
462  }
463  },
464  [&](const node_ds::OR& c) {
465  if (f.c().valuation == Valuation::TRUE) {
466  boost::container::flat_set<cadcells::Constraint> sub_implicant;
467  for (auto sf = c.subformulas.rbegin(); sf != c.subformulas.rend(); sf++) {
468  if (sf->c().valuation == Valuation::TRUE) {
469  sub_implicant.clear();

471  if (std::includes(implicant.begin(), implicant.end(), sub_implicant.begin(), sub_implicant.end())) {
472  sub_implicant.clear();
473  break;
474  }
475  }
476  }
477  implicant.insert(sub_implicant.begin(), sub_implicant.end());
478  } else if (f.c().valuation == Valuation::FALSE) {
479  for (const auto& sf : c.subformulas) {
480  assert(sf.c().valuation == Valuation::FALSE);
481  compute_implicant(sf, implicant);
482  }
483  }
484  },
485  [&](const node_ds::IFF& c) {
486  if (f.c().valuation == Valuation::TRUE) {
487  for (const auto& sf : c.subformulas) {
488  compute_implicant(sf, implicant);
489  }
490  } else {

492  boost::container::flat_set<cadcells::Constraint> sub_implicant_false;
493  bool found_true = false;

496  if (sf->c().valuation == Valuation::TRUE && !found_true) {
497  sub_implicant_true.clear();

499  if (std::includes(implicant.begin(), implicant.end(), sub_implicant_true.begin(), sub_implicant_true.end())) {
500  found_true = true;
501  sub_implicant_true.clear();
502  }
503  } else if (sf->c().valuation == Valuation::FALSE && !found_false) {
504  sub_implicant_false.clear();
505  compute_implicant(*sf, sub_implicant_false);
506  if (std::includes(implicant.begin(), implicant.end(), sub_implicant_false.begin(), sub_implicant_false.end())) {
507  found_false = true;
508  sub_implicant_false.clear();
509  }
510  }
511  }
512  implicant.insert(sub_implicant_true.begin(), sub_implicant_true.end());
513  implicant.insert(sub_implicant_false.begin(), sub_implicant_false.end());
514  }
515  },
516  [&](const node_ds::XOR& c) {
517  for (const auto& sf : c.subformulas) {
518  assert(sf.c().valuation == Valuation::TRUE || sf.c().valuation == Valuation::FALSE);
519  compute_implicant(sf, implicant);
520  }
521  },
522  [&](const node_ds::BOOL&) {
523  assert(false);
524  },
525  [&](const node_ds::CONSTRAINT& c) {
526  if (f.c().valuation == Valuation::TRUE) {
527  implicant.insert(c.constraint);

565 std::vector<boost::container::flat_set<cadcells::Constraint>> compute_implicants(const node_ds::Node& f, std::size_t pruning, ExhaustiveImplicants::ImplicantComplexityOrdering implicant_complexity_ordering) {

568  std::visit(overloaded{

571  },

573  assert(false);
574  },
575  [&](const node_ds::NOT& c) {
576  auto sub_implicants = compute_implicants(c.subformula, pruning, implicant_complexity_ordering);
577  for (const auto& sub_implicant : sub_implicants) {
578  implicants.emplace_back();
579  for (const auto& si : sub_implicant) {
580  implicants.back().insert(si.negation());
581  }
582  }
583  },
584  [&](const node_ds::AND& c) {
585  if (f.c().valuation == Valuation::FALSE) {
586  for (const auto& sf : c.subformulas) {
587  if (sf.c().valuation == Valuation::FALSE) {
588  auto sub_implicants = compute_implicants(sf, pruning, implicant_complexity_ordering);
589  implicants.insert(implicants.end(), sub_implicants.begin(), sub_implicants.end());
590  }
591  }
592  } else if (f.c().valuation == Valuation::TRUE) {
593  for (const auto& sf : c.subformulas) {
594  assert(sf.c().valuation == Valuation::TRUE);
595  auto sub_implicants= compute_implicants(sf, pruning, implicant_complexity_ordering);
596  auto size = implicants.size();
597  if (size == 0) {
598  implicants.insert(implicants.end(), sub_implicants.begin(), sub_implicants.end());
599  } else {
600  for (std::size_t j = 1; j < sub_implicants.size(); j++) {
601  for (std::size_t i = 0; i < size; i++) {
602  implicants.emplace_back(implicants[i]);
603  }
604  }
605  std::size_t i = 0;
606  for (const auto& sub_implicant : sub_implicants) {
607  for (std::size_t j = 0; j < size; j++) {
608  assert(i < implicants.size());
609  implicants[i].insert(sub_implicant.begin(), sub_implicant.end());
610  i++;
611  }
612  }
613  assert(i == implicants.size());
614  }
615  }
616  // TODO remove duplicates
617  }
618  },
619  [&](const node_ds::OR& c) {

646  }

676  }
677  }
678  // TODO remove duplicates
679  } else {

681  std::vector<boost::container::flat_set<cadcells::Constraint>> sub_implicants_false;
682  for (const auto& sf : c.subformulas) {

684  if (sf.c().valuation == Valuation::TRUE) {

704  auto size = implicants.size();

707  } else {

721  assert(i == implicants.size());
722  }
723  }
724  },
725  [&](const node_ds::BOOL&) {
726  assert(false);
727  },
728  [&](const node_ds::CONSTRAINT& c) {
729  if (f.c().valuation == Valuation::TRUE) {

733  } else {

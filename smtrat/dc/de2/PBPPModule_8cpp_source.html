75  void PBPPModule<Settings>::addConstraints(const FormulaT& formula) {
76  if (formula.isBooleanCombination() && formula.isNary()) {
77  SMTRAT_LOG_DEBUG("smtrat.pbc", "Searching for embedded constraints in " << formula);
78  for (const auto& subformula : formula.subformulas()) {
79  addConstraints(subformula);
80  }
81 
82  return;
83  }
84 
85  if (formula.getType() == carl::FormulaType::CONSTRAINT && formula.constraint().isPseudoBoolean()) {
86  ConstraintT constraint = formula.constraint();
87  if (!constraint.isPseudoBoolean()) {
88  // eg an objective function - just forward it
89  addSubformulaToPassedFormula(formula, formula);
90  return;
91  }
92 
93  ConstraintT normalizedConstraint = constraint;
94  if (Settings::NORMALIZE_CONSTRAINTS) {
95  #ifdef DEBUG_PBPP
96  std::cout << "Constraint before normalization: \t" << normalizedConstraint << std::endl;
97  #endif
98  // normalize and hence hopefully simplify the formula
99  std::pair<boost::optional<FormulaT>, ConstraintT> normalizedConstraintWithBoolPart = mNormalizer.normalize(constraint);
100 
101  // extract the normalized constraint and pass along the rest
102  #ifdef DEBUG_PBPP
103  std::cout << "Constraint after normalization: \t"
104  << normalizedConstraintWithBoolPart.first
105  << " and "
106  << normalizedConstraintWithBoolPart.second << std::endl;
107  #endif
108 
109  normalizedConstraint = normalizedConstraintWithBoolPart.second;
110  if (normalizedConstraintWithBoolPart.first) {
111  extractConstraints(*normalizedConstraintWithBoolPart.first);
112  }
113  }

116  formulaByConstraint[normalizedConstraint] = formula;
117  }

121  void PBPPModule<Settings>::extractConstraints(const FormulaT& formula) {
122  if (formula.isBooleanCombination()) {
123  assert(formula.getType() == carl::FormulaType::AND);
124  for (const auto& subformula: formula.subformulas()) {
125  extractConstraints(subformula);
126  }
127  } else if (formula.getType() == carl::FormulaType::CONSTRAINT) {
128  mConstraints.push_back(formula.constraint());
129  formulaByConstraint[formula.constraint()] = formula;
130  } else if (formula.getType() == carl::FormulaType::TRUE) {
131  return;
132  } else {
133  // we only expect a boolean combination of constraints
134  assert("The FormulaType passed was not valid. Expected Constraint, BooleanCombination or TRUE" && false);

138  template<class Settings>
139  Answer PBPPModule<Settings>::checkCore()
140  {
141  // 1. Preprocessing - ignore some constraints and gather informations
142  for (const auto& constraint : mConstraints) {
143 
144  // we can also check a mode which only uses simplex and does not encode
145  if (Settings::USE_LIA_ONLY) {
146  liaConstraints.push_back(constraint);
147  continue;
148  }
149 
150  // store information about the constraint
151  Rational encodingSize = std::numeric_limits<size_t>::max();
152  for (const auto encoder : mEncoders) {
153  Rational curEncoderSize = encoder->encodingSize(constraint);
154  encodingSize = conversionSizeByConstraint[constraint];
155  if (encoder->canEncode(constraint) &&
156  (curEncoderSize <= encodingSize || conversionSizeByConstraint[constraint] == Rational(0)))
157  {
158  encoderByConstraint[constraint] = encoder;
159  conversionSizeByConstraint[constraint] = curEncoderSize;
160  }
161  }
162 
163  if (encoderByConstraint.find(constraint) == encoderByConstraint.end()) {
164  SMTRAT_LOG_INFO("smtrat.pbc", "There is no encoder for constraint " << constraint);
165  // if we do not know how to encode the constraint, use LIA!
166  liaConstraints.push_back(constraint);
167  continue;
168  }
169 
170  // by now we know which encoder we want to use if we actually want to convert to a boolean formula.
171  // Now check whether it is "benefitial" to use the boolean encoding, i.e. whether we introduce more
172  // new formulas than we already have.
173  if (encodeAsBooleanFormula(constraint)) {
174  boolConstraints.push_back(constraint);
175  } else {
176  liaConstraints.push_back(constraint);
177  }
178  }
179 
180  auto constraintComperator = [](const ConstraintT& left, const ConstraintT& right) { return left.variables().size() < right.variables().size();};
181  // sort by number of variables ascending
182  std::sort(boolConstraints.begin(), boolConstraints.end(), constraintComperator);
183  std::sort(liaConstraints.begin(), liaConstraints.end(), constraintComperator);
184 
185  #ifdef DEBUG_PBPP
186  std::cout << "After Step 1 - Encoding as LIA: " << liaConstraints << std::endl;
187  std::cout << "After Step 1 - Encoding as Bool: " << boolConstraints << std::endl;
188  #endif
189 
190  std::set<carl::Variable> variablesInLIA;
191  for (const auto& constraint : liaConstraints) {
192  for (const auto& cvar : constraint.variables()) {
193  variablesInLIA.insert(cvar);
194  }
195  }
196 
197  // 3. add more constraints to LIA part to refine the relaxation
198 
199  // since access to variablesInLIA would invalidate the iterator we instead save which variables we already inspected
200  std::set<carl::Variable> inspectedVariables;
201  std::set<ConstraintT> additionallyLIAEncodedBoolConstraints;
202  std::map<carl::Variable, carl::Variable>& variablesFromNormalization = mNormalizer.substitutedVariables();
203  std::set<carl::Variable> variableSubstitutions;
204  for (const auto& pair : variablesFromNormalization) {
205  variableSubstitutions.insert(pair.second);
206  }
207 
208  for (const auto& var : variablesInLIA) {
209  if (inspectedVariables.find(var) != inspectedVariables.end()) continue;
210 
211  for (const auto& constraint : boolConstraints) {
212  if (additionallyLIAEncodedBoolConstraints.find(constraint) != additionallyLIAEncodedBoolConstraints.end()) {
213  continue;
214  }
215 
216  bool constraintContainsCurrentVariable = constraint.variables().has(var);
217  bool constraintContainsSubstitutedVariable = variableSubstitutions.find(var) != variableSubstitutions.end();
218 
219  if (!isTrivial(constraint) && (constraintContainsCurrentVariable || constraintContainsSubstitutedVariable)) {
220  inspectedVariables.insert(var);
221  liaConstraints.push_back(constraint);
222  additionallyLIAEncodedBoolConstraints.insert(constraint);
223  }
224  }
225  }
226 
227  for (const auto& additionalConstraint : additionallyLIAEncodedBoolConstraints) {
228  auto it = std::find(boolConstraints.begin(), boolConstraints.end(), additionalConstraint);
229  if (it != boolConstraints.end()) {
230  boolConstraints.erase(it);
231  }
232  }
233 
234  #ifdef DEBUG_PBPP
235  std::cout << "After Step 3 - Encoding as LIA: " << liaConstraints << std::endl;
236  std::cout << "After Step 3 - Encoding as Bool: " << boolConstraints << std::endl;
237  #endif
238 
239  // 4. encode all remaining constraints as bool
240  std::set<carl::Variable> variablesInBooleanPart;
241  for (const auto& constraint : boolConstraints){
242  boost::optional<FormulaT> boolEncoding = encoderByConstraint[constraint]->encode(constraint);
243  if (!boolEncoding) {
244  liaConstraints.push_back(constraint);
245  continue;
246  }
247 
248  #ifdef DEBUG_PBPP
249  std::cout << "Encoded using " << encoderByConstraint[constraint]->name() << " " << constraint << " \t as \t " << *boolEncoding << std::endl;
250  #endif
251  for (const auto& var : constraint.variables()) {
252  variablesInBooleanPart.insert(var);
253  }
254 
255  boolEncodings[constraint] = *boolEncoding;
256  }
257 
258  for (const auto& constraint : liaConstraints) {
259  //addSubformulaToPassedFormula(forwardAsArithmetic(constraint, variablesInBooleanPart), formulaByConstraint[constraint]);
260  liaConstraintFormula[constraint] = forwardAsArithmetic(constraint, variablesInBooleanPart);
261  }
262 
263  #ifdef PRINT_STATS
264  std::cout << "Encoded " << mConstraints.size() - liaConstraints.size() << " as bool and "
265  << liaConstraints.size() << " as LIA. "
266  << ((double)(mConstraints.size() - liaConstraints.size()))/(double)mConstraints.size() * 100 << "% are directly encoded." << std::endl;
267  #endif
268 
269  for (const auto& subformula : rReceivedFormula()) {
270  SMTRAT_LOG_INFO("smtrat.pbc", "rFormula " << subformula.formula() << " forwarded as \t" << restoreFormula(subformula.formula()));
271  addSubformulaToPassedFormula(restoreFormula(subformula.formula()), subformula.formula());
272  }
273 
274  Answer ans = runBackends();
275  if (ans == UNSAT) {
276  generateTrivialInfeasibleSubset();
277  }
278 
279  return ans;
280  }

283  FormulaT PBPPModule<Settings>::restoreFormula(const FormulaT& formula) {
284  if (formula.getType() == carl::FormulaType::CONSTRAINT) {
285  // this one we may have to substitute
286  const ConstraintT& constraint = formula.constraint();
287  if (boolEncodings.find(constraint) != boolEncodings.end()) return boolEncodings[constraint];
288  if (liaConstraintFormula.find(constraint) != liaConstraintFormula.end()) return liaConstraintFormula[constraint];
289 
290  return formula;
291  } else if (formula.isBooleanCombination() && formula.isNary()) {
292  FormulasT subforms;
293  for (const auto& subformula : formula.subformulas()) {
294  subforms.push_back(restoreFormula(subformula));
295  }
296 
297  return FormulaT(formula.getType(), subforms);
298  } else {
299  // we don't care and return the formula - for example boolean literals. That's totally fine.
300  return formula;

307  template<typename Settings>
308  FormulaT PBPPModule<Settings>::forwardAsArithmetic(const ConstraintT& formula, const std::set<carl::Variable>& boolVariables){
309  carl::Variables variables = formula.variables().as_set();
310 
311  std::set<carl::Variable> variableSetIntersection;
312  std::set_intersection(variables.begin(), variables.end(),
313  boolVariables.begin(), boolVariables.end(),
314  std::inserter(variableSetIntersection, variableSetIntersection.end()));
315 
316  for(const auto& it : variables){
317  if (mVariablesCache.find(it) != mVariablesCache.end()) continue;
318 
319  // add the variable since there is no integer coupling just yet.
320  mVariablesCache.insert(std::pair<carl::Variable, carl::Variable>(it, carl::freshIntegerVariable()));
321  }
322 
323  Poly lhs;
324  for(const auto& it : formula.lhs()){
325  if (it.isConstant()) {
326  lhs += it.coeff();
327  continue;
328  }
329 
330  lhs = lhs + it.coeff() * mVariablesCache[it.getSingleVariable()];
331  }
332 
333  FormulaT constraintFormula = FormulaT(lhs, formula.relation());
334  FormulaT boolConnection = interconnectVariables(variableSetIntersection);
335  // it remains to specify bounds to on the new integer variables, however, it is enough
336  // to specify bounds to variables \setminus variableSetIntersection
337  FormulasT bounds;
338  for (auto var : variables) {
339  if (variableSetIntersection.find(var) != variableSetIntersection.end()) continue;
340 
341  // variable is not in intersection, add discrete bounds
342  ConstraintT equalOne((Poly(mVariablesCache[var]) - Rational(1)), carl::Relation::EQ);
343  ConstraintT equalZero(Poly(mVariablesCache[var]), carl::Relation::EQ);
344  bounds.push_back(FormulaT(carl::FormulaType::OR, FormulaT(equalOne), FormulaT(equalZero)));
345  }
346 
347  // bounds and connection do hold globally, forward them here
348  addSubformulaToPassedFormula(FormulaT(carl::FormulaType::AND, boolConnection, FormulaT(carl::FormulaType::AND, bounds)));
349 
350  return constraintFormula;
351  }

353  template<typename Settings>
354  FormulaT PBPPModule<Settings>::interconnectVariables(const std::set<carl::Variable>& variables){
355  FormulasT subformulas;
356  for(const auto& var : variables){
357  if(std::find(mConnectedVars.begin(), mConnectedVars.end(), var) == mConnectedVars.end()){
358  //The variables are not interconnected
359  mConnectedVars.push_back(var);
360  FormulaT boolVar = FormulaT(var);
361  Poly intVar(mVariablesCache.find(var)->second);
362  FormulaT subformulaA = FormulaT(intVar - Rational(1), carl::Relation::EQ);
363  FormulaT subformulaB = FormulaT(carl::FormulaType::IMPLIES, boolVar, subformulaA);
364  FormulaT subformulaC = FormulaT(intVar, carl::Relation::EQ);
365  FormulaT subformulaD = FormulaT(carl::FormulaType::IMPLIES, boolVar.negated(), subformulaC);
366  FormulaT newFormula = FormulaT(carl::FormulaType::AND, subformulaB, subformulaD);
367  subformulas.push_back(newFormula);
368 
369  }
370  }
371  return FormulaT(carl::FormulaType::AND, std::move(subformulas));
372  }

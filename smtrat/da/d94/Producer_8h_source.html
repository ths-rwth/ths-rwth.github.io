67  std::size_t operator()(Node& root) {
68  std::size_t skip = 0;

109  /**
110  * Iterate over nodes using DFS.
111  * @param root Root of nodes.
112  * @param n Current node.
113  * @param num Internal counter of node.
114  * @param skip Number of nodes to skip.
115  */
116  void dfs(const Node& root, const Node* n, std::size_t& num, std::size_t skip) {
117  if (consumer.hasResult()) return;
118  progress();
119  if (skip < num) process(root, *n, num);
120  num++;
121  for (const auto& child: n->children) {
122  if (child.immutable()) continue;
123  dfs(root, &child, num, skip);
124  }
125  }
126  /**
127  * Iterate over nodes using BFS.
128  * @param root Root of nodes.
129  * @param num Internal counter of node.
130  * @param skip Number of nodes to skip.
131  */
132  void bfs(const Node& root, std::size_t& num, std::size_t skip) {
133  std::queue<const Node*> q;
134  q.push(&root);
135  while (!q.empty()) {
136  if (consumer.hasResult()) return;
137  progress();
138  if (skip < num) process(root, *q.front(), num);
139  num++;
140  for (const auto& child: q.front()->children) {
141  if (child.immutable()) {
142  progress(child.complexity());
143  continue;
144  }
145  q.push(&child);
146  }
147  q.pop();
148  }
149  }

151  /**
152  * Performs the actual simplifications.
153  * @param root Root of the current smtlib file.
154  * @param n Node that is currently processed.
155  * @param num Current progress.
156  */
157  void process(const Node& root, const Node& n, std::size_t num) {
158  if (n.immutable()) return;
159  if (&root == &n) return;
160  if (delayDeclareFun) {
161  if (n.name == "declare-fun") return;
162  }
163  if (!settings.has("no-removal")) {
164  consumer.consume(root.clone(&n, nullptr), String() << "Removed \"" << n.repr(verbose) << "\"", num);
165  }
166  for (const auto& op: operators) {
167  auto changes = std::get<0>(op)(n);
168  for (const auto& c: changes) {
169  consumer.consume(root.clone(&n, &c), String() << std::get<1>(op) << "\"" << n.repr(verbose) << "\"" << std::get<2>(op) << "\"" << c.repr(verbose) << "\"" << std::get<3>(op), num);
170  }
171  }

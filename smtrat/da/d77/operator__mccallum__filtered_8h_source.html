63 static inline void delineate_properties(datastructures::SampledDerivation<PropertiesSet>& deriv) {
64  SMTRAT_LOG_FUNC("smtrat.cadcells.operators", &deriv);
65  for(const auto& prop : deriv.properties<properties::poly_irreducible_sgn_inv>()) {
66  rules::delineate(*deriv.delineated(), prop);
67  }
68  for(const auto& prop : deriv.properties<properties::poly_irreducible_semi_sgn_inv>()) {
69  rules::delineate(*deriv.delineated(), prop);
70  }
71  for(const auto& prop : deriv.properties<properties::root_ordering_holds>()) {
72  switch(Settings::delineation_function) {
73  case MccallumFilteredSettings::ALL:
74  rules::delineate_all<Settings>(deriv, prop, Settings::enable_weak);
75  break;
76  case MccallumFilteredSettings::BOUNDS_ONLY:
77  rules::delineate_bounds_only(deriv, prop);
78  break;
79  case MccallumFilteredSettings::COMPOUND:
80  rules::delineate_all_compound(deriv, prop, Settings::enable_weak, false);
81  break;
82  case MccallumFilteredSettings::COMPOUND_PWL:
83  rules::delineate_compound_piecewiselinear(deriv, prop, Settings::enable_weak);
84  break;
85  case MccallumFilteredSettings::BC:
86  rules::delineate_all_biggest_cell(deriv, prop, Settings::enable_weak);
87  break;
88  case MccallumFilteredSettings::NOOP:
89  default:
90  rules::delineate_noop(deriv, prop);
91  }

95 static inline bool project_cell_properties(datastructures::CellRepresentation<PropertiesSet>& repr) {
96  SMTRAT_LOG_FUNC("smtrat.cadcells.operators", repr);
97  auto& deriv = *repr.derivation;
98 
99  for (const auto poly : deriv.delin().nullified()) {
100  if (deriv.contains(properties::poly_del{ poly })) {
101  return false;
102  }
103  }
104 
105  for(const auto& poly : repr.description.polys()) {
106  deriv.insert(properties::poly_del{ poly });
107  }
108  for(const auto& poly : repr.ordering.polys()) {
109  deriv.insert(properties::poly_del{ poly });
110  }
111  properties::insert_root_ordering_holds(deriv, repr.ordering);
112 
113  for(const auto& prop : deriv.properties<properties::root_ordering_holds>()) {
114  if (!rules::root_ordering_holds_delineated(deriv, repr.description, repr.ordering, repr.ordering_polys, prop.ordering)) return false;
115  }
116 
117  if (deriv.contains(properties::cell_connected{deriv.level()})) {
118  rules::cell_connected(deriv, repr.description, repr.ordering);
119  }
120  rules::cell_analytic_submanifold(deriv, repr.description);
121  rules::cell_represents(deriv, repr.description);
122 
123  if (!repr.equational.empty()) {
124  deriv.insert(properties::poly_sgn_inv{ deriv.proj().ldcf(repr.description.section_defining().poly) });
125  }
126 
127  for(const auto& prop : deriv.properties<properties::poly_ord_inv_base>()) {
128  rules::poly_ord_inv_base(deriv, repr.description, repr.ordering, prop.poly);
129  }
130  for(const auto& prop : deriv.properties<properties::poly_irreducible_sgn_inv>()) {
131  if (repr.equational.find(prop.poly) != repr.equational.end()) {
132  rules::poly_irreducible_sgn_inv_ec(deriv, repr.description, prop.poly);
133  } else if (deriv.delin().nonzero().find(prop.poly) != deriv.delin().nonzero().end()) {
134  rules::poly_irreducible_nonzero_sgn_inv(*deriv.delineated(), prop.poly);
135  } else if (deriv.delin().nullified().find(prop.poly) != deriv.delin().nullified().end()) {
136  rules::poly_irreducible_null_sgn_inv(deriv, prop.poly);
137  } else {
138  rules::poly_irreducible_sgn_inv_filtered(deriv, repr.description, repr.ordering, prop.poly);
139  }
140  }
141  for(const auto& prop : deriv.properties<properties::poly_irreducible_semi_sgn_inv>()) {
142  if (repr.equational.find(prop.poly) != repr.equational.end()) {
143  rules::poly_irreducible_semi_sgn_inv_ec(deriv, repr.description, prop.poly);
144  } else if (deriv.delin().nonzero().find(prop.poly) != deriv.delin().nonzero().end()) {
145  rules::poly_irreducible_nonzero_semi_sgn_inv(*deriv.delineated(), prop.poly);
146  } else if (deriv.delin().nullified().find(prop.poly) != deriv.delin().nullified().end()) {
147  rules::poly_irreducible_null_semi_sgn_inv(deriv, prop.poly);
148  } else {
149  rules::poly_irreducible_semi_sgn_inv_filtered(deriv, repr.description, repr.ordering, prop.poly);
150  }
151  }
152  return true;
153 }

105  void merge(Value const& a, Value const& b) noexcept
106  {
107  auto ra = find(a);
108  auto rb = find(b);
109  if (ra == rb)
110  return;
111  if (_ranks[ra] < _ranks[rb])
112  std::swap(ra, rb);
113  _parents[rb] = ra;
114  if (_ranks[ra] == _ranks[rb])
115  ++_ranks[ra];
116  }

179  FindState find_impl(Parents parents, Value const& i) const noexcept
180  {

182  if (fi == i) {
183  return {parents, i};
184  } else {

198  PersistentUnionFind merge(Value const& a, Value const& b) noexcept
199  {
200  auto const& ra = find(a);
201  auto const& rb = find(b);
202  if (ra != rb) {
203  auto const& ranka = _ranks[ra];
204  auto const& rankb = _ranks[rb];
205 
206  PersistentUnionFind merged;
207  if (ranka > rankb) {
208  auto parents = _parents.set(rb, ra);
209  merged.update(std::move(parents), _ranks);
210  } else if(ranka < rankb) {
211  auto parents = _parents.set(ra, rb);
212  merged.update(std::move(parents), _ranks);
213  } else {
214  auto ranks = _ranks.set(ra, ranka + 1);
215  auto parents = _parents.set(rb, ra);
216  merged.update(std::move(parents), std::move(ranks));
217  }
218 
219  return merged;
220  }

72  FormulaT LVEModule<Settings>::eliminate_from_separated_weak_inequality(carl::Variable v, const Poly& with, const Poly& without, carl::Relation rel) {
73  auto res = lve::get_root(v, with);
74  if (res) {
75  SMTRAT_LOG_DEBUG("smtrat.lve", "Satisfying equality by selecting root " << v << " = " << *res);
76  mPPModel.assign(v, *res);
77  return FormulaT(carl::FormulaType::TRUE);
78  } else {
79  SMTRAT_LOG_DEBUG("smtrat.lve", "Factor " << with << " has no real root. Set " << v << " = 0 and eliminate factor.");
80  mPPModel.assign(v, 0);
81  carl::Sign sgn = lve::sgn_of_invariant_poly(v, with);
82  assert(sgn == carl::Sign::NEGATIVE || sgn == carl::Sign::POSITIVE);
83  if (sgn == carl::Sign::NEGATIVE) {
84  switch (rel) {
85  case carl::Relation::EQ: rel = carl::Relation::EQ; break;
86  case carl::Relation::LEQ: rel = carl::Relation::GEQ; break;
87  case carl::Relation::GEQ: rel = carl::Relation::LEQ; break;
88  default: assert(false);
89  }
90  SMTRAT_LOG_DEBUG("smtrat.lve", "-> " << FormulaT(without, rel));
91  return FormulaT(without, rel);
92  } else if (sgn == carl::Sign::POSITIVE) {
93  SMTRAT_LOG_DEBUG("smtrat.lve", "-> " << FormulaT(without, rel));
94  return FormulaT(without, rel);
95  } else {
96  assert(false);
97  return FormulaT();
98  }
99  }
100  }

122  template<class Settings>
123  FormulaT LVEModule<Settings>::eliminate_from_separated_strict_inequality(carl::Variable v, const Poly& with, const Poly& without, carl::Relation rel) {
124  if (without.isConstant()) {
125  SMTRAT_LOG_DEBUG("smtrat.lve", "Remaining part " << without << " is constant, we choose a proper value for " << v);
126  carl::Sign target = sign_of(rel, without.constantPart() < 0);
127  auto val = lve::get_value_for_sgn(v, with, target);
128  if (val) {
129  SMTRAT_LOG_DEBUG("smtrat.lve", "Found proper value " << v << " = " << *val << " such that " << with << " " << rel << " 0");
130  mPPModel.emplace(v, *val);
131  return FormulaT(carl::FormulaType::TRUE);
132  } else {
133  SMTRAT_LOG_DEBUG("smtrat.lve", "No value exists for " << v << " such that " << with << " " << rel << " 0");
134  return FormulaT(carl::FormulaType::FALSE);
135  }
136  } else {
137  SMTRAT_LOG_DEBUG("smtrat.lve", "Remaining part " << without << " is not constant, we construct a conditional model for " << v);
138  auto posval = lve::get_value_for_sgn(v, with, carl::Sign::POSITIVE);
139  auto negval = lve::get_value_for_sgn(v, with, carl::Sign::NEGATIVE);
140  assert(posval || negval);
141  if (!posval) {
142  SMTRAT_LOG_DEBUG("smtrat.lve", "Cannot make " << with << " positive.");
143  mPPModel.emplace(v, *negval);
144  switch (rel) {
145  case carl::Relation::LESS: return FormulaT(without, carl::Relation::GREATER);
146  case carl::Relation::GREATER: return FormulaT(without, carl::Relation::LESS);
147  default: assert(false);
148  }
149  return FormulaT();
150  }
151  if (!negval) {
152  SMTRAT_LOG_DEBUG("smtrat.lve", "Cannot make " << with << " negative.");
153  mPPModel.emplace(v, *posval);
154  return FormulaT(without, rel);
155  }
156  SMTRAT_LOG_DEBUG("smtrat.lve", "Possible values for " << v << " are " << *posval << " and " << *negval);
157  if (FormulaT(without, carl::Relation::LESS).isFalse()) {
158  SMTRAT_LOG_DEBUG("smtrat.lve", "Cannot make " << without << " negative.");
159  switch (rel) {
160  case carl::Relation::LESS: mPPModel.emplace(v, *negval); break;
161  case carl::Relation::GREATER: mPPModel.emplace(v, *posval); break;
162  default: assert(false);
163  }
164  return FormulaT(without, carl::Relation::GREATER);
165  }
166  if (FormulaT(without, carl::Relation::GREATER).isFalse()) {
167  SMTRAT_LOG_DEBUG("smtrat.lve", "Cannot make " << without << " positive.");
168  switch (rel) {
169  case carl::Relation::LESS: mPPModel.emplace(v, *posval); break;
170  case carl::Relation::GREATER: mPPModel.emplace(v, *negval); break;
171  default: assert(false);
172  }
173  return FormulaT(without, carl::Relation::LESS);
174  }
175  std::vector<std::pair<FormulaT, ModelValue>> subs;
176  switch (rel) {
177  case carl::Relation::LESS:
178  subs.emplace_back(FormulaT(without, carl::Relation::LESS), *posval);
179  subs.emplace_back(FormulaT(without, carl::Relation::GREATER), *negval);
180  break;
181  case carl::Relation::GREATER:
182  subs.emplace_back(FormulaT(without, carl::Relation::LESS), *negval);
183  subs.emplace_back(FormulaT(without, carl::Relation::GREATER), *posval);
184  break;
185  default: assert(false);
186  }
187  mPPModel.emplace(v, carl::createSubstitution<Rational,Poly,carl::ModelConditionalSubstitution<Rational,Poly>>(subs));
188  SMTRAT_LOG_DEBUG("smtrat.lve", "Constructed conditional model: " << mPPModel);
189  return FormulaT(without, carl::Relation::NEQ);
190  }

194  std::optional<FormulaT> LVEModule<Settings>::eliminate_from_separated_factors(carl::Variable v, const Poly& with, const Poly& without, carl::Relation rel) {
195  SMTRAT_LOG_DEBUG("smtrat.lve", "Considering " << v << " in " << with << " and " << without << " and relation " << rel);
196  if (carl::isWeak(rel)) {
197  return eliminate_from_separated_weak_inequality(v, with, without, rel);
198  } else if (rel == carl::Relation::NEQ) {
199  return eliminate_from_separated_disequality(v, with, without);
200  } else {
201  assert(carl::isStrict(rel));
202  return eliminate_from_separated_strict_inequality(v, with, without, rel);
203  }
204  return std::nullopt;
205  }

207  template<class Settings>
208  std::optional<FormulaT> LVEModule<Settings>::eliminate_from_factors(carl::Variable v, const ConstraintT& c) {
209  Poly with_v = Poly(1);
210  Poly without_v = Poly(1);
211  for (const auto& factor: carl::factorization(c.lhs())) {
212  carl::carlVariables vars = carl::variables(factor.first);
213  if (vars == carl::carlVariables({ v })) {
214  with_v *= carl::pow(factor.first, factor.second);
215  } else if (std::find(vars.begin(), vars.end(), v) == vars.end()) {
216  without_v *= carl::pow(factor.first, factor.second);
217  } else {
218  return std::nullopt;
219  }
220  }
221  assert(!isZero(with_v) && !isZero(without_v));
222  SMTRAT_LOG_DEBUG("smtrat.lve", "Separated " << c << " into " << with_v << " and " << without_v);
223  return eliminate_from_separated_factors(v, with_v, without_v, c.relation());
224  }

226  template<class Settings>
227  std::optional<FormulaT> LVEModule<Settings>::eliminate_linear(carl::Variable v, const ConstraintT& c) {
228  assert(c.maxDegree(v) > 0);
229  if (c.maxDegree(v) > 1) {
230  return std::nullopt;
231  }
232  // Decompose a * v - b ~ 0
233  carl::Relation rel = c.relation();
234  auto a = c.lhs().coeff(v, 1);
235  auto b = a*v - c.lhs();
236  // Now consider a * v ~ b
237  SMTRAT_LOG_DEBUG("smtrat.lve", "Considering " << a << " * " << v << " " << rel << " " << b);
238 
239  if (a.isConstant()) {
240  if (a.constantPart() < 0) {
241  switch (rel) {
242  case carl::Relation::EQ: rel = carl::Relation::EQ; break;
243  case carl::Relation::NEQ: rel = carl::Relation::NEQ; break;
244  case carl::Relation::LESS: rel = carl::Relation::GREATER; break;
245  case carl::Relation::LEQ: rel = carl::Relation::GEQ; break;
246  case carl::Relation::GREATER: rel = carl::Relation::LESS; break;
247  case carl::Relation::GEQ: rel = carl::Relation::LEQ; break;
248  }
249  }
250  b = b / a.constantPart();
251  // Now we have v ~ b
252  SMTRAT_LOG_DEBUG("smtrat.lve", "Transformed to " << v << " " << rel << " " << b);
253  switch (rel) {
254  case carl::Relation::EQ: break;
255  case carl::Relation::NEQ: b = b + Rational(1); break;
256  case carl::Relation::LESS: b = b - Rational(1); break;
257  case carl::Relation::LEQ: break;
258  case carl::Relation::GREATER: b = b + Rational(1); break;
259  case carl::Relation::GEQ: break;
260  }
261  SMTRAT_LOG_DEBUG("smtrat.lve", "Eliminated with " << v << " = " << b);
262  mPPModel.emplace(v, carl::createSubstitution<Rational,Poly,ModelPolynomialSubstitution>(b));
263  return FormulaT(carl::FormulaType::TRUE);
264  }
265 
266  return std::nullopt;
267  }

292  template<class Settings>
293  Answer LVEModule<Settings>::checkCore()
294  {
295  std::vector<carl::Variable> vars = get_lone_variables();
296  if (is_minimizing()) {
297  SMTRAT_LOG_DEBUG("smtrat.lve", "Ignore objective variable " << objective());
298  vars.erase(std::find(vars.begin(), vars.end(), objective()));
299  }
300 #ifdef SMTRAT_DEVOPTION_Statistics
301  mStatistics.lone_variables = std::max(mStatistics.lone_variables, vars.size());
302 #endif
303  for (const auto& f: rReceivedFormula()) {
304  if (f.formula().getType() == carl::FormulaType::CONSTRAINT) {
305  const auto& c = f.formula().constraint();
306  carl::Variable target = carl::Variable::NO_VARIABLE;
307  for (auto v: vars) {
308  if (c.variables().has(v)) {
309  target = v;
310  break;
311  }
312  }
313  if (target != carl::Variable::NO_VARIABLE) {
314  auto res = eliminate_variable(target, c);
315  if (res) {
316  SMTRAT_LOG_DEBUG("smtrat.lve", "Elimination: " << f.formula() << " -> " << *res);
317  addSubformulaToPassedFormula(*res, f.formula());
318  continue;
319  }
320  }
321  }
322  SMTRAT_LOG_DEBUG("smtrat.lve", "No elimination for " << f.formula());
323  addSubformulaToPassedFormula(f.formula(), f.formula());
324  }
325  auto res = runBackends();
326  if (res == UNSAT) getInfeasibleSubsets();
327  #ifdef SMTRAT_DEVOPTION_Statistics
328  mStatistics.eliminated = std::max(mStatistics.eliminated, mPPModel.size());
329  #endif
330  return res;
331  }
332 }

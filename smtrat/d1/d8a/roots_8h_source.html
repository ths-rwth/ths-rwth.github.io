134  /**
135  * Add a CellDescription to the covering.
136  *
137  * The added cell needs to be the rightmost cell of the already added cells and not be contained in any of these cells (or vice versa).
138  *
139  * * The first cell needs to have -oo as left bound.
140  * * The last cell needs to have oo as right bound.
141  * * All cells need to cover the real line under an appropriate sample.
142  * * Evaluated under an appropriate sample, the left bound of the added cell c is strictly greater than the left bounds of already added cells (considering also "strictness" of the bounds).
143  * * The right bound of the added cell c needs to be greater than all right bounds of already added cells (considering also "strictness" of the bounds).
144  */

146  assert(!m_data.empty() || (c.is_sector() && !c.lower()));
147  assert(m_data.empty() || c.is_section() || c.lower());
148  assert(m_data.empty() || m_data.back().is_section() || m_data.back().upper());
149  m_data.push_back(c);
150  }

171  assert(first.poly.level == second.poly.level);
172  if (first != second) {

200  bool leq_transitive(IndexedRoot first, IndexedRoot second) const {
201  boost::container::flat_set<IndexedRoot> reached({first});
202  std::vector<IndexedRoot> active({first});
203  if (first == second) return true;
204  while(!active.empty()) {
205  auto current = active.back();
206  active.pop_back();
207  if (m_leq.contains(current)) {
208  for (const auto& e : m_leq.at(current)) {
209  if (!reached.contains(e)) {
210  if (e == second) return true;
211  reached.insert(e);
212  active.push_back(e);
213  }
214  }
215  }
216  }

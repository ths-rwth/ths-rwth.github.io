45 NodeChangeSet reorderChildren(const Node& n) {
46  std::vector<Node> nodes = n.children;
47  NodeChangeSet res;
48  for (std::size_t i = 1; i < nodes.size(); ++i) {
49  if (nodes[i-1].size > nodes[i].size) {
50  res.emplace_back(n);
51  std::swap(res.back().children[i-1], res.back().children[i]);

59  * @param n Node.
60  * @return A set of replacements.

62 NodeChangeSet mergeChild(const Node& n) {
63  if (n.children.empty()) return NodeChangeSet();
64  if ((n.name == "and") || (n.name == "or") || (n.name == "+") || (n.name == "*")) {
65  NodeChangeSet res;
66  for (auto it = n.children.begin(); it != n.children.end();) {
67  std::vector<Node> newchildren;
68  newchildren.insert(newchildren.end(), n.children.begin(), it);
69  newchildren.insert(newchildren.end(), it->children.begin(), it->children.end());
70  it++;
71  if (it != n.children.end()) {
72  newchildren.insert(newchildren.end(), it, n.children.end());
73  }
74  res.emplace_back(std::make_tuple(n.name, newchildren, n.brackets));
75  }
76  return res;
77  }

81 /**
82  * Node operator that provides meaningful replacements for numbers.
83  * @param n Node.
84  * @return A set of replacements.
85  */
86 NodeChangeSet number(const Node& n) {
87  if (!n.children.empty()) return NodeChangeSet();
88  if (n.name == "") return NodeChangeSet();
89  if (n.brackets) return NodeChangeSet();
90 
91  NodeChangeSet res;
92 
93  // Not a number
94  if (!std::regex_match(n.name, std::regex("[0-9]*\\.?[0-9]*"))) return res;
95 
96  // Handle trailing dot -> remove dot
97  if (std::regex_match(n.name, std::regex("[0-9]+\\."))) {
98  res.emplace_back(n.name.substr(0, n.name.size()-1), false);
99  return res;
100  }
101  // Handle simple numbers -> remove last digits
102  if (std::regex_match(n.name, std::regex("[0-9]+"))) {
103  for (unsigned i = 1; i < n.name.size(); i++) {
104  res.emplace_back(n.name.substr(0, i), false);
105  }
106  return res;
107  }
108 
109  // Handle degenerated floating points -> add zero in front
110  if (std::regex_match(n.name, std::regex("\\.[0-9]+"))) {
111  res.emplace_back("0" + n.name, false);
112  return res;
113  }
114 
115  // Handle floating points -> remove decimal places
116  if (std::regex_match(n.name, std::regex("[0-9]+\\.[0-9]+"))) {
117  std::size_t pos = n.name.find('.');
118  res.emplace_back(n.name.substr(0, pos), false);
119  res.emplace_back(n.name.substr(0, n.name.size()-1));
120  return res;
121  }
122 
123  // Fallthrough
124  return res;
125 }

89  // Return all constraints that are reason for the derivation used in the full covering on level 0
90  inline FormulaSetT getInfeasibleSubset() {
91  assert(mCoveringInformation[0].isFullCovering());
92 
93  SMTRAT_LOG_DEBUG("smtrat.covering", "getInfeasibleSubset");
94  // Use Set to avoid duplicates
95  FormulaSetT infeasibleSubset;
96  // We can just take the constraints used in level 0, as all the constraints of higher levels get pushed down if used in the covering
97  for (auto& infeasibleConstraint : mCoveringInformation[0].getConstraintsOfCovering(mDerivationToConstraint)) {
98  infeasibleSubset.insert(FormulaT(infeasibleConstraint));
99  }
100  SMTRAT_LOG_DEBUG("smtrat.covering", "getInfeasibleSubset done: " << infeasibleSubset);
101  return infeasibleSubset;

163  // Return true if the constraint to remove was used in the current covering
164  void removeConstraint(const ConstraintT& constraint) {
165  // We can substract 1 from level because we dont have constant polynomials
166  std::size_t level = helper::level_of(mVariableOrdering, constraint.lhs()) - 1;
167  SMTRAT_LOG_DEBUG("smtrat.covering", "Removing Constraint : " << constraint << " on level " << level);
168  SMTRAT_LOG_DEBUG("smtrat.covering", "Known Constraints: " << mKnownConstraints);
169  SMTRAT_LOG_DEBUG("smtrat.covering", "Unknown Constraints: " << mUnknownConstraints);
170 
171  // If we find the constraint in the unknown constraints we have the easy case -> Just remove it and not care about the stored data
172  if (mUnknownConstraints[level].find(constraint) != mUnknownConstraints[level].end()) {
173  assert(mKnownConstraints[level].find(constraint) == mKnownConstraints[level].end());
174  // remove all stored information which resulted from the constraint
175  for (std::size_t cur_level = 0; cur_level <= level; cur_level++) {
176  SMTRAT_LOG_DEBUG("smtrat.covering", "Removing on level " << cur_level);
177  mCoveringInformation[cur_level].removeConstraint(constraint, mDerivationToConstraint);
178  }
179  mUnknownConstraints[level].erase(constraint);
180  SMTRAT_LOG_DEBUG("smtrat.covering", "Constraint to remove was in unknown constraints");
181  return;
182  }
183 
184  if (mKnownConstraints[level].find(constraint) == mKnownConstraints[level].end()) {
185  SMTRAT_LOG_DEBUG("smtrat.covering", "Constraint to remove was not in known constraints");
186  // This can happen if the constraint is to be added in the same iteration
187  // TODO: what to do then?
188  return;
189  }
190 
191  // The constraint must be in the known constraints
192  SMTRAT_LOG_DEBUG("smtrat.covering", "Constraint to remove was in known constraints");
193  // remove all stored information which resulted from the constraint
194  for (std::size_t cur_level = 0; cur_level <= level; cur_level++) {
195  SMTRAT_LOG_DEBUG("smtrat.covering", "Removing on level " << cur_level);
196  mCoveringInformation[cur_level].removeConstraint(constraint, mDerivationToConstraint);
197  }
198 
199  // remove the constraint from the known constraints
200  mKnownConstraints[level].erase(constraint);
201  }

220  void processUnknownConstraints(const std::size_t& level, const bool prune_assignment) {
221  SMTRAT_LOG_DEBUG("smtrat.covering", "Processing unknown constraints on level " << level << " with " << mUnknownConstraints[level].size() << " constraints");
222  SMTRAT_LOG_DEBUG("smtrat.covering", "Need to prune the assignment: " << prune_assignment);
223 
224 
225  for (const ConstraintT& constraint : mUnknownConstraints[level]) {
226  std::vector<datastructures::SampledDerivationRef<PropSet>> intervals ;
227  if (prune_assignment) {
228  intervals = algorithms::get_unsat_intervals<op>(constraint, *mProjections, mCurrentAssignment);
229  } else {
230  // create copy of the assignment with mVariableOrdering[level] and following not present
231  carl::ran_assignment<Rational> assignment;
232  for (std::size_t i = 0; i < level; i++) {
233  assignment[mVariableOrdering[i]] = mCurrentAssignment[mVariableOrdering[i]];
234  }
235  intervals = algorithms::get_unsat_intervals<op>(constraint, *mProjections, assignment);
236  }
237  for (const auto& interval : intervals) {
238  SMTRAT_LOG_DEBUG("smtrat.covering", "Found UNSAT Interval: " << interval->cell() << " from constraint: " << constraint);
239  // Insert into the derivation to constraint map
240  mDerivationToConstraint.insert(std::make_pair(interval, std::vector<ConstraintT>{constraint}));
241  mCoveringInformation[level].addDerivation(std::move(interval));
242  }
243  }
244 
245  // Set the unknown constraints to be known, as all new derivations have been calculated
246  setConstraintsKnown(level);
247  }

249  Answer getUnsatCover(const std::size_t level) {
250  SMTRAT_LOG_DEBUG("smtrat.covering", " getUnsatCover for level: " << level << " / " << dimension());
251  SMTRAT_LOG_DEBUG("smtrat.covering", " Variable Ordering: " << mVariableOrdering);
252  SMTRAT_LOG_DEBUG("smtrat.covering", " Unknown Constraints: " << mUnknownConstraints[level]);
253  SMTRAT_LOG_DEBUG("smtrat.covering", " Known Constraints: " << mKnownConstraints[level]);
254  SMTRAT_LOG_DEBUG("smtrat.covering", " Current Covering Information: " << mCoveringInformation[level]);
255  SMTRAT_LOG_DEBUG("smtrat.covering", " Current Assignment: " << mCurrentAssignment);
256 
257  // incase of incremental solving, we need to check if the current level is already assigned
258  // if it is assigned, the current assignment is satisfying all unknown constraints and we dont need to process the unknown constraints
259  // If we know that the assignment is still satisfying we also dont need to recalculate the (partial!) covering.
260  if (mCurrentAssignment.find(mVariableOrdering[level]) == mCurrentAssignment.end()) {
261  processUnknownConstraints(level, false);
262  SMTRAT_LOG_DEBUG("smtrat-covering", "Computing Covering represenentation")
263  bool invalidates = mCoveringInformation[level].computeCovering();
264  if (invalidates) {
265  SMTRAT_LOG_DEBUG("smtrat.covering", "Computed Covering invalidates all higher levels as the underlying sample point changed");
266  resetStoredData(level + 1);
267  }
268  } else {
269  SMTRAT_LOG_DEBUG("smtrat.covering", "Current variable " << mVariableOrdering[level] << " is assigned, skipping processing of new constraints and computing covering representation");
270  // Possible in the case that SAT was reported before and constraints were removed and invalidated the current partial covering
271  if (mCoveringInformation[level].isUnknownCovering()) {
272  SMTRAT_LOG_DEBUG("smtrat.covering", "Covering was invalidated, recomputing covering representation and processing all unknown constraints");
273  processUnknownConstraints(level, true);
274  bool invalidates = mCoveringInformation[level].computeCovering();
275  if (invalidates) {
276  SMTRAT_LOG_DEBUG("smtrat.covering", "Computed Covering invalidates all higher levels as the underlying sample point changed");
277  resetStoredData(level + 1);
278  }
279  assert(mCoveringInformation[level].isPartialCovering());
280  }
281  }
282 
283  SMTRAT_LOG_DEBUG("smtrat.covering", "Got CoveringStatus: " << mCoveringInformation[level].getCoveringStatus());
284  if (mCoveringInformation[level].isFailedCovering()) {
285  SMTRAT_LOG_DEBUG("smtrat.covering", "Covering failed -> Abort");
286  return Answer::UNKNOWN;
287  }
288 
289  while (mCoveringInformation[level].isPartialCovering()) {
290  SMTRAT_LOG_DEBUG("smtrat.covering", "Sample outside " << mCoveringInformation[level].getSampleOutside());
291  updateAssignment(level);
292  if (mCurrentAssignment.size() == mVariableOrdering.size()) {
293  // SAT
294  SMTRAT_LOG_DEBUG("smtrat.covering", "Found satisfying variable assignment: " << mCurrentAssignment);
295  return Answer::SAT;
296  }
297 
298  Answer nextLevelAnswer = getUnsatCover(level + 1);
299  if (nextLevelAnswer == Answer::SAT) {
300  SMTRAT_LOG_DEBUG("smtrat.covering", "Got SAT on level: " << level);
301  // Push down SAT
302  return nextLevelAnswer;
303  } else if (nextLevelAnswer == Answer::UNSAT) {
304 
305  auto new_derivation = mCoveringInformation[level + 1].constructDerivation(mDerivationToConstraint);
306 
307  if (!new_derivation.has_value()) {
308  SMTRAT_LOG_DEBUG("smtrat.covering", "No new derivation found -> Abort");
309  return Answer::UNKNOWN;
310  }
311 
312  SMTRAT_LOG_DEBUG("smtrat.covering", "Found new derivation: " << new_derivation.value()->cell());
313  SMTRAT_LOG_DEBUG("smtrat.covering", "Adding new derivation to Covering Information");
314  mCoveringInformation[level].addDerivation(std::move(new_derivation.value()));
315 
316  // delete the now obsolete data
317  mCurrentAssignment.erase(mVariableOrdering[level]);
318  mCoveringInformation[level + 1].clear();
319  setConstraintsUnknown(level + 1);
320 
321  // If there are unknown constraints on this level, we need to process them now
322  processUnknownConstraints(level, false);
323 
324  // Recalculate the current covering
325  SMTRAT_LOG_DEBUG("smtrat.covering", "Computing covering representation");
326  bool invalidates = mCoveringInformation[level].computeCovering();
327  if (invalidates) {
328  SMTRAT_LOG_DEBUG("smtrat.covering", "Computed Covering invalidates all higher levels as the underlying sample point changed");
329  resetStoredData(level + 1);
330  }
331  SMTRAT_LOG_DEBUG("smtrat.covering", "Got CoveringStatus: " << mCoveringInformation[level].getCoveringStatus());
332  if (mCoveringInformation[level].isFailedCovering()) {
333  SMTRAT_LOG_DEBUG("smtrat.covering", "Covering failed -> Abort");
334  return Answer::UNKNOWN;
335  }
336 
337  } else {
338  // Something went wrong (McCallum failed)
339  return Answer::UNKNOWN;
340  }
341  }
342 
343  assert(mCoveringInformation[level].isFullCovering());
344  SMTRAT_LOG_DEBUG("smtrat.covering", "Cells cover the numberline ");
345 
346  return Answer::UNSAT;
347  }
